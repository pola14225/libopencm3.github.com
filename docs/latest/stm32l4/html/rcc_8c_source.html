178 void rcc_wait_for_sysclk_status(enum rcc_osc osc)
179 {
180  switch (osc) {
181  case RCC_PLL:
182  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK)
183  != RCC_CFGR_SWS_PLL);
184  break;
185  case RCC_HSE:
186  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK)
187  != RCC_CFGR_SWS_HSE);
188  break;
189  case RCC_HSI16:
190  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK)
191  != RCC_CFGR_SWS_HSI16);
192  break;
193  case RCC_MSI:
194  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK)
195  != RCC_CFGR_SWS_MSI);
196  break;
197  default:
198  /* Shouldn't be reached. */
199  break;
200  }
201 }

359 /**
360  * Set the msi run time range.
361  * Can only be called when MSI is either OFF, or when MSI is on _and_
362  * ready. (RCC_CR_MSIRDY bit). @sa rcc_set_msi_range_standby
363  * @param msi_range range number @ref rcc_cr_msirange
364  */
365 void rcc_set_msi_range(uint32_t msi_range)
366 {
367  uint32_t reg = RCC_CR;
368  reg &= ~(RCC_CR_MSIRANGE_MASK << RCC_CR_MSIRANGE_SHIFT);
369  reg |= msi_range << RCC_CR_MSIRANGE_SHIFT;
370  RCC_CR = reg | RCC_CR_MSIRGSEL;
371 }

373 /**

376  * Note that not all MSI range values are allowed here!

378  * @param msi_range range number valid for post standby @ref rcc_csr_msirange

380 void rcc_set_msi_range_standby(uint32_t msi_range)
