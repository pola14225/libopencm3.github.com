70 /** @brief Set the Number of Wait States
71 
72 Used to match the system clock to the FLASH memory access time. See the
73 programming manual for more information on clock speed ranges. The latency must
74 be changed to the appropriate value <b>before</b> any increase in clock
75 speed, or <b>after</b> any decrease in clock speed.
76 
77 @param[in] ws values from @ref flash_latency.
78 */
79 void flash_set_ws(uint32_t ws)
80 {
81  uint32_t reg32;
82 
83  reg32 = FLASH_ACR;
84  reg32 &= ~(FLASH_ACR_LATENCY_MASK << FLASH_ACR_LATENCY_SHIFT);
85  reg32 |= (ws << FLASH_ACR_LATENCY_SHIFT);
86  FLASH_ACR = reg32;
87 }

245 /** @brief Program a 32 bit Word to FLASH
246  * This performs all operations necessary to program a 32 bit word to FLASH
247  * memory. The program error flag should be checked separately for the event
248  * that memory was not properly erased.
249  *
250  * @param[in] address Starting address in Flash.
251  * @param[in] data word to write
252  */
253 void flash_program_word(uint32_t address, uint32_t data)
254 {
255  /* Ensure that all flash operations are complete. */
256  flash_wait_for_last_operation();
257 
258  /* Enable writes to flash. */
259  FLASH_CR |= FLASH_CR_PG;
260 
261  /* Program the word. */
262  MMIO32(address) = data;
263 
264  /* Wait for the write to complete. */
265  flash_wait_for_last_operation();
266 
267  /* Disable writes to flash. */
268  FLASH_CR &= ~FLASH_CR_PG;
269 }

271 /** @brief Program a Data Block to FLASH
272  * This programs an arbitrary length data block to FLASH memory.
273  * The program error flag should be checked separately for the event that
274  * memory was not properly erased.
275  * @param[in] address Starting address in Flash.
276  * @param[in] data Pointer to start of data block.
277  * @param[in] len Length of data block.
278  */
279 void flash_program(uint32_t address, uint8_t *data, uint32_t len)
280 {
281  /* TODO: Use dword and word size program operations where possible for
282  * turbo speed.
283  */
284  uint32_t i;
285  for (i = 0; i < len; i++) {
286  flash_program_word(address+i, data[i]);
287  }
288 }

290 /** @brief Erase a Sector of FLASH
291  * This performs all operations necessary to erase a sector in FLASH memory.
292  * The page should be checked to ensure that it was properly erased. A sector
293  * must first be fully erased before attempting to program it.
294  * See the reference manual or the FLASH programming manual for details.
295  * @param[in] sector (0 - 11 for some parts, 0-23 on others)
296  */
297 void flash_erase_sector(uint8_t sector)
298 {
299  flash_wait_for_last_operation();
300 
301  FLASH_CR &= ~(FLASH_CR_PNB_MASK << FLASH_CR_PNB_SHIFT);
302  FLASH_CR |= (sector & FLASH_CR_PNB_MASK) << FLASH_CR_PNB_SHIFT;
303  FLASH_CR |= FLASH_CR_PER;
304  FLASH_CR |= FLASH_CR_START;
305 
306  flash_wait_for_last_operation();
307  FLASH_CR &= ~FLASH_CR_PER;
308  FLASH_CR &= ~(FLASH_CR_PNB_MASK << FLASH_CR_PNB_SHIFT);
309 }

311 /** @brief Erase All FLASH
312  * This performs all operations necessary to erase all sectors in the FLASH
313  * memory.
314  */
315 void flash_erase_all_sectors(void)
316 {
317  flash_wait_for_last_operation();
318 
319  FLASH_CR |= FLASH_CR_MER1 | FLASH_CR_MER2;
320  FLASH_CR |= FLASH_CR_START;
321 
322  flash_wait_for_last_operation();
323  FLASH_CR &= ~FLASH_CR_MER1 & ~FLASH_CR_MER2;
324 }

326 /** @brief Program the Option Bytes
327  * This performs all operations necessary to program the option bytes.
328  * The option bytes do not need to be erased first.
329  * @param[in] data value to be programmed.
330  */
331 void flash_program_option_bytes(uint32_t data)
332 {
333  flash_wait_for_last_operation();
334 
335  if (FLASH_CR & FLASH_CR_OPTLOCK) {
336  flash_unlock_option_bytes();
337  }
338 
339  FLASH_OPTR = data & ~0x3;
340  FLASH_OPTR |= FLASH_CR_OPTSTRT;
341  flash_wait_for_last_operation();
342 }
