52 /*
53  * bit[N] == 0, trace port width of (N+1) not supported
54  * bit[N] == 1, trace port width of (N+1) supported
55  */
56 #define TPIU_SSPSR_BYTE (1 << 0)

64 #define TPIU_CSPSR_HALFWORD (1 << 1)
65 #define TPIU_CSPSR_WORD (1 << 3)

67 /* --- TPIU_ACPR values ---------------------------------------------------- */

69 /* Bits 31:16 - Reserved */
