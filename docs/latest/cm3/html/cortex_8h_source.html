110 /** @brief Cortex M Mask interrupts
111  *
112  * This function switches the mask of the interrupts. If mask is true, the
113  * interrupts will be disabled. The result of this function can be used for
114  * restoring previous state of the mask.
115  *
116  * @param[in] mask bool New state of the interrupt mask
117  * @returns bool old state of the interrupt mask
118  */
119 __attribute__((always_inline))
120 static inline bool cm_mask_interrupts(bool mask)
121 {
122  register bool old;
123  __asm__ __volatile__("MRS %0, PRIMASK" : "=r" (old));
124  __asm__ __volatile__("" : : : "memory");
125  __asm__ __volatile__("MSR PRIMASK, %0" : : "r" (mask));
126  return old;
127 }

176 /** @brief Cortex M Atomic Declare block
177  *
178  * This macro disables interrupts for the next command or block of code. The
179  * interrupt mask is automatically restored after exit of the boundary of the
180  * code block. Therefore restore of interrupt is done automatically after call
181  * of return or goto control sentence jumping outside of the block.
182  *
183  * @warning The usage of sentences break or continue is prohibited in the block
184  * due to implementation of this macro!
185  *
186  * @note It is safe to use this block inside normal code and in interrupt
187  * routine.
188  *
189  * @example 1: Basic usage of atomic block
190  *
191  * @code
192  * uint64_t value; // This value is used somewhere in interrupt
193  *
194  * ...
195  *
196  * CM_ATOMIC_BLOCK() { // interrupts are masked in this block
197  * value = value * 1024 + 651; // access value as atomic
198  * } // interrupts is restored automatically
199  * @endcode
200  *
201  * @example 2: Use of return inside block:
202  *
203  * @code
204  * uint64_t value; // This value is used somewhere in interrupt
205  *
206  * ...
207  *
208  * uint64_t allocval(void)
209  * {
210  * CM_ATOMIC_BLOCK() { // interrupts are masked in this block
211  * value = value * 1024 + 651; // do long atomic operation
212  * return value; // interrupts is restored automatically
213  * }
214  * }
215  * @endcode
216  */
217 #if defined(__DOXYGEN__)
218 #define CM_ATOMIC_BLOCK()
219 #else /* defined(__DOXYGEN__) */
220 #define CM_ATOMIC_BLOCK() \
221  for (bool __CM_SAVER(true), __my = true; __my; __my = false)
222 #endif /* defined(__DOXYGEN__) */

225 /** @brief Cortex M Atomic Declare context
226  *
227  * This macro disables interrupts in the current block of code from the place
228  * where it is defined to the end of the block. The interrupt mask is
229  * automatically restored after exit of the boundary of the code block.
230  * Therefore restore of interrupt is done automatically after call of return,
231  * continue, break, or goto control sentence jumping outside of the block.
232  *
233  * @note This function is intended for use in for- cycles to enable the use of
234  * break and contine sentences inside the block, and for securing the atomic
235  * reader-like functions.
236  *
237  * @note It is safe to use this block inside normal code and in interrupt
238  * routine.
239  *
240  * @example 1: Basic usage of atomic context
241  *
242  * @code
243  * uint64_t value; // This value is used somewhere in interrupt
244  *
245  * ...
246  *
247  * for (int i=0;i < 100; i++) {
248  * CM_ATOMIC_CONTEXT(); // interrupts are masked in this block
249  * value += 100; // access value as atomic
250  * if ((value % 16) == 0) {
251  * break; // restore interrupts and break cycle
252  * }
253  * } // interrupts is restored automatically
254  * @endcode
255  *
256  * @example 2: Usage of atomic context inside atomic reader fcn.
257  *
258  * @code
259  * uint64_t value; // This value is used somewhere in interrupt
260  *
261  * ...
262  *
263  * uint64_t getnextval(void)
264  * {
265  * CM_ATOMIC_CONTEXT(); // interrupts are masked in this block
266  * value = value + 3; // do long atomic operation
267  * return value; // interrupts is restored automatically
268  * }
269  * @endcode
270  */
