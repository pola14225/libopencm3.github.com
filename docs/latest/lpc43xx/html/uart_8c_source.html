44 /*
45 * UART Init function
46 */
47 void uart_init(uart_num_t uart_num, uart_databit_t data_nb_bits,
48  uart_stopbit_t data_nb_stop, uart_parity_t data_parity,
49  uint16_t uart_divisor, uint8_t uart_divaddval, uint8_t uart_mulval)
50 {
51  uint32_t lcr_config;
52  uint32_t uart_port;
53 
54  uart_port = uart_num;
55 
56  switch (uart_num) {
57  case UART0_NUM:
58  /* use PLL1 as clock source for UART0 */
59  CGU_BASE_UART0_CLK = (1<<UART_CGU_AUTOBLOCK_CLOCK_BIT) |
60  (CGU_SRC_PLL1<<UART_CGU_BASE_CLK_SEL_SHIFT);
61  break;
62 
63  case UART1_NUM:
64  /* use PLL1 as clock source for UART1 */
65  CGU_BASE_UART1_CLK = (1<<UART_CGU_AUTOBLOCK_CLOCK_BIT) |
66  (CGU_SRC_PLL1<<UART_CGU_BASE_CLK_SEL_SHIFT);
67  break;
68 
69  case UART2_NUM:
70  /* use PLL1 as clock source for UART2 */
71  CGU_BASE_UART2_CLK = (1<<UART_CGU_AUTOBLOCK_CLOCK_BIT) |
72  (CGU_SRC_PLL1<<UART_CGU_BASE_CLK_SEL_SHIFT);
73  break;
74 
75  case UART3_NUM:
76  /* use PLL1 as clock source for UART3 */
77  CGU_BASE_UART3_CLK = (1<<UART_CGU_AUTOBLOCK_CLOCK_BIT) |
78  (CGU_SRC_PLL1<<UART_CGU_BASE_CLK_SEL_SHIFT);
79  break;
80 
81  default:
82  return; /* error */
83  }
84 
85  /* FIFOs RX/TX Enabled and Reset RX/TX FIFO (DMA Mode is also cleared)*/
86  UART_FCR(uart_port) = (UART_FCR_FIFO_EN | UART_FCR_RX_RS |
87  UART_FCR_TX_RS);
88  /* Disable FIFO */
89  UART_FCR(uart_port) = 0;
90 
91  /* Dummy read (to clear existing data) */
92  while (UART_LSR(uart_port) & UART_LSR_RDR) {
93  dummy_read = UART_RBR(uart_port);
94  }
95 
96  /* Wait end of TX & disable TX */
97  UART_TER(uart_port) = UART_TER_TXEN;
98 
99  /* Wait for current transmit complete */
100  while (!(UART_LSR(uart_port) & UART_LSR_THRE));
101 
102  /* Disable Tx */
103  UART_TER(uart_port) = 0;
104 
105  /* Disable interrupt */
106  UART_IER(uart_port) = 0;
107 
108  /* Set LCR to default state */
109  UART_LCR(uart_port) = 0;
110 
111  /* Set ACR to default state */
112  UART_ACR(uart_port) = 0;
113 
114  /* Dummy Read to Clear Status */
115  dummy_read = UART_LSR(uart_port);
116 
117  /*
118  Table 835. USART Fractional Divider Register:
119  UARTbaudrate = PCLK / ( 16* (((256*DLM)+ DLL)*(1+(DivAddVal/MulVal))) )
120  The value of MULVAL and DIVADDVAL should comply to the following
121  conditions:
122  1. 1 <= MULVAL <= 15
123  2. 0 <= DIVADDVAL <= 14
124  3. DIVADDVAL < MULVAL
125  */
126 
127  /* Set DLAB Bit */
128  UART_LCR(uart_port) |= UART_LCR_DLAB_EN;
129  UART_DLM(uart_port) = UART_LOAD_DLM(uart_divisor);
130  UART_DLL(uart_port) = UART_LOAD_DLL(uart_divisor);
131  /* Clear DLAB Bit */
132  UART_LCR(uart_port) &= (~UART_LCR_DLAB_EN) & UART_LCR_BITMASK;
133  UART_FDR(uart_port) = UART_FDR_BITMASK &
134  (UART_FDR_MULVAL(uart_mulval) | UART_FDR_DIVADDVAL(uart_divaddval));
135 
136  /* Read LCR config & Force Enable of Divisor Latches Access */
137  lcr_config = (UART_LCR(uart_port) & UART_LCR_DLAB_EN) &
138  UART_LCR_BITMASK;
139  lcr_config |= data_nb_bits; /* Set Nb Data Bits */
140  lcr_config |= data_nb_stop; /* Set Nb Stop Bits */
141  lcr_config |= data_parity; /* Set Data Parity */
142 
143  /* Write LCR (only 8bits) */
144  UART_LCR(uart_port) = (lcr_config & UART_LCR_BITMASK);
145 
146  /* Enable TX */
147  UART_TER(uart_port) = UART_TER_TXEN;
148 }

150 /*
151 * This Function return if data are received or not received.
152 */
153 uart_rx_data_ready_t uart_rx_data_ready(uart_num_t uart_num)
154 {
155  uint32_t uart_port;
156  uint8_t uart_status;
157  uart_rx_data_ready_t data_ready;
158 
159  uart_port = uart_num;
160 
161  uart_status = UART_LSR(uart_port) & 0xFF;
162 
163  /* Check Error */
164  if ((uart_status & UART_LSR_ERROR_MASK) == 0) {
165  /* No errors check if data is ready */
166  if ((uart_status & UART_LSR_RDR) == 0) {
167  data_ready = UART_RX_NO_DATA;
168  } else {
169  data_ready = UART_RX_DATA_READY;
170  }
171  } else {
172  /* UART Error */
173  data_ready = UART_RX_DATA_ERROR;
174  }
175 
176  return data_ready;
177 }

179 /*
180 * This Function Wait until Data RX Ready, and return Data Read from UART.
181 */
182 uint8_t uart_read(uart_num_t uart_num)
183 {
184  uint32_t uart_port;
185  uint8_t uart_val;
186 
187  uart_port = uart_num;
188 
189  /* Wait Until Data Received (Rx Data Not Ready) */
190  while ((UART_LSR(uart_port) & UART_LSR_RDR) == 0);
191 
192  uart_val = (UART_RBR(uart_port) & UART_RBR_MASKBIT);
193 
194  return uart_val;
195 }

197 /*
198 * This Function Wait until Data RX Ready, and return Data Read from UART.
199 */
200 uint8_t uart_read_timeout(uart_num_t uart_num, uint32_t rx_timeout_nb_cycles,
201  uart_error_t *error)
202 {
203  uint32_t uart_port;
204  uint8_t uart_val;
205  uint32_t counter;
206 
207  uart_port = uart_num;
208 
209  /* Wait Until Data Received (Rx Data Not Ready) */
210  counter = 0;
211  while ((UART_LSR(uart_port) & UART_LSR_RDR) == 0) {
212  if (rx_timeout_nb_cycles > 0) {
213  counter++;
214  if (counter >= rx_timeout_nb_cycles) {
215  *error = UART_TIMEOUT_ERROR;
216  return 0;
217  }
218  }
219  }
220 
221  uart_val = (UART_RBR(uart_port) & UART_RBR_MASKBIT);
222 
223  /* Clear error */
224  *error = UART_NO_ERROR;
225 
226  return uart_val;
227 }

237 
238  /* Wait Until FIFO not full */
