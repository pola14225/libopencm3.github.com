44 void i2c0_init(const uint16_t duty_cycle_count)
45 {
46  /* enable input on SCL and SDA pins */
47  SCU_SFSI2C0 = SCU_I2C0_NOMINAL;
48 
49  I2C0_SCLH = duty_cycle_count;
50  I2C0_SCLL = duty_cycle_count;
51 
52  /* clear the control bits */
53  I2C0_CONCLR = (I2C_CONCLR_AAC | I2C_CONCLR_SIC
54  | I2C_CONCLR_STAC | I2C_CONCLR_I2ENC);
55 
56  /* enable I2C0 */
57  I2C0_CONSET = I2C_CONSET_I2EN;
58 }

60 /* transmit start bit */
61 void i2c0_tx_start(void)
62 {
63  I2C0_CONCLR = I2C_CONCLR_SIC;
64  I2C0_CONSET = I2C_CONSET_STA;
65  while (!(I2C0_CONSET & I2C_CONSET_SI));
66  I2C0_CONCLR = I2C_CONCLR_STAC;
67 }

69 /* transmit data byte */
70 void i2c0_tx_byte(uint8_t byte)
71 {
72  if (I2C0_CONSET & I2C_CONSET_STA) {
73  I2C0_CONCLR = I2C_CONCLR_STAC;
74  }
75  I2C0_DAT = byte;
76  I2C0_CONCLR = I2C_CONCLR_SIC;
77  while (!(I2C0_CONSET & I2C_CONSET_SI));
78 }
