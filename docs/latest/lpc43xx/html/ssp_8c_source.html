62 */
63 void ssp_init(ssp_num_t ssp_num,
64  ssp_datasize_t data_size,
65  ssp_frame_format_t frame_format,
66  ssp_cpol_cpha_t cpol_cpha_format,
67  u8 serial_clock_rate,
68  u8 clk_prescale,
69  ssp_mode_t mode,
70  ssp_master_slave_t master_slave,
71  ssp_slave_option_t slave_option)
72 {
73  u32 ssp_port;
74  u32 clock;
75 
76  if(ssp_num == SSP0_NUM)
77  {
78  ssp_port = SSP0;
79  }else
80  {
81  ssp_port = SSP1;
82  }
83 
84  /* use PLL1 as clock source for SSP1 */
85  CGU_BASE_SSP1_CLK = (CGU_SRC_PLL1<<CGU_BASE_CLK_SEL_SHIFT) | (1<<CGU_AUTOBLOCK_CLOCK_BIT);
86 
87  /* Disable SSP before to configure it */
88  SSP_CR1(ssp_port) = 0x0;
89 
90  /* Configure SSP */
91  clock = serial_clock_rate;
92  SSP_CPSR(ssp_port) = clk_prescale;
93  SSP_CR0(ssp_port) = (data_size | frame_format | cpol_cpha_format | (clock<<8) );
94 
95  /* Enable SSP */
96  SSP_CR1(ssp_port) = (SSP_ENABLE | mode | master_slave | slave_option);
97 }

102 u16 ssp_read(ssp_num_t ssp_num)
103 {
104  u32 ssp_port;
105 
106  if(ssp_num == SSP0_NUM)
107  {
108  ssp_port = SSP0;

110  {
111  ssp_port = SSP1;

134 /* This Function Wait Data TX Ready, and Write Data to SSP */
135 void ssp_write(ssp_num_t ssp_num, u16 data)
136 {
137  u32 ssp_port;
138 
139  if(ssp_num == SSP0_NUM)
140  {
141  ssp_port = SSP0;
142  }else
143  {
144  ssp_port = SSP1;
145  }
146 
147  /* Wait Until FIFO not full */
148  while( (SSP_SR(ssp_port) & SSP_SR_TNF) == 0);
149 
150  SSP_DR(ssp_port) = data;
151 
152  /* Wait for not busy, since we're controlling CS# of
153  * devices manually and need to wait for the data to
154  * be sent. It may also be important to wait here
155  * in case we're configuring devices via SPI and also
156  * with GPIO control -- we need to know when SPI
157  * commands are effective before altering a device's
158  * state with GPIO. I'm thinking the MAX2837, for
159  * example...
160  */
161  ssp_wait_until_not_busy(ssp_num);
162 }
