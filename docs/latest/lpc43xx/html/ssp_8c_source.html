99  /* use PLL1 as clock source for SSP1 */
100  CGU_BASE_SSP1_CLK = (CGU_SRC_PLL1<<CGU_BASE_CLK_SEL_SHIFT) | (1<<CGU_AUTOBLOCK_CLOCK_BIT);
101 
102  /* Disable SSP before to configure it */
103  SSP_CR1(ssp_port) = 0x0;
104 
105  /* Configure SSP */
106  clock = serial_clock_rate;
107  SSP_CPSR(ssp_port) = clk_prescale;
108  SSP_CR0(ssp_port) = (data_size | frame_format | cpol_cpha_format | (clock<<8) );
109 
110  /* Enable SSP */
111  SSP_CR1(ssp_port) = (SSP_ENABLE | mode | master_slave | slave_option);

149 /* This Function Wait Data TX Ready, and Write Data to SSP */
150 void ssp_write(ssp_num_t ssp_num, u16 data)
151 {
152  u32 ssp_port;
153 
154  if(ssp_num == SSP0_NUM)
155  {
156  ssp_port = SSP0;
157  }else
158  {
159  ssp_port = SSP1;
160  }
161 
162  /* Wait Until FIFO not full */
163  while( (SSP_SR(ssp_port) & SSP_SR_TNF) == 0);
164 
165  SSP_DR(ssp_port) = data;
166 
167  /* Wait for not busy, since we're controlling CS# of
168  * devices manually and need to wait for the data to
169  * be sent. It may also be important to wait here
170  * in case we're configuring devices via SPI and also
171  * with GPIO control -- we need to know when SPI
172  * commands are effective before altering a device's
173  * state with GPIO. I'm thinking the MAX2837, for
174  * example...
175  */
176  ssp_wait_until_not_busy(ssp_num);
