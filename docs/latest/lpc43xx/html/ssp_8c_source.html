105 /* This Function Wait Data TX Ready, and Write Data to SSP */
106 uint16_t ssp_transfer(ssp_num_t ssp_num, uint16_t data)
107 {
108  uint32_t ssp_port;
109 
110  if (ssp_num == SSP0_NUM) {
111  ssp_port = SSP0;
112  } else {
113  ssp_port = SSP1;
114  }
115 
116  /* Wait Until FIFO not full */
117  while ((SSP_SR(ssp_port) & SSP_SR_TNF) == 0);
118 
119  SSP_DR(ssp_port) = data;
120 
121  /* Wait for not busy, since we're controlling CS# of
122  * devices manually and need to wait for the data to
123  * be sent. It may also be important to wait here
124  * in case we're configuring devices via SPI and also
125  * with GPIO control -- we need to know when SPI
126  * commands are effective before altering a device's
127  * state with GPIO. I'm thinking the MAX2837, for
128  * example...
129  */
130  ssp_wait_until_not_busy(ssp_num);
131 
132  /* Wait Until Data Received (Rx FIFO not Empty) */
133  while ((SSP_SR(ssp_port) & SSP_SR_RNE) == 0);
134 
135  return SSP_DR(ssp_port);
136 }
