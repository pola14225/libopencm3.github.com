24 /* Set M0 in reset mode */
25 void ipc_halt_m0(void)
26 {
27  volatile uint32_t rst_active_status1;
28 
29  /* Check if M0 is reset by reading status */
30  rst_active_status1 = RESET_ACTIVE_STATUS1;
31 
32  /* If the M0 has reset not asserted, halt it... */
33  while (rst_active_status1 & RESET_CTRL1_M0APP_RST) {
34  RESET_CTRL1 = ((~rst_active_status1) | RESET_CTRL1_M0APP_RST);
35  rst_active_status1 = RESET_ACTIVE_STATUS1;
36  }
37 }

40 {
41  volatile uint32_t rst_active_status1;
42 
43  /* Set M0 memory mapping to point to start of M0 image */
44  CREG_M0APPMEMMAP = cm0_baseaddr;
45 
46  /* Start/run M0 core */
47 
48  /* Release Slave from reset, first read status */
49  rst_active_status1 = RESET_ACTIVE_STATUS1;
50 
51  /* If the M0 is being held in reset, release it */
52  /* 1 = no reset, 0 = reset */
53  while (!(rst_active_status1 & RESET_CTRL1_M0APP_RST)) {
54  RESET_CTRL1 = ((~rst_active_status1) & ~RESET_CTRL1_M0APP_RST);
55  rst_active_status1 = RESET_ACTIVE_STATUS1;
56  }
