37 /** @brief ADC Set the Sample Time for a Single Channel
38  *
39  * The sampling time can be selected in ADC clock cycles, exact values
40  * depend on the device.
41  *
42  * @param[in] adc ADC block register address base @ref adc_reg_base
43  * @param[in] channel ADC Channel integer @ref adc_channel
44  * @param[in] time Sampling time selection from @ref adc_sample
45  */
46 void adc_set_sample_time(uint32_t adc, uint8_t channel, uint8_t time)
47 {
48  uint32_t reg32;
49 
50  if (channel < 10) {
51  reg32 = ADC_SMPR1(adc);
52  reg32 &= ~(0x7 << (channel * 3));
53  reg32 |= (time << (channel * 3));
54  ADC_SMPR1(adc) = reg32;
55  } else {
56  reg32 = ADC_SMPR2(adc);
57  reg32 &= ~(0x7 << ((channel - 10) * 3));
58  reg32 |= (time << ((channel - 10) * 3));
59  ADC_SMPR2(adc) = reg32;
60  }
61 }

63 /** @brief ADC Set the Sample Time for All Channels
64  *
65  * The sampling time can be selected in ADC clock cycles, exact values
66  * depend on the device.
67  *
68  * @param[in] adc ADC block register address base @ref adc_reg_base
69  * @param[in] time Sampling time selection from @ref adc_sample
70  */
71 void adc_set_sample_time_on_all_channels(uint32_t adc, uint8_t time)
72 {
73  uint8_t i;
74  uint32_t reg32 = 0;
75 
76  for (i = 0; i <= 9; i++) {
77  reg32 |= (time << (i * 3));
78  }
79  ADC_SMPR1(adc) = reg32;
80 
81  for (i = 10; i <= 17; i++) {
82  reg32 |= (time << ((i - 10) * 3));
83  }
84  ADC_SMPR2(adc) = reg32;
85 }

88 /** @brief ADC Set a Regular Channel Conversion Sequence
89  *
90  * Define a sequence of channels to be converted as a regular group with a
91  * length from 1 to 16 channels. If this is called during conversion, the
92  * current conversion is reset and conversion begins again with the newly
93  * defined group.
94  *
95  * @param[in] adc ADC block register address base @ref adc_reg_base
96  * @param[in] length Number of channels in the group, range 0..16
97  * @param[in] channel Set of channels in sequence, range @ref adc_channel
98  */
99 void adc_set_regular_sequence(uint32_t adc, uint8_t length, uint8_t channel[])
100 {
101  uint32_t reg32_1 = 0, reg32_2 = 0, reg32_3 = 0, reg32_4 = 0;
102  uint8_t i = 0;
103 
104  /* Maximum sequence length is 16 channels. */
105  if (length > 16) {
106  return;
107  }
108 
109  for (i = 1; i <= length; i++) {
110  if (i <= 4) {
111  reg32_1 |= (channel[i - 1] << (i * 6));
112  }
113  if ((i > 4) & (i <= 9)) {
114  reg32_2 |= (channel[i - 1] << ((i - 4 - 1) * 6));
115  }
116  if ((i > 9) & (i <= 14)) {
117  reg32_3 |= (channel[i - 1] << ((i - 9 - 1) * 6));
118  }
119  if ((i > 14) & (i <= 16)) {
120  reg32_4 |= (channel[i - 1] << ((i - 14 - 1) * 6));
121  }
122  }
123  reg32_1 |= ((length - 1) << ADC_SQR1_L_SHIFT);
124 
125  ADC_SQR1(adc) = reg32_1;
126  ADC_SQR2(adc) = reg32_2;
127  ADC_SQR3(adc) = reg32_3;
128  ADC_SQR4(adc) = reg32_4;
129 }
