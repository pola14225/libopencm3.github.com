29 /** @brief Enable the FLASH Prefetch Buffer
30 
31 This buffer is used for instruction fetches and is enabled by default after
32 reset.
33 
34 Note carefully the clock restrictions under which the prefetch buffer may be
35 enabled or disabled. Changes are normally made while the clock is running in
36 the power-on low frequency mode before being set to a higher speed mode.
37 See the reference manual for details.
38 */

57 /** @brief Set the Number of Wait States
58

185 /*---------------------------------------------------------------------------*/
186 /** @brief Erase All Option Bytes
187 
188 This performs all operations necessary to erase the option bytes. These must
189 first be fully erased before attempting to program them, therefore it is
190 recommended to check these after an erase attempt.
191 */
192 
193 void flash_erase_option_bytes(void)
194 {
195  flash_wait_for_last_operation();
196 
197  if ((FLASH_CR & FLASH_CR_OPTWRE) == 0) {
198  flash_unlock_option_bytes();
199  }
200 
201  FLASH_CR |= FLASH_CR_OPTER; /* Enable option byte erase. */
202  FLASH_CR |= FLASH_CR_STRT;
203  flash_wait_for_last_operation();
204  FLASH_CR &= ~FLASH_CR_OPTER; /* Disable option byte erase. */
205 }
206 
207 /*---------------------------------------------------------------------------*/
208 /** @brief Program the Option Bytes
209 
210 This performs all operations necessary to program the option bytes.
211 The write protect error flag should be checked separately for the event that
212 an option byte had not been properly erased before calling this function.
213 
214 Only the lower 8 bits of the data is significant.
215 
216 @param[in] address Address of option byte from @ref flash_options.
217 @param[in] data value to write
218 */
219 
220 void flash_program_option_bytes(uint32_t address, uint16_t data)
221 {
222  flash_wait_for_last_operation();
223 
224  if ((FLASH_CR & FLASH_CR_OPTWRE) == 0) {
225  flash_unlock_option_bytes();
226  }
227 
228  FLASH_CR |= FLASH_CR_OPTPG; /* Enable option byte programming. */
229  MMIO16(address) = data;
230  flash_wait_for_last_operation();
231  FLASH_CR &= ~FLASH_CR_OPTPG; /* Disable option byte programming. */
232 }
