377 void rcc_set_pll_configuration(uint32_t source, uint32_t multiplier,
378  uint32_t divisor)

381

384  reg32 &= ~(RCC_CFGR_PLLMUL_MASK << RCC_CFGR_PLLMUL_SHIFT);

452

454  reg &= ~(RCC_ICSCR_MSIRANGE_MASK << RCC_ICSCR_MSIRANGE_SHIFT);
455  reg |= (clock->msi_range << RCC_ICSCR_MSIRANGE_SHIFT);
456  RCC_ICSCR = reg;
457

459  rcc_wait_for_osc_ready(MSI);
460 
461  /* Select MSI as SYSCLK source. */
462  rcc_set_sysclk_source(RCC_CFGR_SW_SYSCLKSEL_MSICLK);
463 
464  /*

466  * Do this before touching the PLL (TODO: why?).
467  */
468  rcc_set_hpre(clock->hpre);
469  rcc_set_ppre1(clock->ppre1);
470  rcc_set_ppre2(clock->ppre2);
471

473  pwr_set_vos_scale(clock->voltage_scale);
474
