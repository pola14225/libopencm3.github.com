179 /** @brief Write a word to eeprom
180  *
181  * @param address assumed to be in the eeprom space, no checking
182  * @param data word to write
183  */
184 void eeprom_program_word(uint32_t address, uint32_t data)
185 {
186  flash_unlock_pecr();
187  /* erase only if needed */
188  FLASH_PECR &= ~FLASH_PECR_FTDW;
189  MMIO32(address) = data;
190  flash_lock_pecr();
191 }

193 /** @brief Write a block of words to eeprom
194  *
195  * Writes a block of words to EEPROM at the requested address, erasing if
196  * necessary, and locking afterwards. Only wordwise writing is safe for
197  * writing any value
198  *
199  * @param[in] address must point to EEPROM space, no checking!
200  * @param[in] data pointer to data to write
201  * @param[in] length_in_words size of of data in WORDS!
202  */
203 void eeprom_program_words(uint32_t address, uint32_t *data, int length_in_words)
204 {
205  int i;
206  flash_unlock_pecr();
207  while (FLASH_SR & FLASH_SR_BSY);
208  /* erase only if needed */
209  FLASH_PECR &= ~FLASH_PECR_FTDW;
210  for (i = 0; i < length_in_words; i++) {
211  MMIO32(address + (i * sizeof(uint32_t))) = *(data+i);
212  while (FLASH_SR & FLASH_SR_BSY);
213  }
214  flash_lock_pecr();
215 }
