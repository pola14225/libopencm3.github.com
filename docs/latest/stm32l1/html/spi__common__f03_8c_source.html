92 int spi_init_master(uint32_t spi, uint32_t br, uint32_t cpol, uint32_t cpha,
93  uint32_t crcl, uint32_t lsbfirst)
94 {
95  uint32_t reg32 = SPI_CR1(spi);
96 
97  /* Reset all bits omitting SPE, CRCEN and CRCNEXT bits. */
98  reg32 &= SPI_CR1_SPE | SPI_CR1_CRCEN | SPI_CR1_CRCNEXT;
99 
100  reg32 |= SPI_CR1_MSTR; /* Configure SPI as master. */
101 
102  reg32 |= br; /* Set baud rate bits. */
103  reg32 |= cpol; /* Set CPOL value. */
104  reg32 |= cpha; /* Set CPHA value. */
105  reg32 |= crcl; /* Set crc length (8 or 16 bits). */
106  reg32 |= lsbfirst; /* Set frame format (LSB- or MSB-first). */
107 
108  /* TODO: NSS pin handling. */
109 
110  SPI_CR1(spi) = reg32;
111 
112  return 0; /* TODO */
113 }

115 void spi_send8(uint32_t spi, uint8_t data)
116 {
117  /* Wait for transfer finished. */
118  while (!(SPI_SR(spi) & SPI_SR_TXE));
