37 
38 If the ADC is in power-down mode then it is powered up. The application needs

54 /** @brief ADC Set the Sample Time for a Single Channel
55 
56 The sampling time can be selected in ADC clock cycles from 4 to 384.
57 
58 @param[in] adc Unsigned int32. ADC block base address @ref adc_reg_base.
59 @param[in] channel uint8. ADC Channel integer 0..18 or from @ref adc_channel.
60 @param[in] time Unsigned int8. Sampling time selection from @ref adc_sample_rg.
61  */
62 void adc_set_sample_time(uint32_t adc, uint8_t channel, uint8_t time)
63 {
64  uint32_t reg32;
65 
66  if (channel < 10) {
67  reg32 = ADC_SMPR3(adc);
68  reg32 &= ~(0x7 << (channel * 3));
69  reg32 |= (time << (channel * 3));
70  ADC_SMPR3(adc) = reg32;
71  } else if (channel < 20) {
72  reg32 = ADC_SMPR2(adc);
73  reg32 &= ~(0x7 << ((channel - 10) * 3));
74  reg32 |= (time << ((channel - 10) * 3));
75  ADC_SMPR2(adc) = reg32;
76  } else {
77  reg32 = ADC_SMPR1(adc);
78  reg32 &= ~(0x7 << ((channel - 20) * 3));
79  reg32 |= (time << ((channel - 20) * 3));
80  ADC_SMPR1(adc) = reg32;
81  }
82 }

85 /** @brief ADC Set the Sample Time for All Channels
86 
87 The sampling time can be selected in ADC clock cycles, same for
88 all channels.
89 
90 @param[in] adc Unsigned int32. ADC block base address @ref adc_reg_base.
91 @param[in] time Unsigned int8. Sampling time selection from @ref adc_sample_rg.
92 */
93 
94 void adc_set_sample_time_on_all_channels(uint32_t adc, uint8_t time)
95 {
96  uint8_t i;
97  uint32_t reg32 = 0;
98 
99  for (i = 0; i <= 9; i++) {
100  reg32 |= (time << (i * 3));
101  }
102  ADC_SMPR0(adc) = reg32;
103  ADC_SMPR1(adc) = reg32;
104  ADC_SMPR2(adc) = reg32;
105  ADC_SMPR3(adc) = reg32;
106 }
