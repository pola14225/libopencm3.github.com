86 /** @brief Program a Half Word to FLASH
87 
88 This performs all operations necessary to program a 16 bit word to FLASH memory.
89 The program error flag should be checked separately for the event that memory
90 was not properly erased.
91 
92 Status bit polling is used to detect end of operation.
93 
94 @param[in] uint32_t address. Full address of flash half word to be programmed.
95 @param[in] uint16_t data.
96 */
97 
98 void flash_program_half_word(uint32_t address, uint16_t data)
99 {
100  flash_wait_for_last_operation();
101 
102  FLASH_CR |= FLASH_CR_PG;
103 
104  MMIO16(address) = data;
105 
106  flash_wait_for_last_operation();
107 
108  FLASH_CR &= ~FLASH_CR_PG;
109 }

112 /** @brief Erase a Page of FLASH
113 
114 This performs all operations necessary to erase a page in FLASH memory.
115 The page should be checked to ensure that it was properly erased. A page must
116 first be fully erased before attempting to program it.
117 
118 Note that the page sizes differ between devices. See the reference manual or
119 the FLASH programming manual for details.
120 
121 @param[in] uint32_t page_address. Full address of flash page to be erased.
122 */
123 
124 void flash_erase_page(uint32_t page_address)
125 {
126  flash_wait_for_last_operation();
127 
128  FLASH_CR |= FLASH_CR_PER;
129  FLASH_AR = page_address;
130  FLASH_CR |= FLASH_CR_STRT;
131 
132  flash_wait_for_last_operation();
133 
134  FLASH_CR &= ~FLASH_CR_PER;
135 }

137 /*---------------------------------------------------------------------------*/
138 /** @brief Erase All FLASH
139 
140 This performs all operations necessary to erase all user pages in the FLASH
141 memory. The information block is unaffected.
142 */
143 
144 void flash_erase_all_pages(void)
145 {
146  flash_wait_for_last_operation();
147 
148  FLASH_CR |= FLASH_CR_MER; /* Enable mass erase. */
149  FLASH_CR |= FLASH_CR_STRT; /* Trigger the erase. */
150 
151  flash_wait_for_last_operation();
152  FLASH_CR &= ~FLASH_CR_MER; /* Disable mass erase. */
153 
154 }
