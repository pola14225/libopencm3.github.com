70 /** ADC Set operation mode
71  *
72  * There are some operation modes, common for entire stm32 branch. In the text
73  * the braces are describing result to single trigger event. The trigger event
74  * is described by character T in the description. The ADC is configured to
75  * convert list of inputs [0, 1, 2, 3]. In Grouped modes, there is used group
76  * size of 2 conversions in the examples
77  *
78  * @li @c ADC_MODE_SEQUENTIAL: T(0) T(1) T(2) T(3)[EOSEQ] T(0) T(1) T(2) ...
79  *
80  * In this mode, after the trigger event a single channel is converted and the
81  * next channel in the list is prepared to convert on next trigger edge.
82  *
83  * @note This mode can be emulated by ADC_MODE_GROUPED with group size
84  * of 1. @par
85  *
86  * @li @c ADC_MODE_SCAN: T(0123)[EOSEQ] T(0123)[EOSEQ] T(0123)[EOSEQ]
87  *
88  * In this mode, after the trigger event, all channels will be converted once,
89  * storing results sequentially.
90  *
91  * @note The DMA must be configured properly for more than single channel to
92  * convert. @par
93  *
94  * @li @c ADC_MODE_SCAN_INFINITE: T(0123[EOSEQ]0123[EOSEQ]0123[EOSEQ]...)
95  *
96  * In this mode, after the trigger event, all channels from the list are
97  * converted. At the end of list, the conversion continues from the beginning.
98  *
99  * @note The DMA must be configured properly to operate in this mode.@par
100  *
101  * @li @c ADC_MODE_GROUPED: T(12) T(34)[EOSEQ] T(12) T(34)[EOSEQ] T(12)
102  *
103  * In this mode, after the trigger event, a specified group size of channels
104  * are converted. If the end of channel list occurs, the EOSEQ is generated
105  * and on the next trigger it wraps to the beginning.
106  *
107  * @note The DMA must be configured properly to operate on more than single
108  * channel conversion groups.@par
109  *
110  * @warning not all families supports all modes of operation of ADC.
111  *
112  * @par
113  *
114  */
115 
116 /*---------------------------------------------------------------------------*/
117 /** @brief ADC Set conversion operation mode
118  *
119  * @note on SEQUENTIAL mode, the trigger event is necessary to start conversion.
120  * @par
121  *
122  * @param[in] adc Unsigned int32. ADC base address (@ref adc_reg_base)
123  * @param[in] adc ::adc_opmode. ADC operation mode (@ref adc_opmode)
124  */
125 
126 void adc_set_operation_mode(uint32_t adc, enum adc_opmode opmode)
127 {
128  switch (opmode) {
129  case ADC_MODE_SEQUENTIAL:
130  ADC_CFGR1(adc) &= ~ADC_CFGR1_CONT;
131  ADC_CFGR1(adc) |= ADC_CFGR1_DISCEN;
132  break;
133 
134  case ADC_MODE_SCAN:
135  ADC_CFGR1(adc) &= ~(ADC_CFGR1_CONT | ADC_CFGR1_DISCEN);
136  break;
137 
138  case ADC_MODE_SCAN_INFINITE:
139  ADC_CFGR1(adc) &= ~ADC_CFGR1_DISCEN;
140  ADC_CFGR1(adc) |= ADC_CFGR1_CONT;
141  break;
142  }
143 }

311 /** @brief ADC Set a Regular Channel Conversion Sequence
312  *
313  * Define a sequence of channels to be converted as a regular group with a
314  * length from 1 to 18 channels. If this is called during conversion, the
315  * current conversion is reset and conversion begins again with the newly
316  * defined group.
317  *
318  * @warning This core doesn't support the random order of ADC conversions.
319  * The channel list must be ordered by channel number.
320  *
321  * @param[in] adc Unsigned int32. ADC base address (@ref adc_reg_base)
322  * @param[in] length Unsigned int8. Number of channels in the group.
323  * @param[in] channel Unsigned int8[]. Set of channels to convert, integers
324  * 0..18.
325  */
326 
327 void adc_set_regular_sequence(uint32_t adc, uint8_t length, uint8_t channel[])
328 {
329  uint32_t reg32 = 0;
330  uint8_t i = 0;
331  bool stepup = false, stepdn = false;
332 
333  if (length == 0) {
334  ADC_CHSELR(adc) = 0;
335  return;
336  }
337 
338  reg32 |= (1 << channel[0]);
339 
340  for (i = 1; i < length; i++) {
341  reg32 |= (1 << channel[i]);
342  stepup |= channel[i-1] < channel[i];
343  stepdn |= channel[i-1] > channel[i];
344  }
345 
346  /* Check, if the channel list is in order */
347  if (stepup && stepdn) {
348  cm3_assert_not_reached();
349  }
350 
351  /* Update the scan direction flag */
352  if (stepdn) {
353  ADC_CFGR1(adc) |= ADC_CFGR1_SCANDIR;
354  } else {
355  ADC_CFGR1(adc) &= ~ADC_CFGR1_SCANDIR;
356  }
357 
358  ADC_CHSELR(adc) = reg32;
359 }

426  * @defgroup adc_api_wdg ADC Analog watchdog API
427  * @ingroup adc_file
428  *
429  * @brief ADC analog watchdog API definitions.
430  *
431  * The analog watchdog allows the monitoring of an analog signal between two
432  * threshold levels. The thresholds must be preset. Analog watchdog is disabled
433  * by default.
434  *
435  * @warning Comparison is done before data alignment takes place, so the
436  * thresholds are left-aligned.
437  *
438  * Example 1: Enable watchdog checking on all channels
439  *
440  * @code
441  * // in configuration
442  * adc_enable_analog_watchdog_on_all_channels(ADC1);
443  * adc_set_watchdog_high_threshold(ADC1, 0xE00);
444  * adc_set_watchdog_low_threshold(ADC1, 0x200);
445  *
446  * // in the main application thread
447  * if (adc_get_watchdog_flag(ADC1)) {
448  * // the converted signal is out of AWD ranges
449  * adc_clear_watchdog_flag(ADC1);
450  * }
451  * @endcode
452  *
453  * Example 2: Enable watchdog checking on channel 5
454  *
455  * @code
456  * // in configuration
457  * adc_enable_analog_watchdog_on_selected_channel(ADC1,5);
458  * adc_set_watchdog_high_threshold(ADC1, 0xE00);
459  * adc_set_watchdog_low_threshold(ADC1, 0x200);
460  *
461  * // in the main application thread
462  * if (adc_get_watchdog_flag(ADC1)) {
