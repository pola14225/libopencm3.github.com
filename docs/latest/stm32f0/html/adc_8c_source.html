99 /** ADC Set operation mode
100  *
101  * There are some operation modes, common for entire stm32 branch. In the text
102  * the braces are describing result to single trigger event. The trigger event
103  * is described by character T in the description. The ADC is configured to
104  * convert list of inputs [0, 1, 2, 3]. In Grouped modes, there is used group
105  * size of 2 conversions in the examples
106  *
107  * @li @c ADC_MODE_SEQUENTIAL: T(0) T(1) T(2) T(3)[EOSEQ] T(0) T(1) T(2) ...
108  *
109  * In this mode, after the trigger event a single channel is converted and the
110  * next channel in the list is prepared to convert on next trigger edge.
111  *
112  * @note This mode can be emulated by ADC_MODE_GROUPED with group size
113  * of 1. @par
114  *
115  * @li @c ADC_MODE_SCAN: T(0123)[EOSEQ] T(0123)[EOSEQ] T(0123)[EOSEQ]
116  *
117  * In this mode, after the trigger event, all channels will be converted once,
118  * storing results sequentially.
119  *
120  * @note The DMA must be configured properly for more than single channel to
121  * convert. @par
122  *
123  * @li @c ADC_MODE_SCAN_INFINITE: T(0123[EOSEQ]0123[EOSEQ]0123[EOSEQ]...)
124  *
125  * In this mode, after the trigger event, all channels from the list are
126  * converted. At the end of list, the conversion continues from the beginning.
127  *
128  * @note The DMA must be configured properly to operate in this mode.@par
129  *
130  * @li @c ADC_MODE_GROUPED: T(12) T(34)[EOSEQ] T(12) T(34)[EOSEQ] T(12)
131  *
132  * In this mode, after the trigger event, a specified group size of channels
133  * are converted. If the end of channel list occurs, the EOSEQ is generated
134  * and on the next trigger it wraps to the beginning.
135  *
136  * @note The DMA must be configured properly to operate on more than single
137  * channel conversion groups.@par
138  *
139  * @warning not all families supports all modes of operation of ADC.
140  *
141  * @par
142  *
143  */
144 
145 /*---------------------------------------------------------------------------*/
146 /** @brief ADC Set conversion operation mode
147  *
148  * @note on SEQUENTIAL mode, the trigger event is neccesary to start conversion.
149  * @par
150  *
151  * @param[in] adc Unsigned int32. ADC base address (@ref adc_reg_base)
152  * @param[in] adc ::adc_opmode. ADC operation mode (@ref adc_opmode)
153  */
154 
155 void adc_set_operation_mode(uint32_t adc, enum adc_opmode opmode)
156 {
157  switch (opmode) {
158  case ADC_MODE_SEQUENTIAL:
159  ADC_CFGR1(adc) &= ~ADC_CFGR1_CONT;
160  ADC_CFGR1(adc) |= ADC_CFGR1_DISCEN;
161  break;
162 
163  case ADC_MODE_SCAN:
164  ADC_CFGR1(adc) &= ~(ADC_CFGR1_CONT | ADC_CFGR1_DISCEN);
165  break;
166 
167  case ADC_MODE_SCAN_INFINITE:
168  ADC_CFGR1(adc) &= ~ADC_CFGR1_DISCEN;
169  ADC_CFGR1(adc) |= ADC_CFGR1_CONT;
170  break;
171  }
172 }

507 /** @brief ADC Set a Regular Channel Conversion Sequence
508  *
509  * Define a sequence of channels to be converted as a regular group with a
510  * length from 1 to 18 channels. If this is called during conversion, the
511  * current conversion is reset and conversion begins again with the newly
512  * defined group.
513  *
514  * @warning This core doesn't support the random order of ADC conversions.
515  * The channel list must be ordered by channel number.
516  *
517  * @param[in] adc Unsigned int32. ADC base address (@ref adc_reg_base)
518  * @param[in] length Unsigned int8. Number of channels in the group.
519  * @param[in] channel Unsigned int8[]. Set of channels to convert, integers
520  * 0..18.
521  */
522 
523 void adc_set_regular_sequence(uint32_t adc, uint8_t length, uint8_t channel[])
524 {
525  uint32_t reg32 = 0;
526  uint8_t i = 0;
527  bool stepup = false, stepdn = false;
528 
529  if (length == 0) {
530  ADC_CHSELR(adc) = 0;
531  return;
532  }
533 
534  reg32 |= (1 << channel[0]);
535 
536  for (i = 1; i < length; i++) {
537  reg32 |= (1 << channel[i]);
538  stepup |= channel[i-1] < channel[i];
539  stepdn |= channel[i-1] > channel[i];
540  }
541 
542  /* Check, if the channel list is in order */
543  if (stepup && stepdn) {
544  cm3_assert_not_reached();
545  }
546 
547  /* Update the scan direction flag */
548  if (stepdn) {
549  ADC_CFGR1(adc) |= ADC_CFGR1_SCANDIR;
550  } else {
551  ADC_CFGR1(adc) &= ~ADC_CFGR1_SCANDIR;
552  }
553 
554  ADC_CHSELR(adc) = reg32;
555 }

726 /**
727  * @defgroup adc_api_wdg ADC Analog watchdog API
728  * @ingroup adc_file
729  *
730  * @brief ADC analog watchdog API definitions.
731  *
732  * The analog watchdog allows the monitoring of an analog signal between two
733  * threshold levels. The thresholds must be preset. Analog watchdog is disabled
734  * by default.
735  *
736  * @warning Comparison is done before data alignment takes place, so the
737  * thresholds are left-aligned.
738  *
739  * Example 1: Enable watchdog checking on all channels
740  *
741  * @code
742  * // in configuration
743  * adc_enable_analog_watchdog_on_all_channels(ADC1);
744  * adc_set_watchdog_high_threshold(ADC1, 0xE00);
745  * adc_set_watchdog_low_threshold(ADC1, 0x200);
746  *
747  * // in the main application thread
748  * if (adc_get_watchdog_flag(ADC1)) {
749  * // the converted signal is out of AWD ranges
750  * adc_clear_watchdog_flag(ADC1);
751  * }
752  * @endcode
753  *
754  * Example 2: Enable watchdog checking on channel 5
755  *
756  * @code
757  * // in configuration
758  * adc_enable_analog_watchdog_on_selected_channel(ADC1,5);
759  * adc_set_watchdog_high_threshold(ADC1, 0xE00);
760  * adc_set_watchdog_low_threshold(ADC1, 0x200);
761  *
762  * // in the main application thread
763  * if (adc_get_watchdog_flag(ADC1)) {
764  * // the converted signal is out of AWD ranges
765  * adc_clear_watchdog_flag(ADC1);
766  * }
767  * @endcode
768  *@{*/
