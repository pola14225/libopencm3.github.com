223 /** @brief RCC Turn on an Oscillator.
224  *
225  * Enable an oscillator and power on. Each oscillator requires an amount of
226  * time to settle to a usable state. Refer to datasheets for time delay
227  * information. A status flag is available to indicate when the oscillator
228  * becomes ready (see @ref rcc_osc_ready_int_flag and @ref
229  * rcc_wait_for_osc_ready).
230  *
231  * @param[in] osc enum ::osc_t. Oscillator ID
232  */
233 
234 void rcc_osc_on(enum rcc_osc osc)
235 {
236  switch (osc) {
237  case HSI14:
238  RCC_CR2 |= RCC_CR2_HSI14ON;
239  break;
240  case HSI:
241  RCC_CR |= RCC_CR_HSION;
242  break;
243  case HSE:
244  RCC_CR |= RCC_CR_HSEON;
245  break;
246  case LSE:
247  RCC_BDCR |= RCC_BDCR_LSEON;
248  break;
249  case LSI:
250  RCC_CSR |= RCC_CSR_LSION;
251  break;
252  case PLL:
253  RCC_CR |= RCC_CR_PLLON;
254  break;
255  }
256 }

259 /** @brief RCC Turn off an Oscillator.
260  *
261  * Disable an oscillator and power off.
262  *
263  * @note An oscillator cannot be turned off if it is selected as the system
264  * clock.
265  *
266  * @param[in] osc enum ::osc_t. Oscillator ID
267  */
268 
269 void rcc_osc_off(enum rcc_osc osc)

274  break;
275  case HSI:
276  RCC_CR &= ~RCC_CR_HSION;
277  break;
278  case HSE:
279  RCC_CR &= ~RCC_CR_HSEON;
280  break;
281  case LSE:
282  RCC_BDCR &= ~RCC_BDCR_LSEON;
283  break;
284  case LSI:
285  RCC_CSR &= ~RCC_CSR_LSION;
286  break;
287  case PLL:
288  /* don't do anything */
289  break;
290  }
291 }
