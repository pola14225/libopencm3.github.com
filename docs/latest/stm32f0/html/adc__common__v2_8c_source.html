99 /**
100  * Turn off the ADC (async)
101  * This will actually block if it needs to turn off a currently running
102  * conversion, as per ref man. (Handles injected on hardware that supports
103  * injected conversions.
104  * @sa adc_wait_power_off
105  * @param adc ADC Block register address base @ref adc_reg_base
106  */
107 void adc_power_off_async(uint32_t adc)
108 {
109  uint32_t checks = ADC_CR_ADSTART;
110  uint32_t stops = ADC_CR_ADSTP;
111 #if defined (ADC_CR_JADSTART)
112  checks |= ADC_CR_JADSTART;
113  stops |= ADC_CR_JADSTP;
114 #endif
115  if (ADC_CR(adc) & checks) {
116  ADC_CR(adc) |= stops;
117  while (ADC_CR(adc) & checks);
118  }
119  ADC_CR(adc) |= ADC_CR_ADDIS;
120 }

341 /** @brief ADC Software Triggered Conversion on Regular Channels
342  *

344  * by hardware once conversion starts.
345  *
346  * @param[in] adc ADC block register address base @ref adc_reg_base
347  */

349 {
350  /* Start conversion on regular channels. */
