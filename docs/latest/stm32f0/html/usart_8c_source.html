42 void usart_set_baudrate(uint32_t usart, uint32_t baud)
43 {
44  uint32_t clock = rcc_apb1_frequency;
45 
46  if (usart == USART1) {
47  clock = rcc_apb1_frequency;

50

56 /** @brief USART Set Word Length.
57  *
58  * The word length is set to 8 or 9 bits. Note that the last bit will be a
59  * parity bit if parity is enabled, in which case the data length will be 7
60  * or 8 bits respectively.
61  *
62  * @param[in] usart unsigned 32 bit. USART block register address base @ref
63  * usart_reg_base
64  * @param[in] bits unsigned 32 bit. Word length in bits 8 or 9.
65  */
66 
67 void usart_set_databits(uint32_t usart, uint32_t bits)
68 {
69  if (bits == 8) {
70  USART_CR1(usart) &= ~USART_CR1_M; /* 8 data bits */
71  } else {
72  USART_CR1(usart) |= USART_CR1_M; /* 9 data bits */
73  }
74 }

194 /** @brief USART Wait for Transmit Data Buffer Empty
195  *
196  * Blocks until the transmit data buffer becomes empty and is ready to accept

198  *
199  * @param[in] usart unsigned 32 bit. USART block register address base @ref

401 /*---------------------------------------------------------------------------*/
402 /** @brief USART Return Interrupt Source.
403  *
404  * Returns true if the specified interrupt flag (IDLE, RXNE, TC, TXE or OE) was
405  * set and the interrupt was enabled. If the specified flag is not an interrupt
406  * flag, the function returns false.
407  *
408  * @param[in] usart unsigned 32 bit. USART block register address base @ref
409  * usart_reg_base
410  * @param[in] flag Unsigned int32. Status register flag @ref usart_sr_flags.
411  * @returns boolean: flag and interrupt enable both set.
412  */
413 
414 bool usart_get_interrupt_source(uint32_t usart, uint32_t flag)
415 {
416  uint32_t flag_set = (USART_ISR(usart) & flag);
417  /* IDLE, RXNE, TC, TXE interrupts */
418  if ((flag >= USART_ISR_IDLE) && (flag <= USART_ISR_TXE)) {
419  return ((flag_set & USART_CR1(usart)) != 0);
420  /* Overrun error */
421  } else if (flag == USART_ISR_ORE) {
422  return flag_set && (USART_CR3(usart) & USART_CR3_CTSIE);
423  }
424 
425  return false;
426 }
