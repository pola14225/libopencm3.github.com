10 This library supports one style of the Analog to Digital Conversion System in
11 the STM32 series of ARM Cortex Microcontrollers by ST Microelectronics.
12 
13 The style of ADC Peripheral supported by this code is found in the F1, F2,
14 F37x, F38x, F4, and L1 series devices (at the time of writing) but is quite
15 different to the style found on the F0 and F30x and F31x.
16 Devices can have up to three A/D converters each with their own set of
17 registers.
18 However all the A/D converters share a common clock. On most devices, this is
19 prescaled from the APB2 clock by default by a minimum factor of 2 to a maximum
20 of 8, though on the L1 this is always a divider from the HSI. (And therefore HSI
21 _must_ be enabled before attempting to enable the ADC)
22 
23 Each A/D converter has up to ADC_MAX_CHANNELS channels:
24 @li On ADC1 the analog channels 16 and 17 are internally connected to the
25 temperature sensor and V<sub>REFINT</sub>, respectively.
26 @li On ADC2 (if available) the analog channels 16 and 17 are internally
27 connected to V<sub>SS</sub>.
28 @li On ADC3 (if available) the analog channels 9, 14, 15, 16 and 17 are
29 internally connected to V<sub>SS</sub>.
30 
31 The conversions can occur as a one-off conversion whereby the process stops once
32 conversion is complete. The conversions can also be continuous wherein a new
33 conversion starts immediately the previous conversion has ended.
34 
35 Conversion can occur as a single channel conversion or a scan of a group of
36 channels in either continuous or one-off mode. If more than one channel is
37 converted in a scan group, DMA must be used to transfer the data as there is
38 only one result register available. An interrupt can be set to occur at the end
39 of conversion, which occurs after all channels have been scanned.
40 
41 A discontinuous mode allows a subgroup of group of a channels to be converted in
42 bursts of a given length.
43 
44 Injected conversions allow a second group of channels to be converted separately
45 from the regular group. An interrupt can be set to occur at the end of
46 conversion, which occurs after all channels have been scanned.
47 
48 @section adc_api_ex Basic ADC Handling API.
49 
50 Example 1: Simple single channel conversion polled. Enable the peripheral clock
51 and ADC, reset ADC and set the prescaler divider. Set dual mode to independent
52 (default). Enable triggering for a software trigger.
53 
54 @code
55  rcc_periph_clock_enable(RCC_ADC1);
56  adc_off(ADC1);
57  rcc_periph_reset_pulse(RST_ADC1);
58  rcc_set_adcpre(RCC_CFGR_ADCPRE_PCLK2_DIV2);
59  adc_set_dual_mode(ADC_CR1_DUALMOD_IND);
60  adc_disable_scan_mode(ADC1);
61  adc_set_single_conversion_mode(ADC1);
62  adc_set_sample_time(ADC1, ADC_CHANNEL0, ADC_SMPR1_SMP_1DOT5CYC);
63  adc_enable_trigger(ADC1, ADC_CR2_EXTSEL_SWSTART);
64  adc_power_on(ADC1);
65  adc_reset_calibration(ADC1);
66  adc_calibration(ADC1);
67  adc_start_conversion_regular(ADC1);
68  while (! adc_eoc(ADC1));
69  reg16 = adc_read_regular(ADC1);
70 @endcode

562 /** @brief ADC Set a Regular Channel Conversion Sequence
563 
564 Define a sequence of channels to be converted as a regular group with a length
565 from 1 to ADC_REGULAR_SEQUENCE_MAX channels. If this is called during
566 conversion, the current conversion is reset and conversion begins again with
567 the newly defined group.
568 
569 @param[in] adc Unsigned int32. ADC block base address @ref adc_reg_base.
570 @param[in] length Unsigned int8. Number of channels in the group.
571 @param[in] channel Unsigned int8[]. Set of channels in sequence, integers 0..31.
572  */
573 
574 void adc_set_regular_sequence(uint32_t adc, uint8_t length, uint8_t channel[])
575 {
576  uint32_t fifth6 = 0;
577  uint32_t fourth6 = 0;
578  uint32_t third6 = 0;
579  uint32_t second6 = 0;
580  uint32_t first6 = 0;
581  uint8_t i = 0;
582 
583  if (length > ADC_SQR_MAX_CHANNELS_REGULAR) {
584  return;
585  }
586 
587  for (i = 1; i <= length; i++) {
588  if (i <= 6) {
589  first6 |= (channel[i - 1] << ((i - 1) * 5));
590  }
591  if ((i > 6) & (i <= 12)) {
592  second6 |= (channel[i - 1] << ((i - 6 - 1) * 5));
593  }
594  if ((i > 12) & (i <= 18)) {
595  third6 |= (channel[i - 1] << ((i - 12 - 1) * 5));
596  }
597  if ((i > 18) & (i <= 24)) {
598  fourth6 |= (channel[i - 1] << ((i - 18 - 1) * 5));
599  }
600  if ((i > 24) & (i <= 28)) {
601  fifth6 |= (channel[i - 1] << ((i - 24 - 1) * 5));
602  }
603  }
604 #if defined(ADC_SQR5)
605  ADC_SQR1(adc) = fifth6 | ((length - 1) << ADC_SQR1_L_LSB);
606  ADC_SQR2(adc) = fourth6;
607  ADC_SQR3(adc) = third6;
608  ADC_SQR4(adc) = second6;
609  ADC_SQR5(adc) = first6;
610 #else
611  ADC_SQR1(adc) = third6 | ((length - 1) << ADC_SQR1_L_LSB);
612  ADC_SQR2(adc) = second6;
613  ADC_SQR3(adc) = first6;
614 #endif
615 }
