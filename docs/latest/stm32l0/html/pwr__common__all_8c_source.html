31 /** @brief Disable Backup Domain Write Protection.
32 
33 This allows backup domain registers to be changed. These registers are write
34 protected after a reset.
35 */
36 
37 void pwr_disable_backup_domain_write_protect(void)
38 {
39  PWR_CR |= PWR_CR_DBP;
40 }
41 
42 /*---------------------------------------------------------------------------*/
43 /** @brief Re-enable Backup Domain Write Protection.
44 
45 This protects backup domain registers from inadvertent change.
46 */
47 
48 void pwr_enable_backup_domain_write_protect(void)
49 {
50  PWR_CR &= ~PWR_CR_DBP;
51 }

54 /** @brief Enable Power Voltage Detector.
55 
56 This provides voltage level threshold detection. The result of detection is
57 provided in the power voltage detector output flag (see @ref pwr_voltage_high)
58 or by setting the EXTI16 interrupt (see datasheet for configuration details).
59 
60 @param[in] pvd_level uint32_t. Taken from @ref pwr_pls.
61 */
62 
63 void pwr_enable_power_voltage_detect(uint32_t pvd_level)
64 {
65  PWR_CR &= ~PWR_CR_PLS_MASK;
66  PWR_CR |= (PWR_CR_PVDE | pvd_level);
67 }
