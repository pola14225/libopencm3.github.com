1 /** @addtogroup dac_file
2 
3 @author @htmlonly &copy; @endhtmlonly 2012 Ken Sarkies ksarkies@internode.on.net
4 
5 This library supports the Digital to Analog Conversion System in the
6 STM32F series of ARM Cortex Microcontrollers by ST Microelectronics.
7 
8 The DAC is present only in a limited set of devices, notably some
9 of the connection line, high density and XL devices.
10 
11 Two DAC channels are available, however unlike the ADC channels these
12 are separate DAC devices controlled by the same register block.
13 
14 The DAC is on APB1. Its clock must be enabled in RCC and the GPIO
15 ports set to alternate function output before it can be used.
16 The digital output driver is disabled so the output driver mode
17 (push-pull/open drain) is arbitrary.
18 
19 The DAC has a holding (buffer) register and an output register from
20 which the analog output is derived. The holding register must be
21 loaded first. If triggering is enabled the output register is loaded
22 from the holding register after a trigger occurs. If triggering is
23 not enabled the holding register contents are transferred directly
24 to the output register.
25 
26 @note To avoid nonlinearities, do not allow outputs to range close
27 to zero or V_analog.
28 
29 @section dac_api_dual Dual Channel Conversion
30 
31 There are dual modes in which both DACs are used to output data
32 simultaneously or independently on both channels. The data must be
33 presented according to the formats described in the datasheets. A
34 convenience function @ref dac_load_data_buffer_dual is provided
35 for software controlled use.
36 
37 A variety of modes are available depending on whether independent
38 or simultaneous output is desired, and whether waveforms are to be
39 superimposed. Refer to the datasheets.
40 
41 If DMA is used, only enable it for one of the channels. The DMA
42 requests will then serve data in dual format to the data register
43 dedicated to dual mode. The data will then be split and loaded to the
44 appropriate DAC following the next trigger. There are three registers
45 available, one for each of the formats: 12 bit right-aligned, 12 bit
46 left-aligned and 8 bit right-aligned. The desired format is determined
47 by specifying the appropriate register to the DMA controller.
48 
49 @section dac_api_basic_ex Basic DAC handling API.
50 
51 Set the DAC's GPIO port to any alternate function output mode. Enable the
52 DAC clock. Enable the DAC, set a trigger source and load the buffer
53 with the first value. After the DAC is triggered, load the buffer with
54 the next value. This example uses software triggering and added noise.
55 The trigger and further buffer load calls are made when data is to be
56 sent out.
57 
58 @code
59  gpio_set_mode(GPIOA, GPIO_MODE_OUTPUT_50_MHZ,
60  GPIO_CNF_OUTPUT_ALTFN_PUSHPULL, GPIO4);
61  rcc_peripheral_enable_clock(&RCC_APB1ENR, RCC_APB1ENR_DACEN);
62  dac_disable(CHANNEL_1);
63  dac_set_waveform_characteristics(DAC_CR_MAMP1_8);
64  dac_set_waveform_generation(DAC_CR_WAVE1_NOISE);
65  dac_enable(CHANNEL_1);
66  dac_set_trigger_source(DAC_CR_TSEL1_SW);
67  dac_load_data_buffer_single(0, RIGHT12, CHANNEL_1);
68  ....
69  dac_software_trigger(CHANNEL_1);
70  dac_load_data_buffer_single(value, RIGHT12, CHANNEL_1);
71 @endcode
72 
73 @section dac_api_dma_ex Simultaneous Dual DAC with DMA.
74 
75 This example in part sets up the DAC channel 1 DMA (DMA2 channel 3) to read
76 16 bit data from memory into the right-aligned 8 bit dual register DAC_DHR8RD.
77 Both DAC channels are enabled, and both triggers are set to the same timer
78 2 input as required for simultaneous operation. DMA is enabled for DAC channel
79 1 only to ensure that only one DMA request is generated.
80 
81 @code
82  dma_set_memory_size(DMA2,DMA_CHANNEL3,DMA_CCR_MSIZE_16BIT);
83  dma_set_peripheral_size(DMA2,DMA_CHANNEL3,DMA_CCR_PSIZE_16BIT);
84  dma_set_read_from_memory(DMA2,DMA_CHANNEL3);
85  dma_set_peripheral_address(DMA2,DMA_CHANNEL3,(uint32_t) &DAC_DHR8RD);
86  dma_enable_channel(DMA2,DMA_CHANNEL3);
87  ...
88  dac_trigger_enable(CHANNEL_D);
89  dac_set_trigger_source(DAC_CR_TSEL1_T2 | DAC_CR_TSEL2_T2);
90  dac_dma_enable(CHANNEL_1);
91  dac_enable(CHANNEL_D);
92 @endcode

403 Loads the appropriate digital to analog converter data register with 12 or 8 bit
404 data to be converted on a channel. The data can be aligned as follows:
405 @li right-aligned 8 bit data in bits 0-7
406 @li right-aligned 12 bit data in bits 0-11
407 @li left aligned 12 bit data in bits 4-15
408 
409 @param[in] dac_data uint16_t with appropriate alignment.
410 @param[in] dac_data_format enum ::data_align. Alignment and size.
411 @param[in] dac_channel enum ::data_channel.

415  data_channel dac_channel)
416 {

420  DAC_DHR8R1 = dac_data;
421  break;
422  case RIGHT12:
423  DAC_DHR12R1 = dac_data;
424  break;
425  case LEFT12:
426  DAC_DHR12L1 = dac_data;
427  break;
428  }
429  } else if (dac_channel == CHANNEL_2) {
430  switch (dac_data_format) {
431  case RIGHT8:
432  DAC_DHR8R2 = dac_data;
433  break;

445 /** @brief Load DAC Dual Data Register.
446 
447 Loads the appropriate digital to analog converter dual data register with 12 or
448 8 bit data to be converted for both channels. This allows high bandwidth
449 simultaneous or independent analog output. The data in both channels are aligned
450 identically.
451 
452 @param[in] dac_data1 uint16_t for channel 1 with appropriate alignment.
453 @param[in] dac_data2 uint16_t for channel 2 with appropriate alignment.
454 @param[in] dac_data_format enum ::data_align. Right or left aligned, and 8 or
455 12 bit.
456 */
457 
458 void dac_load_data_buffer_dual(uint16_t dac_data1, uint16_t dac_data2,
459  data_align dac_data_format)
460 {
461  switch (dac_data_format) {
462  case RIGHT8:
463  DAC_DHR8RD = ((dac_data1 & MASK8) | ((dac_data2 & MASK8) << 8));
464  break;
465  case RIGHT12:
466  DAC_DHR12RD = ((dac_data1 & MASK12) |
467  ((dac_data2 & MASK12) << 16));
468  break;
469  case LEFT12:
470  DAC_DHR12LD = ((dac_data1 & MASK12) |
471  ((dac_data2 & MASK12) << 16));
472  break;
473  }
474 }
