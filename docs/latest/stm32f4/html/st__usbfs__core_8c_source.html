40 /**
41  * Set the receive buffer size for a given USB endpoint.
42  *
43  * @param ep Index of endpoint to configure.
44  * @param size Size in bytes of the RX buffer.
45  */
46 void st_usbfs_set_ep_rx_bufsize(usbd_device *dev, uint8_t ep, uint32_t size)
47 {
48  (void)dev;
49  if (size > 62) {
50  if (size & 0x1f) {
51  size -= 32;
52  }
53  USB_SET_EP_RX_COUNT(ep, (size << 5) | 0x8000);
54  } else {
55  if (size & 1) {
56  size++;
57  }
58  USB_SET_EP_RX_COUNT(ep, size << 10);
59  }
60 }

62 void st_usbfs_ep_setup(usbd_device *dev, uint8_t addr, uint8_t type,
63  uint16_t max_size,
64  void (*callback) (usbd_device *usbd_dev,
65  uint8_t ep))
66 {
67  /* Translate USB standard type codes to STM32. */
68  const uint16_t typelookup[] = {
69  [USB_ENDPOINT_ATTR_CONTROL] = USB_EP_TYPE_CONTROL,
70  [USB_ENDPOINT_ATTR_ISOCHRONOUS] = USB_EP_TYPE_ISO,
71  [USB_ENDPOINT_ATTR_BULK] = USB_EP_TYPE_BULK,
72  [USB_ENDPOINT_ATTR_INTERRUPT] = USB_EP_TYPE_INTERRUPT,
73  };
74  uint8_t dir = addr & 0x80;
75  addr &= 0x7f;
76 
77  /* Assign address. */
78  USB_SET_EP_ADDR(addr, addr);
79  USB_SET_EP_TYPE(addr, typelookup[type]);
80 
81  if (dir || (addr == 0)) {
82  USB_SET_EP_TX_ADDR(addr, dev->pm_top);
83  if (callback) {
84  dev->user_callback_ctr[addr][USB_TRANSACTION_IN] =
85  (void *)callback;
86  }
87  USB_CLR_EP_TX_DTOG(addr);
88  USB_SET_EP_TX_STAT(addr, USB_EP_TX_STAT_NAK);
89  dev->pm_top += max_size;
90  }
91 
92  if (!dir) {
93  USB_SET_EP_RX_ADDR(addr, dev->pm_top);
94  st_usbfs_set_ep_rx_bufsize(dev, addr, max_size);
95  if (callback) {
96  dev->user_callback_ctr[addr][USB_TRANSACTION_OUT] =
97  (void *)callback;
98  }
99  USB_CLR_EP_RX_DTOG(addr);
100  USB_SET_EP_RX_STAT(addr, USB_EP_RX_STAT_VALID);
101  dev->pm_top += max_size;
102  }
103 }

117 void st_usbfs_ep_stall_set(usbd_device *dev, uint8_t addr,
118  uint8_t stall)
119 {
120  (void)dev;
121  if (addr == 0) {
122  USB_SET_EP_TX_STAT(addr, stall ? USB_EP_TX_STAT_STALL :
123  USB_EP_TX_STAT_NAK);
124  }
125 
126  if (addr & 0x80) {
127  addr &= 0x7F;
128 
129  USB_SET_EP_TX_STAT(addr, stall ? USB_EP_TX_STAT_STALL :
130  USB_EP_TX_STAT_NAK);
131 
132  /* Reset to DATA0 if clearing stall condition. */
133  if (!stall) {
134  USB_CLR_EP_TX_DTOG(addr);
135  }
136  } else {
137  /* Reset to DATA0 if clearing stall condition. */
138  if (!stall) {
139  USB_CLR_EP_RX_DTOG(addr);
140  }
141 
142  USB_SET_EP_RX_STAT(addr, stall ? USB_EP_RX_STAT_STALL :
143  USB_EP_RX_STAT_VALID);
144  }
145 }

147 uint8_t st_usbfs_ep_stall_get(usbd_device *dev, uint8_t addr)
148 {
149  (void)dev;
150  if (addr & 0x80) {
151  if ((*USB_EP_REG(addr & 0x7F) & USB_EP_TX_STAT) ==
152  USB_EP_TX_STAT_STALL) {
153  return 1;
154  }
155  } else {
156  if ((*USB_EP_REG(addr) & USB_EP_RX_STAT) ==
157  USB_EP_RX_STAT_STALL) {
158  return 1;
159  }
160  }
161  return 0;
162 }

164 void st_usbfs_ep_nak_set(usbd_device *dev, uint8_t addr, uint8_t nak)
165 {
166  (void)dev;
167  /* It does not make sence to force NAK on IN endpoints. */
168  if (addr & 0x80) {
169  return;
170  }
171 
172  st_usbfs_force_nak[addr] = nak;
173 
174  if (nak) {
175  USB_SET_EP_RX_STAT(addr, USB_EP_RX_STAT_NAK);
176  } else {
177  USB_SET_EP_RX_STAT(addr, USB_EP_RX_STAT_VALID);
178  }

181 uint16_t st_usbfs_ep_write_packet(usbd_device *dev, uint8_t addr,
182  const void *buf, uint16_t len)
183 {
184  (void)dev;
185  addr &= 0x7F;
186 
187  if ((*USB_EP_REG(addr) & USB_EP_TX_STAT) == USB_EP_TX_STAT_VALID) {
188  return 0;
189  }
190 
191  st_usbfs_copy_to_pm(USB_GET_EP_TX_BUFF(addr), buf, len);
192  USB_SET_EP_TX_COUNT(addr, len);
193  USB_SET_EP_TX_STAT(addr, USB_EP_TX_STAT_VALID);
194 
195  return len;
196 }

198 uint16_t st_usbfs_ep_read_packet(usbd_device *dev, uint8_t addr,
199  void *buf, uint16_t len)
200 {
201  (void)dev;
202  if ((*USB_EP_REG(addr) & USB_EP_RX_STAT) == USB_EP_RX_STAT_VALID) {
203  return 0;
204  }
205 
206  len = MIN(USB_GET_EP_RX_COUNT(addr) & 0x3ff, len);
207  st_usbfs_copy_from_pm(buf, USB_GET_EP_RX_BUFF(addr), len);
208  USB_CLR_EP_RX_CTR(addr);
209 
210  if (!st_usbfs_force_nak[addr]) {
211  USB_SET_EP_RX_STAT(addr, USB_EP_RX_STAT_VALID);
212  }
213 
214  return len;
215 }

217 void st_usbfs_poll(usbd_device *dev)
218 {
219  uint16_t istr = *USB_ISTR_REG;
220 
221  if (istr & USB_ISTR_RESET) {
222  USB_CLR_ISTR_RESET();
223  dev->pm_top = USBD_PM_TOP;
224  _usbd_reset(dev);
225  return;
226  }
227 
228  if (istr & USB_ISTR_CTR) {
229  uint8_t ep = istr & USB_ISTR_EP_ID;
230  uint8_t type;
231 
232  if (istr & USB_ISTR_DIR) {
233  /* OUT or SETUP? */
234  if (*USB_EP_REG(ep) & USB_EP_SETUP) {
235  type = USB_TRANSACTION_SETUP;
236  } else {
237  type = USB_TRANSACTION_OUT;
238  }
239  } else {
240  type = USB_TRANSACTION_IN;
241  USB_CLR_EP_TX_CTR(ep);
242  }
243 
244  if (dev->user_callback_ctr[ep][type]) {
245  dev->user_callback_ctr[ep][type] (dev, ep);
246  } else {
247  USB_CLR_EP_RX_CTR(ep);
248  }
249  }
250 
251  if (istr & USB_ISTR_SUSP) {
252  USB_CLR_ISTR_SUSP();
253  if (dev->user_callback_suspend) {
254  dev->user_callback_suspend();
255  }
256  }
257 
258  if (istr & USB_ISTR_WKUP) {
259  USB_CLR_ISTR_WKUP();
260  if (dev->user_callback_resume) {
261  dev->user_callback_resume();
262  }
263  }
264 
265  if (istr & USB_ISTR_SOF) {
266  USB_CLR_ISTR_SOF();
267  if (dev->user_callback_sof) {
268  dev->user_callback_sof();
269  }
270  }
271 
272  if (dev->user_callback_sof) {
273  *USB_CNTR_REG |= USB_CNTR_SOFM;
274  } else {
275  *USB_CNTR_REG &= ~USB_CNTR_SOFM;
276  }
277 }
