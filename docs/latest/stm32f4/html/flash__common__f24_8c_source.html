30 /** @brief Set the Program Parallelism Size
31 
32 Set the programming word width. Note carefully the power supply voltage
33 restrictions under which the different word sizes may be used. See the
34 programming manual for more information.
35 
36 @param psize: 0 (8-bit), 1 (16-bit), 2 (32-bit), 3 (64-bit)

39 static inline void flash_set_program_size(uint32_t psize)
40 {
41  FLASH_CR &= ~(((1 << 0) | (1 << 1)) << 8);
42  FLASH_CR |= psize;

213 /** @brief Program a 64 bit Word to FLASH
214 
215 This performs all operations necessary to program a 64 bit word to FLASH memory.
216 The program error flag should be checked separately for the event that memory
217 was not properly erased.
218 
219 @param[in] uint32_t address
220 @param[in] uint64_t data.
221 */
222 
223 void flash_program_double_word(uint32_t address, uint64_t data)
224 {
225  /* Ensure that all flash operations are complete. */
226  flash_wait_for_last_operation();
227  flash_set_program_size(FLASH_CR_PROGRAM_X64);
228 
229  /* Enable writes to flash. */
230  FLASH_CR |= FLASH_CR_PG;
231 
232  /* Program the double_word. */
233  MMIO64(address) = data;
234 
235  /* Wait for the write to complete. */
236  flash_wait_for_last_operation();
237 
238  /* Disable writes to flash. */
239  FLASH_CR &= ~FLASH_CR_PG;
240 }

345 /*---------------------------------------------------------------------------*/
346 /** @brief Erase a Sector of FLASH
347 
348 This performs all operations necessary to erase a sector in FLASH memory.
349 The page should be checked to ensure that it was properly erased. A sector must
350 first be fully erased before attempting to program it.
351 
352 See the reference manual or the FLASH programming manual for details.
353 
354 @param[in] uint32_t sector (0 - 11).
355 @param program_size: 0 (8-bit), 1 (16-bit), 2 (32-bit), 3 (64-bit)
356 */
357 
358 void flash_erase_sector(uint8_t sector, uint32_t program_size)
359 {
360  flash_wait_for_last_operation();
361  flash_set_program_size(program_size);
362 
363  FLASH_CR &= ~(0xF << 3);
364  FLASH_CR |= (sector << 3) & 0x78;
365  FLASH_CR |= FLASH_CR_SER;
366  FLASH_CR |= FLASH_CR_STRT;
367 
368  flash_wait_for_last_operation();
369  FLASH_CR &= ~FLASH_CR_SER;
370  FLASH_CR &= ~(0xF << 3);
371 }
372 
373 /*---------------------------------------------------------------------------*/
374 /** @brief Erase All FLASH
375 
376 This performs all operations necessary to erase all sectors in the FLASH
377 memory.
378 
379 @param program_size: 0 (8-bit), 1 (16-bit), 2 (32-bit), 3 (64-bit)
380 */
381 
382 void flash_erase_all_sectors(uint32_t program_size)
383 {
384  flash_wait_for_last_operation();
385  flash_set_program_size(program_size);
386 
387  FLASH_CR |= FLASH_CR_MER; /* Enable mass erase. */
388  FLASH_CR |= FLASH_CR_STRT; /* Trigger the erase. */
389 
390  flash_wait_for_last_operation();
391  FLASH_CR &= ~FLASH_CR_MER; /* Disable mass erase. */
392 }

394 /*---------------------------------------------------------------------------*/
395 /** @brief Program the Option Bytes
396 
397 This performs all operations necessary to program the option bytes.
398 The option bytes do not need to be erased first.
399 
400 @param[in] uint32_t data to be programmed.
401 */
402 
403 void flash_program_option_bytes(uint32_t data)
404 {
405  flash_wait_for_last_operation();
406 
407  if (FLASH_OPTCR & FLASH_OPTCR_OPTLOCK) {
408  flash_unlock_option_bytes();
409  }
410 
411  FLASH_OPTCR = data & ~0x3;
412  FLASH_OPTCR |= FLASH_OPTCR_OPTSTRT; /* Enable option byte prog. */
413  flash_wait_for_last_operation();
414 }
