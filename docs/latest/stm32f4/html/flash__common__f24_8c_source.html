109 void flash_program_double_word(uint32_t address, uint64_t data)
110 {
111  /* Ensure that all flash operations are complete. */
112  flash_wait_for_last_operation();
113  flash_set_program_size(FLASH_CR_PROGRAM_X64);
114 
115  /* Enable writes to flash. */
116  FLASH_CR |= FLASH_CR_PG;
117 
118  /* Program the first half of the word. */
119  MMIO64(address) = data;
120 
121  /* Wait for the write to complete. */
122  flash_wait_for_last_operation();
123 
124  /* Disable writes to flash. */
125  FLASH_CR &= ~FLASH_CR_PG;
126 }

186 void flash_erase_sector(uint8_t sector, uint32_t program_size)
187 {
188  flash_wait_for_last_operation();
189  flash_set_program_size(program_size);
190 
191  FLASH_CR &= ~(0xF << 3);
192  FLASH_CR |= (sector << 3) & 0x78;
193  FLASH_CR |= FLASH_CR_SER;
194  FLASH_CR |= FLASH_CR_STRT;
195 
196  flash_wait_for_last_operation();
197  FLASH_CR &= ~FLASH_CR_SER;
198  FLASH_CR &= ~(0xF << 3);
199 }

201 void flash_erase_all_sectors(uint32_t program_size)
202 {
203  flash_wait_for_last_operation();
204  flash_set_program_size(program_size);
205 
206  FLASH_CR |= FLASH_CR_MER; /* Enable mass erase. */
207  FLASH_CR |= FLASH_CR_STRT; /* Trigger the erase. */
208 
209  flash_wait_for_last_operation();
210  FLASH_CR &= ~FLASH_CR_MER; /* Disable mass erase. */
211 }

213 void flash_program_option_bytes(uint32_t data)
214 {
215  flash_wait_for_last_operation();
216 
217  if (FLASH_OPTCR & FLASH_OPTCR_OPTLOCK) {
218  flash_unlock_option_bytes();
219  }
220 
221  FLASH_OPTCR = data & ~0x3;
222  FLASH_OPTCR |= FLASH_OPTCR_OPTSTRT; /* Enable option byte prog. */
223  flash_wait_for_last_operation();
224 }
