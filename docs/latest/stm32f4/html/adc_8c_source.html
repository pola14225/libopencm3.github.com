1 /** @defgroup adc_file ADC
2 
3 @ingroup STM32F4xx
4 
5 @brief <b>libopencm3 STM32F4xx Analog to Digital Converters</b>
6 
7 @author @htmlonly &copy; @endhtmlonly 2012
8 Ken Sarkies <ksarkies@internode.on.net>
9 
10 @date 30 August 2012
11 
12 This library supports the A/D Converter Control System in the STM32 series
13 of ARM Cortex Microcontrollers by ST Microelectronics.
14 
15 Devices can have up to three A/D converters each with their own set of
16 registers. However all the A/D converters share a common clock which is
17 prescaled from the APB2 clock by default by a minimum factor of 2 to a maximum
18 of 8. The ADC resolution can be set to 12, 10, 8 or 6 bits.
19 
20 Each A/D converter has up to 19 channels:
21 @li On ADC1 the analog channels 16 is internally connected to the temperature
22 sensor, channel 17 to V<sub>REFINT</sub>, and channel 18 to V<sub>BATT</sub>.
23 @li On ADC2 and ADC3 the analog channels 16 - 18 are not used.
24 
25 The conversions can occur as a one-off conversion whereby the process stops
26 once conversion is complete. The conversions can also be continuous wherein a
27 new conversion starts immediately the previous conversion has ended.
28 
29 Conversion can occur as a single channel conversion or a scan of a group of
30 channels in either continuous or one-off mode. If more than one channel is
31 converted in a scan group, DMA must be used to transfer the data as there is
32 only one result register available. An interrupt can be set to occur at the end
33 of conversion, which occurs after all channels have been scanned.
34 
35 A discontinuous mode allows a subgroup of group of a channels to be converted
36 in bursts of a given length.
37 
38 Injected conversions allow a second group of channels to be converted
39 separately from the regular group. An interrupt can be set to occur at the end
40 of conversion, which occurs after all channels have been scanned.
41 
42 @section adc_f4_api_ex Basic ADC Handling API.
43 
44 Example 1: Simple single channel conversion polled. Enable the peripheral clock
45 and ADC, reset ADC and set the prescaler divider. Set multiple mode to
46 independent.
47 
48 @code
49 gpio_mode_setup(GPIOA, GPIO_MODE_ANALOG, GPIO_PUPD_NONE, GPIO1);
50 rcc_peripheral_enable_clock(&RCC_APB2ENR, RCC_APB2ENR_ADC1EN);
51 adc_set_clk_prescale(RCC_CFGR_ADCPRE_BY2);
52 adc_disable_scan_mode(ADC1);
53 adc_set_single_conversion_mode(ADC1);
54 adc_set_sample_time(ADC1, ADC_CHANNEL0, ADC_SMPR1_SMP_1DOT5CYC);
55 uint8_t channels[] = ADC_CHANNEL0;
56 adc_set_regular_sequence(ADC1, 1, channels);
57 adc_set_multi_mode(ADC_CCR_MULTI_INDEPENDENT);
58 adc_power_on(ADC1);
59 adc_start_conversion_regular(ADC1);
60 while (! adc_eoc(ADC1));
61 reg16 = adc_read_regular(ADC1);
62 @endcode

90 /** @brief ADC Set the Sample Time for a Single Channel
91 
92 The sampling time can be selected in ADC clock cycles from 1.5 to 239.5.
93 
94 @param[in] adc Unsigned int32. ADC block register address base @ref adc_reg_base
95 @param[in] channel Unsigned int8. ADC Channel integer 0..18 or from @ref
96 adc_channel
97 @param[in] time Unsigned int8. Sampling time selection from @ref adc_sample_rg
98  * NOTE Common with f1, f2 and f37x
99 */
100 
101 void adc_set_sample_time(uint32_t adc, uint8_t channel, uint8_t time)
102 {
103  uint32_t reg32;
104 
105  if (channel < 10) {
106  reg32 = ADC_SMPR2(adc);
107  reg32 &= ~(0x7 << (channel * 3));
108  reg32 |= (time << (channel * 3));
109  ADC_SMPR2(adc) = reg32;
110  } else {
111  reg32 = ADC_SMPR1(adc);
112  reg32 &= ~(0x7 << ((channel - 10) * 3));
113  reg32 |= (time << ((channel - 10) * 3));
114  ADC_SMPR1(adc) = reg32;
115  }
116 }
117 
118 /*---------------------------------------------------------------------------*/
119 /** @brief ADC Set the Sample Time for All Channels
120 
121 The sampling time can be selected in ADC clock cycles from 1.5 to 239.5, same
122 for all channels.
123 
124 @param[in] adc Unsigned int32. ADC block register address base @ref adc_reg_base
125 @param[in] time Unsigned int8. Sampling time selection from @ref adc_sample_rg
126  * NOTE Common with f1, f2 and f37x
127 */
128 
129 void adc_set_sample_time_on_all_channels(uint32_t adc, uint8_t time)
130 {
131  uint8_t i;
132  uint32_t reg32 = 0;
133 
134  for (i = 0; i <= 9; i++) {
135  reg32 |= (time << (i * 3));
136  }
137  ADC_SMPR2(adc) = reg32;
138 
139  for (i = 10; i <= 17; i++) {
140  reg32 |= (time << ((i - 10) * 3));
141  }
142  ADC_SMPR1(adc) = reg32;
143 }

193 /** @brief ADC Enable an External Trigger for Regular Channels
194 
195 This enables an external trigger for set of defined regular channels, and sets
196 the polarity of the trigger event: rising or falling edge or both. Note that if
197 the trigger polarity is zero, triggering is disabled.
198 
199 @param[in] adc Unsigned int32. ADC block register address base @ref adc_reg_base
200 @param[in] trigger Unsigned int32. Trigger identifier @ref adc_trigger_regular

205 void adc_enable_external_trigger_regular(uint32_t adc, uint32_t trigger,
206  uint32_t polarity)
207 {
208  uint32_t reg32 = ADC_CR2(adc);
209 
210  reg32 &= ~(ADC_CR2_EXTSEL_MASK | ADC_CR2_EXTEN_MASK);
211  reg32 |= (trigger | polarity);
212  ADC_CR2(adc) = reg32;
213 }
