1 /** @defgroup STM32F4xx_adc_file ADC
2 
3 @ingroup STM32F4xx
4 
5 @brief <b>libopencm3 STM32F4xx Analog to Digital Converters</b>
6 
7 @author @htmlonly &copy; @endhtmlonly 2012 Ken Sarkies <ksarkies@internode.on.net>
8 
9 @date 30 August 2012
10 
11 This library supports the A/D Converter Control System in the STM32 series
12 of ARM Cortex Microcontrollers by ST Microelectronics.
13 
14 Devices can have up to three A/D converters each with their own set of registers.
15 However all the A/D converters share a common clock which is prescaled from the APB2
16 clock by default by a minimum factor of 2 to a maximum of 8. The ADC resolution
17 can be set to 12, 10, 8 or 6 bits.
18 
19 Each A/D converter has up to 19 channels:
20 @li On ADC1 the analog channels 16 is internally connected to the temperature
21 sensor, channel 17 to V<sub>REFINT</sub>, and channel 18 to V<sub>BATT</sub>.
22 @li On ADC2 and ADC3 the analog channels 16 - 18 are not used.
23 
24 The conversions can occur as a one-off conversion whereby the process stops once
25 conversion is complete. The conversions can also be continuous wherein a new
26 conversion starts immediately the previous conversion has ended.
27 
28 Conversion can occur as a single channel conversion or a scan of a group of
29 channels in either continuous or one-off mode. If more than one channel is converted
30 in a scan group, DMA must be used to transfer the data as there is only one
31 result register available. An interrupt can be set to occur at the end of
32 conversion, which occurs after all channels have been scanned.
33 
34 A discontinuous mode allows a subgroup of group of a channels to be converted in
35 bursts of a given length.
36 
37 Injected conversions allow a second group of channels to be converted separately
38 from the regular group. An interrupt can be set to occur at the end of
39 conversion, which occurs after all channels have been scanned.
40 
41 @section adc_f4_api_ex Basic ADC Handling API.
42 
43 Example 1: Simple single channel conversion polled. Enable the peripheral clock
44 and ADC, reset ADC and set the prescaler divider. Set multiple mode to independent.
45 
46 @code
47 gpio_mode_setup(GPIOA, GPIO_MODE_ANALOG, GPIO_PUPD_NONE, GPIO1);
48 rcc_peripheral_enable_clock(&RCC_APB2ENR, RCC_APB2ENR_ADC1EN);
49 adc_set_clk_prescale(RCC_CFGR_ADCPRE_BY2);
50 adc_disable_scan_mode(ADC1);
51 adc_set_single_conversion_mode(ADC1);
52 adc_set_sample_time(ADC1, ADC_CHANNEL0, ADC_SMPR1_SMP_1DOT5CYC);
53 u8 channels[] = ADC_CHANNEL0;
54 adc_set_regular_sequence(ADC1, 1, channels);
55 adc_set_multi_mode(ADC_CCR_MULTI_INDEPENDENT);
56 adc_power_on(ADC1);
57 adc_start_conversion_regular(ADC1);
58 while (! adc_eoc(ADC1));
59 reg16 = adc_read_regular(ADC1);
60 @endcode

496 {
497  u32 reg32;
498 
499  if (channel < 10) {

501  reg32 &= ~(0x7 << (channel * 3));

503  ADC_SMPR2(adc) = reg32;

506  reg32 &= ~(0x7 << ((channel - 10) * 3));
507  reg32 |= (time << ((channel - 10) * 3));
508  ADC_SMPR1(adc) = reg32;
509  }
510 }

569 /** @brief ADC Set a Regular Channel Conversion Sequence
570 
571 Define a sequence of channels to be converted as a regular group with a length
572 from 1 to 16 channels. If this is called during conversion, the current conversion
573 is reset and conversion begins again with the newly defined group.
574 
575 @param[in] adc Unsigned int32. ADC block register address base @ref adc_reg_base
576 @param[in] length Unsigned int8. Number of channels in the group.
577 @param[in] channel Unsigned int8[]. Set of channels in sequence, integers 0..18.
578 */
579 
580 void adc_set_regular_sequence(u32 adc, u8 length, u8 channel[])
581 {
582  u32 reg32_1 = 0, reg32_2 = 0, reg32_3 = 0;
583  u8 i = 0;
584 
585  /* Maximum sequence length is 16 channels. */
586  if (length > 16)
587  return;
588 
589  for (i = 1; i <= length; i++) {
590  if (i <= 6)
591  reg32_3 |= (channel[i - 1] << ((i - 1) * 5));
592  if ((i > 6) & (i <= 12))
593  reg32_2 |= (channel[i - 1] << ((i - 6 - 1) * 5));
594  if ((i > 12) & (i <= 16))
595  reg32_1 |= (channel[i - 1] << ((i - 12 - 1) * 5));
596  }
597  reg32_1 |= ((length -1) << ADC_SQR1_L_LSB);
598 
599  ADC_SQR1(adc) = reg32_1;
600  ADC_SQR2(adc) = reg32_2;
601  ADC_SQR3(adc) = reg32_3;
602 }

605 /** @brief ADC Set an Injected Channel Conversion Sequence
606 
607 Defines a sequence of channels to be converted as an injected group with a length
608 from 1 to 4 channels. If this is called during conversion, the current conversion
609 is reset and conversion begins again with the newly defined group.
610 
611 @param[in] adc Unsigned int32. ADC block register address base @ref adc_reg_base
612 @param[in] length Unsigned int8. Number of channels in the group.
613 @param[in] channel Unsigned int8[]. Set of channels in sequence, integers 0..18
614 */
615 
616 void adc_set_injected_sequence(u32 adc, u8 length, u8 channel[])
617 {
618  u32 reg32 = 0;
619  u8 i = 0;
620 
621  /* Maximum sequence length is 4 channels. */
622  if ((length-1) > 3)
623  return;
624 
625  for (i = 1; i <= length; i++)
626  reg32 |= (channel[4 - i] << ((4 - i) * 5));
627 
628  reg32 |= ((length - 1) << ADC_JSQR_JL_LSB);
629 
630  ADC_JSQR(adc) = reg32;
631 }
