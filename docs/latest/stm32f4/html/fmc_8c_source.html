24 /*
25  * Install various timing values into the correct place in the
26  * SDRAM Timing Control Register format.
27  *
28  * Note that the register is 'zero' based to save bits so 1 cycle
29  * is stored as '0'. This command takes actual cycles and adjusts
30  * by subtracting 1.
31  */
32 uint32_t
33 sdram_timing(struct sdram_timing *t) {
34  uint32_t result;
35 
36  result = 0;
37  result |= ((t->trcd - 1) & 0xf) * FMC_SDTR_TRCD_SHIFT;
38  result |= ((t->trp - 1) & 0xf) * FMC_SDTR_TRP_SHIFT;
39  result |= ((t->twr - 1) & 0xf) * FMC_SDTR_TWR_SHIFT;
40  result |= ((t->trc - 1) & 0xf) * FMC_SDTR_TRC_SHIFT;
41  result |= ((t->tras - 1) & 0xf) * FMC_SDTR_TRAS_SHIFT;
42  result |= ((t->txsr - 1) & 0xf) * FMC_SDTR_TXSR_SHIFT;
43  result |= ((t->tmrd - 1) & 0xf) * FMC_SDTR_TMRD_SHIFT;
44  return result;
45 }

47 /*
48  * Send a command to the SDRAM controller, wait until it is not
49  * busy before sending. This allows you to chain sending commands
50  * and the code will pause as needed between them.
51  */
52 void
53 sdram_command(enum fmc_sdram_bank bank,
54  enum fmc_sdram_command cmd, int autorefresh, int modereg) {
55  uint32_t tmp_reg = 0;
56 
57  switch (bank) {
58  case SDRAM_BANK1:
59  tmp_reg = FMC_SDCMR_CTB1;
60  break;
61  case SDRAM_BANK2:
62  tmp_reg = FMC_SDCMR_CTB2;
63  break;
64  case SDRAM_BOTH_BANKS:
65  tmp_reg = FMC_SDCMR_CTB1 | FMC_SDCMR_CTB2;
66  break;
67  }
68  tmp_reg |= autorefresh * FMC_SDCMR_NRFS_SHIFT;
69  tmp_reg |= modereg * FMC_SDCMR_MRD_SHIFT;
70  switch (cmd) {
71  case SDRAM_CLK_CONF:
72  tmp_reg |= FMC_SDCMR_MODE_CLOCK_CONFIG_ENA;
73  break;
74  case SDRAM_AUTO_REFRESH:
75  tmp_reg |= FMC_SDCMR_MODE_AUTO_REFRESH;
76  break;
77  case SDRAM_LOAD_MODE:
78  tmp_reg |= FMC_SDCMR_MODE_LOAD_MODE_REGISTER;
79  break;
80  case SDRAM_PALL:
81  tmp_reg |= FMC_SDCMR_MODE_PALL;
82  break;
83  case SDRAM_SELF_REFRESH:
84  tmp_reg |= FMC_SDCMR_MODE_SELF_REFRESH;
85  break;
86  case SDRAM_POWER_DOWN:
87  tmp_reg |= FMC_SDCMR_MODE_POWER_DOWN;
88  break;
89  case SDRAM_NORMAL:
90  default:
91  break;
92  }
93 
94  /* Wait for the next chance to talk to the controller */
95  while (FMC_SDSR & FMC_SDSR_BUSY) ;
96 
97  /* Send the next command */
98  FMC_SDCMR = tmp_reg;
99 }
