30 /** @brief Set the Number of Wait States
31 
32 Used to match the system clock to the FLASH memory access time. See the
33 programming manual for more information on clock speed ranges. The latency must
34 be changed to the appropriate value <b>before</b> any increase in clock
35 speed, or <b>after</b> any decrease in clock speed.
36 
37 @param[in] ws values from @ref flash_latency.
38 */
39 void flash_set_ws(uint32_t ws)
40 {
41  uint32_t reg32;
42 
43  reg32 = FLASH_ACR;
44  reg32 &= ~((1 << 0) | (1 << 1) | (1 << 2));
45  reg32 |= ws;
46  FLASH_ACR = reg32;
47 }

51

56 void flash_unlock(void)
