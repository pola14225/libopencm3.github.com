148 void flash_program_double_word(u32 address, u64 data, u32 program_size)
149 {
150  /* Ensure that all flash operations are complete. */
151  flash_wait_for_last_operation();
152  flash_set_program_size(program_size);
153 
154  /* Enable writes to flash. */
155  FLASH_CR |= FLASH_CR_PG;
156 
157  /* Program the first half of the word. */
158  MMIO64(address) = data;
159 
160  /* Wait for the write to complete. */
161  flash_wait_for_last_operation();
162 
163  /* Disable writes to flash. */
164  FLASH_CR &= ~FLASH_CR_PG;
165 }

214 void flash_erase_sector(u32 sector, u32 program_size)
215 {
216  flash_wait_for_last_operation();
217  flash_set_program_size(program_size);
218 
219  FLASH_CR &= ~(((1 << 0) | (1 << 1) | (1 << 2) | (1 << 3)) << 3);
220  FLASH_CR |= sector;
221  FLASH_CR |= FLASH_CR_SER;
222  FLASH_CR |= FLASH_CR_STRT;
223 
224  flash_wait_for_last_operation();
225  FLASH_CR &= ~FLASH_CR_SER;
226  FLASH_CR &= ~(((1 << 0) | (1 << 1) | (1 << 2) | (1 << 3)) << 3);
227 }

241 void flash_program_option_bytes(u32 data)
242 {
243  flash_wait_for_last_operation();
244 
245  if (FLASH_OPTCR & FLASH_OPTCR_OPTLOCK)
246  flash_unlock_option_bytes();
247 
248  FLASH_OPTCR = data & ~0x3;
249  FLASH_OPTCR |= FLASH_OPTCR_OPTSTRT; /* Enable option byte programming. */
250  flash_wait_for_last_operation();
