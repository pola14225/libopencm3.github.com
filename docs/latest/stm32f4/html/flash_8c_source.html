155 void flash_program_double_word(u32 address, u64 data, u32 program_size)
156 {
157  /* Ensure that all flash operations are complete. */
158  flash_wait_for_last_operation();
159  flash_set_program_size(program_size);
160 
161  /* Enable writes to flash. */
162  FLASH_CR |= FLASH_CR_PG;
163 
164  /* Program the first half of the word. */
165  MMIO64(address) = data;
166 
167  /* Wait for the write to complete. */
168  flash_wait_for_last_operation();
169 
170  /* Disable writes to flash. */
171  FLASH_CR &= ~FLASH_CR_PG;
172 }

221 void flash_erase_sector(u32 sector, u32 program_size)
222 {
223  flash_wait_for_last_operation();
224  flash_set_program_size(program_size);
225 
226  FLASH_CR &= ~(((1 << 0) | (1 << 1) | (1 << 2) | (1 << 3)) << 3);
227  FLASH_CR |= sector;
228  FLASH_CR |= FLASH_CR_SER;
229  FLASH_CR |= FLASH_CR_STRT;
230 
231  flash_wait_for_last_operation();
232  FLASH_CR &= ~FLASH_CR_SER;
233  FLASH_CR &= ~(((1 << 0) | (1 << 1) | (1 << 2) | (1 << 3)) << 3);
234 }

237 {
238  flash_wait_for_last_operation();

248 void flash_program_option_bytes(u32 data)
249 {
250  flash_wait_for_last_operation();
251 
252  if (FLASH_OPTCR & FLASH_OPTCR_OPTLOCK)
253  flash_unlock_option_bytes();
254 
255  FLASH_OPTCR = data & ~0x3;

257  flash_wait_for_last_operation();
