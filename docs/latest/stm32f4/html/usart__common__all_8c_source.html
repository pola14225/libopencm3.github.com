37 /** @brief USART Set Baudrate.
38 
39 The baud rate is computed from the APB high-speed prescaler clock (for
40 USART1/6) or the APB low-speed prescaler clock (for other USARTs). These values
41 must be correctly set before calling this function (refer to the
42 rcc_clock_setup-* functions in RCC).
43 
44 @param[in] usart unsigned 32 bit. USART block register address base @ref
45 usart_reg_base
46 @param[in] baud unsigned 32 bit. Baud rate specified in Hz.
47 */
48 
49 void usart_set_baudrate(uint32_t usart, uint32_t baud)
50 {
51  uint32_t clock = rcc_ppre1_frequency;
52 
53 #if defined STM32F2 || defined STM32F4
54  if ((usart == USART1) ||
55  (usart == USART6)) {
56  clock = rcc_ppre2_frequency;
57  }
58 #else
59  if (usart == USART1) {
60  clock = rcc_ppre2_frequency;
61  }
62 #endif
63 
64  /*
65  * Yes it is as simple as that. The reference manual is
66  * talking about fractional calculation but it seems to be only
67  * marketting babble to sound awesome. It is nothing else but a
68  * simple divider to generate the correct baudrate.
69  *
70  * Note: We round() the value rather than floor()ing it, for more
71  * accurate divisor selection.
72  */
73  USART_BRR(usart) = ((2 * clock) + baud) / (2 * baud);
74 }

77 /** @brief USART Set Word Length.
78 
79 The word length is set to 8 or 9 bits. Note that the last bit will be a parity
80 bit if parity is enabled, in which case the data length will be 7 or 8 bits
81 respectively.
82 
83 @param[in] usart unsigned 32 bit. USART block register address base @ref
84 usart_reg_base
85 @param[in] bits unsigned 32 bit. Word length in bits 8 or 9.
86 */
87 
88 void usart_set_databits(uint32_t usart, uint32_t bits)
89 {
90  if (bits == 8) {
91  USART_CR1(usart) &= ~USART_CR1_M; /* 8 data bits */
92  } else {
93  USART_CR1(usart) |= USART_CR1_M; /* 9 data bits */
94  }
95 }

230 /*---------------------------------------------------------------------------*/
231 /** @brief USART Wait for Transmit Data Buffer Empty
232 
233 Blocks until the transmit data buffer becomes empty and is ready to accept the
234 next data word.
235 
236 @param[in] usart unsigned 32 bit. USART block register address base @ref
237 usart_reg_base
238 */
239 
240 void usart_wait_send_ready(uint32_t usart)
241 {
242  /* Wait until the data has been transferred into the shift register. */
243  while ((USART_SR(usart) & USART_SR_TXE) == 0);
244 }

442 /*---------------------------------------------------------------------------*/
443 /** @brief USART Return Interrupt Source.
444 
445 Returns true if the specified interrupt flag (IDLE, RXNE, TC, TXE or OE) was
446 set and the interrupt was enabled. If the specified flag is not an interrupt
447 flag, the function returns false.
448 
449 @todo These are the most important interrupts likely to be used. Others
450 relating to LIN break, and error conditions in multibuffer communication, need
451 to be added for completeness.
452 
453 @param[in] usart unsigned 32 bit. USART block register address base @ref
454 usart_reg_base
455 @param[in] flag Unsigned int32. Status register flag @ref usart_sr_flags.
456 @returns boolean: flag and interrupt enable both set.
457 */
458 
459 bool usart_get_interrupt_source(uint32_t usart, uint32_t flag)
460 {
461 uint32_t flag_set = (USART_SR(usart) & flag);
462  /* IDLE, RXNE, TC, TXE interrupts */
463  if ((flag >= USART_SR_IDLE) && (flag <= USART_SR_TXE)) {
464  return ((flag_set & USART_CR1(usart)) != 0);
465  /* Overrun error */
466  } else if (flag == USART_SR_ORE) {
467  return flag_set && (USART_CR3(usart) & USART_CR3_CTSIE);
468  }
469 
470  return false;
471 }
