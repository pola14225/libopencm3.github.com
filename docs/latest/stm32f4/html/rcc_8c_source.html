345 void rcc_wait_for_sysclk_status(enum rcc_osc osc)
346 {
347  switch (osc) {
348  case PLL:
349  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_PLL);
350  break;
351  case HSE:
352  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_HSE);
353  break;
354  case HSI:
355  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_HSI);
356  break;
357  default:
358  /* Shouldn't be reached. */
359  break;
360  }

530 void rcc_clock_setup_hse_3v3(const clock_scale_t *clock)
531 {
532  /* Enable internal high-speed oscillator. */
533  rcc_osc_on(HSI);
534  rcc_wait_for_osc_ready(HSI);
535 
536  /* Select HSI as SYSCLK source. */
537  rcc_set_sysclk_source(RCC_CFGR_SW_HSI);
538 
539  /* Enable external high-speed oscillator 8MHz. */
540  rcc_osc_on(HSE);
541  rcc_wait_for_osc_ready(HSE);
542 
543  /* Enable/disable high performance mode */
544  if (!clock->power_save) {
545  pwr_set_vos_scale(SCALE1);
546  } else {
547  pwr_set_vos_scale(SCALE2);
548  }
549 
550  /*
551  * Set prescalers for AHB, ADC, ABP1, ABP2.
552  * Do this before touching the PLL (TODO: why?).
553  */
554  rcc_set_hpre(clock->hpre);
555  rcc_set_ppre1(clock->ppre1);
556  rcc_set_ppre2(clock->ppre2);
557 
558  rcc_set_main_pll_hse(clock->pllm, clock->plln,
559  clock->pllp, clock->pllq);
560 
561  /* Enable PLL oscillator and wait for it to stabilize. */
562  rcc_osc_on(PLL);
563  rcc_wait_for_osc_ready(PLL);
564 
565  /* Configure flash settings. */
566  flash_set_ws(clock->flash_config);
567 
568  /* Select PLL as SYSCLK source. */
569  rcc_set_sysclk_source(RCC_CFGR_SW_PLL);
570 
571  /* Wait for PLL clock to be selected. */
572  rcc_wait_for_sysclk_status(PLL);
573 
574  /* Set the peripheral clock frequencies used. */
575  rcc_apb1_frequency = clock->apb1_frequency;
576  rcc_apb2_frequency = clock->apb2_frequency;
577 
578  /* Disable internal high-speed oscillator. */
579  rcc_osc_off(HSI);
580 }
