397 void rcc_wait_for_sysclk_status(enum rcc_osc osc)
398 {
399  switch (osc) {
400  case PLL:
401  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_PLL);
402  break;
403  case HSE:
404  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_HSE);
405  break;
406  case HSI:
407  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_HSI);
408  break;
409  default:
410  /* Shouldn't be reached. */
411  break;
412  }
413 }

582 void rcc_clock_setup_hse_3v3(const clock_scale_t *clock)
583 {
584  /* Enable internal high-speed oscillator. */
585  rcc_osc_on(HSI);
586  rcc_wait_for_osc_ready(HSI);
587 
588  /* Select HSI as SYSCLK source. */
589  rcc_set_sysclk_source(RCC_CFGR_SW_HSI);
590 
591  /* Enable external high-speed oscillator 8MHz. */
592  rcc_osc_on(HSE);
593  rcc_wait_for_osc_ready(HSE);
594 
595  /* Enable/disable high performance mode */
596  if (!clock->power_save) {
597  pwr_set_vos_scale(SCALE1);
598  } else {
599  pwr_set_vos_scale(SCALE2);
600  }
601 
602  /*
603  * Set prescalers for AHB, ADC, ABP1, ABP2.
604  * Do this before touching the PLL (TODO: why?).
605  */
606  rcc_set_hpre(clock->hpre);
607  rcc_set_ppre1(clock->ppre1);
608  rcc_set_ppre2(clock->ppre2);
609 
610  rcc_set_main_pll_hse(clock->pllm, clock->plln,
611  clock->pllp, clock->pllq);
612 
613  /* Enable PLL oscillator and wait for it to stabilize. */
614  rcc_osc_on(PLL);
615  rcc_wait_for_osc_ready(PLL);
616 
617  /* Configure flash settings. */
618  flash_set_ws(clock->flash_config);
619 
620  /* Select PLL as SYSCLK source. */
621  rcc_set_sysclk_source(RCC_CFGR_SW_PLL);
622 
623  /* Wait for PLL clock to be selected. */
624  rcc_wait_for_sysclk_status(PLL);
625 
626  /* Set the peripheral clock frequencies used. */
627  rcc_apb1_frequency = clock->apb1_frequency;
628  rcc_apb2_frequency = clock->apb2_frequency;
629 
630  /* Disable internal high-speed oscillator. */
631  rcc_osc_off(HSI);
632 }
