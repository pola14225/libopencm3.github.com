59 /** @brief Enable the wakeup timer interrupt
60  @warning You must unlock the registers before using this function
61 
62 */
63 void rtc_enable_wakeup_timer_interrupt(void)
64 {
65  /* FTFM:
66  * To enable the RTC Wakeup interrupt, the following sequence is
67  * required:
68  * 1. Configure and enable the EXTI Line 22 in interrupt mode and
69  * select the rising edge sensitivity.
70  */
71  exti_enable_request(EXTI22);
72  exti_set_trigger(EXTI22, EXTI_TRIGGER_RISING);
73 
74  /* 2. Configure and enable the RTC_WKUP IRQ channel in the NVIC. */
75  nvic_enable_irq(NVIC_RTC_WKUP_IRQ);
76  nvic_set_priority(NVIC_RTC_WKUP_IRQ, 1);
77 
78  /* 3. Configure the RTC to generate the RTC wakeup timer event. */
79  RTC_CR |= RTC_CR_WUTIE; /* Enable the interrupt */
80 }
