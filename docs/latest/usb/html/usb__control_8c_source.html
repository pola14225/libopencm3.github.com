42 /*
43  * According to the USB 2.0 specification, section 8.5.3, when a control
44  * transfer is stalled, the pipe becomes idle. We provide one utility to stall
45  * a transaction to reduce boilerplate code.
46  */
47 static void stall_transaction(usbd_device *usbd_dev)
48 {
49  usbd_ep_stall_set(usbd_dev, 0, 1);
50  usbd_dev->control_state.state = IDLE;
51 }

74 static void usb_control_send_chunk(usbd_device *usbd_dev)
75 {
76  if (usbd_dev->desc->bMaxPacketSize0 <
77  usbd_dev->control_state.ctrl_len) {
78  /* Data stage, normal transmission */
79  usbd_ep_write_packet(usbd_dev, 0,
80  usbd_dev->control_state.ctrl_buf,
81  usbd_dev->desc->bMaxPacketSize0);
82  usbd_dev->control_state.state = DATA_IN;
83  usbd_dev->control_state.ctrl_buf +=
84  usbd_dev->desc->bMaxPacketSize0;
85  usbd_dev->control_state.ctrl_len -=
86  usbd_dev->desc->bMaxPacketSize0;
87  } else {
88  /* Data stage, end of transmission */
89  usbd_ep_write_packet(usbd_dev, 0,
90  usbd_dev->control_state.ctrl_buf,
91  usbd_dev->control_state.ctrl_len);
92  usbd_dev->control_state.state = LAST_DATA_IN;
93  usbd_dev->control_state.ctrl_len = 0;
94  usbd_dev->control_state.ctrl_buf = NULL;
95  }
96 }

98 static int usb_control_recv_chunk(usbd_device *usbd_dev)
99 {
100  uint16_t packetsize = MIN(usbd_dev->desc->bMaxPacketSize0,
101  usbd_dev->control_state.req.wLength -
102  usbd_dev->control_state.ctrl_len);
103  uint16_t size = usbd_ep_read_packet(usbd_dev, 0,
104  usbd_dev->control_state.ctrl_buf +
105  usbd_dev->control_state.ctrl_len,
106  packetsize);
107 
108  if (size != packetsize) {
109  stall_transaction(usbd_dev);
110  return -1;
111  }
112 
113  usbd_dev->control_state.ctrl_len += size;
114 
115  return packetsize;
116 }

118 static int usb_control_request_dispatch(usbd_device *usbd_dev,
119  struct usb_setup_data *req)
120 {
121  int i, result = 0;
122  struct user_control_callback *cb = usbd_dev->user_control_callback;
123 
124  /* Call user command hook function. */
125  for (i = 0; i < MAX_USER_CONTROL_CALLBACK; i++) {
126  if (cb[i].cb == NULL) {
127  break;
128  }
129 
130  if ((req->bmRequestType & cb[i].type_mask) == cb[i].type) {
131  result = cb[i].cb(usbd_dev, req,
132  &(usbd_dev->control_state.ctrl_buf),
133  &(usbd_dev->control_state.ctrl_len),
134  &(usbd_dev->control_state.complete));
135  if (result == USBD_REQ_HANDLED ||
136  result == USBD_REQ_NOTSUPP) {
137  return result;
138  }
139  }
140  }
141 
142  /* Try standard request if not already handled. */
143  return _usbd_standard_request(usbd_dev, req,
144  &(usbd_dev->control_state.ctrl_buf),
145  &(usbd_dev->control_state.ctrl_len));
146 }

148 /* Handle commands and read requests. */
149 static void usb_control_setup_read(usbd_device *usbd_dev,
150  struct usb_setup_data *req)
151 {
152  usbd_dev->control_state.ctrl_buf = usbd_dev->ctrl_buf;
153  usbd_dev->control_state.ctrl_len = req->wLength;
154 
155  if (usb_control_request_dispatch(usbd_dev, req)) {
156  if (usbd_dev->control_state.ctrl_len) {
157  /* Go to data out stage if handled. */
158  usb_control_send_chunk(usbd_dev);
159  } else {
160  /* Go to status stage if handled. */
161  usbd_ep_write_packet(usbd_dev, 0, NULL, 0);
162  usbd_dev->control_state.state = STATUS_IN;
163  }
164  } else {
165  /* Stall endpoint on failure. */
166  stall_transaction(usbd_dev);
167  }
168 }

170 static void usb_control_setup_write(usbd_device *usbd_dev,
171  struct usb_setup_data *req)
172 {
173  if (req->wLength > usbd_dev->ctrl_buf_len) {
174  stall_transaction(usbd_dev);
175  return;
176  }
177 
178  /* Buffer into which to write received data. */
179  usbd_dev->control_state.ctrl_buf = usbd_dev->ctrl_buf;
180  usbd_dev->control_state.ctrl_len = 0;
181  /* Wait for DATA OUT stage. */
182  if (req->wLength > usbd_dev->desc->bMaxPacketSize0) {
183  usbd_dev->control_state.state = DATA_OUT;
184  } else {
185  usbd_dev->control_state.state = LAST_DATA_OUT;
186  }
187 }

192 void _usbd_control_setup(usbd_device *usbd_dev, uint8_t ea)
193 {
194  struct usb_setup_data *req = &usbd_dev->control_state.req;
195  (void)ea;
196 
197  usbd_dev->control_state.complete = NULL;
198 
199  if (usbd_ep_read_packet(usbd_dev, 0, req, 8) != 8) {
200  stall_transaction(usbd_dev);
201  return;
202  }
203 
204  if (req->wLength == 0) {
205  usb_control_setup_read(usbd_dev, req);
206  } else if (req->bmRequestType & 0x80) {
207  usb_control_setup_read(usbd_dev, req);
208  } else {
209  usb_control_setup_write(usbd_dev, req);
210  }

213 void _usbd_control_out(usbd_device *usbd_dev, uint8_t ea)
214 {
215  (void)ea;
216 
217  switch (usbd_dev->control_state.state) {
218  case DATA_OUT:
219  if (usb_control_recv_chunk(usbd_dev) < 0) {
220  break;
221  }
222  if ((usbd_dev->control_state.req.wLength -
223  usbd_dev->control_state.ctrl_len) <=
224  usbd_dev->desc->bMaxPacketSize0) {
225  usbd_dev->control_state.state = LAST_DATA_OUT;
226  }
227  break;
228  case LAST_DATA_OUT:
229  if (usb_control_recv_chunk(usbd_dev) < 0) {
230  break;
231  }
232  /*
233  * We have now received the full data payload.
234  * Invoke callback to process.
235  */
236  if (usb_control_request_dispatch(usbd_dev,
237  &(usbd_dev->control_state.req))) {
238  /* Got to status stage on success. */
239  usbd_ep_write_packet(usbd_dev, 0, NULL, 0);
240  usbd_dev->control_state.state = STATUS_IN;
241  } else {
242  stall_transaction(usbd_dev);
243  }
244  break;
245  case STATUS_OUT:
246  usbd_ep_read_packet(usbd_dev, 0, NULL, 0);
247  usbd_dev->control_state.state = IDLE;
248  if (usbd_dev->control_state.complete) {
249  usbd_dev->control_state.complete(usbd_dev,
250  &(usbd_dev->control_state.req));
251  }
252  usbd_dev->control_state.complete = NULL;
253  break;
254  default:
255  stall_transaction(usbd_dev);
256  }
257 }

259 void _usbd_control_in(usbd_device *usbd_dev, uint8_t ea)
260 {
261  (void)ea;
262  struct usb_setup_data *req = &(usbd_dev->control_state.req);
263 
264  switch (usbd_dev->control_state.state) {
265  case DATA_IN:
266  usb_control_send_chunk(usbd_dev);
267  break;
268  case LAST_DATA_IN:
269  usbd_dev->control_state.state = STATUS_OUT;
270  break;
271  case STATUS_IN:
272  if (usbd_dev->control_state.complete) {
273  usbd_dev->control_state.complete(usbd_dev,
274  &(usbd_dev->control_state.req));
275  }
276 
277  /* Exception: Handle SET ADDRESS function here... */
278  if ((req->bmRequestType == 0) &&
279  (req->bRequest == USB_REQ_SET_ADDRESS)) {
280  usbd_dev->driver->set_address(usbd_dev, req->wValue);
281  }
282  usbd_dev->control_state.state = IDLE;
283  break;
284  default:
285  stall_transaction(usbd_dev);
286  }
287 }
