61 /** Initialize the USB device controller hardware of the STM32. */
62 static usbd_device *stm32f103_usbd_init(void)
63 {
64  rcc_peripheral_enable_clock(&RCC_APB1ENR, RCC_APB1ENR_USBEN);
65  SET_REG(USB_CNTR_REG, 0);
66  SET_REG(USB_BTABLE_REG, 0);
67  SET_REG(USB_ISTR_REG, 0);
68 
69  /* Enable RESET, SUSPEND, RESUME and CTR interrupts. */
70  SET_REG(USB_CNTR_REG, USB_CNTR_RESETM | USB_CNTR_CTRM |
71  USB_CNTR_SUSPM | USB_CNTR_WKUPM);
72  return &usbd_dev;
73 }

82 /**
83  * Set the receive buffer size for a given USB endpoint.
84  *
85  * @param ep Index of endpoint to configure.
86  * @param size Size in bytes of the RX buffer.
87  */
88 static void usb_set_ep_rx_bufsize(usbd_device *dev, uint8_t ep, uint32_t size)
89 {
90  (void)dev;
91  if (size > 62) {
92  if (size & 0x1f) {
93  size -= 32;
94  }
95  USB_SET_EP_RX_COUNT(ep, (size << 5) | 0x8000);
96  } else {
97  if (size & 1) {
98  size++;
99  }
100  USB_SET_EP_RX_COUNT(ep, size << 10);
101  }
102 }

104 static void stm32f103_ep_setup(usbd_device *dev, uint8_t addr, uint8_t type,
105  uint16_t max_size,
106  void (*callback) (usbd_device *usbd_dev,
107  uint8_t ep))
108 {
109  /* Translate USB standard type codes to STM32. */
110  const uint16_t typelookup[] = {
111  [USB_ENDPOINT_ATTR_CONTROL] = USB_EP_TYPE_CONTROL,
112  [USB_ENDPOINT_ATTR_ISOCHRONOUS] = USB_EP_TYPE_ISO,
113  [USB_ENDPOINT_ATTR_BULK] = USB_EP_TYPE_BULK,
114  [USB_ENDPOINT_ATTR_INTERRUPT] = USB_EP_TYPE_INTERRUPT,
115  };
116  uint8_t dir = addr & 0x80;
117  addr &= 0x7f;
118 
119  /* Assign address. */
120  USB_SET_EP_ADDR(addr, addr);
121  USB_SET_EP_TYPE(addr, typelookup[type]);
122 
123  if (dir || (addr == 0)) {
124  USB_SET_EP_TX_ADDR(addr, dev->pm_top);
125  if (callback) {
126  dev->user_callback_ctr[addr][USB_TRANSACTION_IN] =
127  (void *)callback;
128  }
129  USB_CLR_EP_TX_DTOG(addr);
130  USB_SET_EP_TX_STAT(addr, USB_EP_TX_STAT_NAK);
131  dev->pm_top += max_size;
132  }
133 
134  if (!dir) {
135  USB_SET_EP_RX_ADDR(addr, dev->pm_top);
136  usb_set_ep_rx_bufsize(dev, addr, max_size);
137  if (callback) {
138  dev->user_callback_ctr[addr][USB_TRANSACTION_OUT] =
139  (void *)callback;
140  }
141  USB_CLR_EP_RX_DTOG(addr);
142  USB_SET_EP_RX_STAT(addr, USB_EP_RX_STAT_VALID);
143  dev->pm_top += max_size;
144  }
145 }

148 {
149  int i;
150

159 static void stm32f103_ep_stall_set(usbd_device *dev, uint8_t addr,
160  uint8_t stall)
161 {
162  (void)dev;
163  if (addr == 0) {
164  USB_SET_EP_TX_STAT(addr, stall ? USB_EP_TX_STAT_STALL :
165  USB_EP_TX_STAT_NAK);
166  }
167 
168  if (addr & 0x80) {
169  addr &= 0x7F;
170 
171  USB_SET_EP_TX_STAT(addr, stall ? USB_EP_TX_STAT_STALL :
172  USB_EP_TX_STAT_NAK);
173 
174  /* Reset to DATA0 if clearing stall condition. */
175  if (!stall) {
176  USB_CLR_EP_TX_DTOG(addr);
177  }
178  } else {
179  /* Reset to DATA0 if clearing stall condition. */
180  if (!stall) {
181  USB_CLR_EP_RX_DTOG(addr);
182  }
183 
184  USB_SET_EP_RX_STAT(addr, stall ? USB_EP_RX_STAT_STALL :
185  USB_EP_RX_STAT_VALID);
186  }
187 }

189 static uint8_t stm32f103_ep_stall_get(usbd_device *dev, uint8_t addr)
190 {
191  (void)dev;
192  if (addr & 0x80) {
193  if ((*USB_EP_REG(addr & 0x7F) & USB_EP_TX_STAT) ==
194  USB_EP_TX_STAT_STALL) {
195  return 1;
196  }
197  } else {
198  if ((*USB_EP_REG(addr) & USB_EP_RX_STAT) ==
199  USB_EP_RX_STAT_STALL) {
200  return 1;
201  }
202  }
203  return 0;

206 static void stm32f103_ep_nak_set(usbd_device *dev, uint8_t addr, uint8_t nak)
207 {
208  (void)dev;
209  /* It does not make sence to force NAK on IN endpoints. */
210  if (addr & 0x80) {

213

216  if (nak) {

230 static void usb_copy_to_pm(volatile void *vPM, const void *buf, uint16_t len)
231 {
232  const uint16_t *lbuf = buf;
233  volatile uint16_t *PM = vPM;
234 
235  for (len = (len + 1) >> 1; len; PM += 2, lbuf++, len--) {
236  *PM = *lbuf;

240 static uint16_t stm32f103_ep_write_packet(usbd_device *dev, uint8_t addr,
241  const void *buf, uint16_t len)
242 {
243  (void)dev;
244  addr &= 0x7F;
245 
246  if ((*USB_EP_REG(addr) & USB_EP_TX_STAT) == USB_EP_TX_STAT_VALID) {
247  return 0;
248  }
249 
250  usb_copy_to_pm(USB_GET_EP_TX_BUFF(addr), buf, len);
251  USB_SET_EP_TX_COUNT(addr, len);
252  USB_SET_EP_TX_STAT(addr, USB_EP_TX_STAT_VALID);
253 
254  return len;
255 }

257 /**
258  * Copy a data buffer from packet memory.
259  *
260  * @param buf Source pointer to data buffer.
261  * @param vPM Destination pointer into packet memory.
262  * @param len Number of bytes to copy.
263  */
264 static void usb_copy_from_pm(void *buf, const volatile void *vPM, uint16_t len)
265 {
266  uint16_t *lbuf = buf;
267  const volatile uint16_t *PM = vPM;
268  uint8_t odd = len & 1;
269 
270  for (len >>= 1; len; PM += 2, lbuf++, len--) {
271  *lbuf = *PM;
272  }
273 
274  if (odd) {
275  *(uint8_t *) lbuf = *(uint8_t *) PM;
276  }
277 }

279 static uint16_t stm32f103_ep_read_packet(usbd_device *dev, uint8_t addr,
280  void *buf, uint16_t len)
281 {
282  (void)dev;
283  if ((*USB_EP_REG(addr) & USB_EP_RX_STAT) == USB_EP_RX_STAT_VALID) {
284  return 0;
285  }
286 
287  len = MIN(USB_GET_EP_RX_COUNT(addr) & 0x3ff, len);
288  usb_copy_from_pm(buf, USB_GET_EP_RX_BUFF(addr), len);
289  USB_CLR_EP_RX_CTR(addr);
290 
291  if (!force_nak[addr]) {
292  USB_SET_EP_RX_STAT(addr, USB_EP_RX_STAT_VALID);
293  }
294 
295  return len;
296 }

298 static void stm32f103_poll(usbd_device *dev)
299 {
300  uint16_t istr = *USB_ISTR_REG;
301 
302  if (istr & USB_ISTR_RESET) {
303  dev->pm_top = 0x40;
304  _usbd_reset(dev);
305  USB_CLR_ISTR_RESET();
306  return;
307  }
308 
309  if (istr & USB_ISTR_CTR) {
310  uint8_t ep = istr & USB_ISTR_EP_ID;
311  uint8_t type = (istr & USB_ISTR_DIR) ? 1 : 0;
312 
313  if (type) { /* OUT or SETUP transaction */
314  type += (*USB_EP_REG(ep) & USB_EP_SETUP) ? 1 : 0;
315  } else { /* IN transaction */
316  USB_CLR_EP_TX_CTR(ep);
317  }
318 
319  if (dev->user_callback_ctr[ep][type]) {
320  dev->user_callback_ctr[ep][type] (dev, ep);
321  } else {
322  USB_CLR_EP_RX_CTR(ep);
323  }
324  }
325 
326  if (istr & USB_ISTR_SUSP) {
327  USB_CLR_ISTR_SUSP();
328  if (dev->user_callback_suspend) {
329  dev->user_callback_suspend();
330  }
331  }
332 
333  if (istr & USB_ISTR_WKUP) {
334  USB_CLR_ISTR_WKUP();
335  if (dev->user_callback_resume) {
336  dev->user_callback_resume();
337  }
338  }
339 
340  if (istr & USB_ISTR_SOF) {
341  if (dev->user_callback_sof) {
342  dev->user_callback_sof();
343  }
344  USB_CLR_ISTR_SOF();
345  }
346 }
