46 /** Initialize the USB_FS device controller hardware of the STM32. */
47 static usbd_device *efm32lg_usbd_init(void)
48 {
49  /* Enable clock */
50  CMU_HFCORECLKEN0 |= CMU_HFCORECLKEN0_USB | CMU_HFCORECLKEN0_USBC;
51  CMU_CMD = CMU_CMD_USBCCLKSEL_HFCLKNODIV;
52 
53  /* wait till clock not selected */
54  while (!(CMU_STATUS & CMU_STATUS_USBCHFCLKSEL));
55 
56  USB_GINTSTS = USB_GINTSTS_MMIS;
57 
58  USB_CTRL &= ~USB_CTRL_DMPUAP;
59  USB_ROUTE = USB_ROUTE_DMPUPEN | USB_ROUTE_PHYPEN;
60 
61  /* Wait for AHB idle. */
62  while (!(USB_GRSTCTL & USB_GRSTCTL_AHBIDL));
63  /* Do core soft reset. */
64  USB_GRSTCTL |= USB_GRSTCTL_CSRST;
65  while (USB_GRSTCTL & USB_GRSTCTL_CSRST);
66 
67  /* Force peripheral only mode. */
68  USB_GUSBCFG |= USB_GUSBCFG_FDMOD | USB_GUSBCFG_TRDT_16BIT;
69 
70  /* Full speed device. */
71  USB_DCFG |= USB_DCFG_DSPD;
72 
73  /* Restart the PHY clock. */
74  USB_PCGCCTL = 0;
75 
76  USB_GRXFSIZ = efm32lg_usb_driver.rx_fifo_size;
77  _usbd_dev.fifo_mem_top = efm32lg_usb_driver.rx_fifo_size;
78 
79  /* Unmask interrupts for TX and RX. */
80  USB_GAHBCFG |= USB_GAHBCFG_GLBLINTRMSK;
81  USB_GINTMSK = USB_GINTMSK_ENUMDNEM |
82  USB_GINTMSK_RXFLVLM |
83  USB_GINTMSK_IEPINT |
84  USB_GINTMSK_USBSUSPM |
85  USB_GINTMSK_WUIM;
86  USB_DAINTMSK = 0xF;
87  USB_DIEPMSK = USB_DIEPMSK_XFRCM;
88 
89  return &_usbd_dev;
90 }

99 static void efm32lg_ep_setup(usbd_device *usbd_dev, uint8_t addr, uint8_t type,
100  uint16_t max_size,
101  void (*callback) (usbd_device *usbd_dev, uint8_t ep))
102 {
103  /*
104  * Configure endpoint address and type. Allocate FIFO memory for
105  * endpoint. Install callback funciton.
106  */
107  uint8_t dir = addr & 0x80;
108  addr &= 0x7f;
109 
110  if (addr == 0) { /* For the default control endpoint */
111  /* Configure IN part. */
112  if (max_size >= 64) {
113  USB_DIEP0CTL = USB_DIEP0CTL_MPSIZ_64;
114  } else if (max_size >= 32) {
115  USB_DIEP0CTL = USB_DIEP0CTL_MPSIZ_32;
116  } else if (max_size >= 16) {
117  USB_DIEP0CTL = USB_DIEP0CTL_MPSIZ_16;
118  } else {
119  USB_DIEP0CTL = USB_DIEP0CTL_MPSIZ_8;
120  }
121 
122  USB_DIEP0TSIZ =
123  (max_size & USB_DIEP0TSIZ_XFRSIZ_MASK);
124  USB_DIEP0CTL |=
125  USB_DIEP0CTL_EPENA | USB_DIEP0CTL_SNAK;
126 
127  /* Configure OUT part. */
128  usbd_dev->doeptsiz[0] = USB_DIEP0TSIZ_STUPCNT_1 |
129  USB_DIEP0TSIZ_PKTCNT |
130  (max_size & USB_DIEP0TSIZ_XFRSIZ_MASK);
131  USB_DOEPx_TSIZ(0) = usbd_dev->doeptsiz[0];
132  USB_DOEPx_CTL(0) |=
133  USB_DOEP0CTL_EPENA | USB_DIEP0CTL_SNAK;
134 
135  USB_GNPTXFSIZ = ((max_size / 4) << 16) |
136  usbd_dev->driver->rx_fifo_size;
137  usbd_dev->fifo_mem_top += max_size / 4;
138  usbd_dev->fifo_mem_top_ep0 = usbd_dev->fifo_mem_top;
139 
140  return;
141  }
142 
143  if (dir) {
144  USB_DIEPTXF(addr) = ((max_size / 4) << 16) |
145  usbd_dev->fifo_mem_top;
146  usbd_dev->fifo_mem_top += max_size / 4;
147 
148  USB_DIEPx_TSIZ(addr) =
149  (max_size & USB_DIEP0TSIZ_XFRSIZ_MASK);
150  USB_DIEPx_CTL(addr) |=
151  USB_DIEP0CTL_EPENA | USB_DIEP0CTL_SNAK | (type << 18)
152  | USB_DIEP0CTL_USBAEP | USB_DIEP0CTL_SD0PID
153  | (addr << 22) | max_size;
154 
155  if (callback) {
156  usbd_dev->user_callback_ctr[addr][USB_TRANSACTION_IN] =
157  (void *)callback;
158  }
159  }
160 
161  if (!dir) {
162  usbd_dev->doeptsiz[addr] = USB_DIEP0TSIZ_PKTCNT |
163  (max_size & USB_DIEP0TSIZ_XFRSIZ_MASK);
164  USB_DOEPx_TSIZ(addr) = usbd_dev->doeptsiz[addr];
165  USB_DOEPx_CTL(addr) |= USB_DOEP0CTL_EPENA |
166  USB_DOEP0CTL_USBAEP | USB_DIEP0CTL_CNAK |
167  USB_DOEP0CTL_SD0PID | (type << 18) | max_size;
168 
169  if (callback) {
170  usbd_dev->user_callback_ctr[addr][USB_TRANSACTION_OUT] =
171  (void *)callback;
172  }
173  }
174 }

183  uint8_t stall)
184 {
185  (void)usbd_dev;
186  if (addr == 0) {
187  if (stall) {
188  USB_DIEPx_CTL(addr) |= USB_DIEP0CTL_STALL;
189  } else {
190  USB_DIEPx_CTL(addr) &= ~USB_DIEP0CTL_STALL;
191  }
192  }
193 
194  if (addr & 0x80) {
195  addr &= 0x7F;
196 
197  if (stall) {
198  USB_DIEPx_CTL(addr) |= USB_DIEP0CTL_STALL;
199  } else {
200  USB_DIEPx_CTL(addr) &= ~USB_DIEP0CTL_STALL;
201  USB_DIEPx_CTL(addr) |= USB_DIEP0CTL_SD0PID;
202  }
203  } else {
204  if (stall) {
205  USB_DOEPx_CTL(addr) |= USB_DOEP0CTL_STALL;
206  } else {
207  USB_DOEPx_CTL(addr) &= ~USB_DOEP0CTL_STALL;

243 static uint16_t efm32lg_ep_write_packet(usbd_device *usbd_dev, uint8_t addr,
244  const void *buf, uint16_t len)
245 {
246  (void)usbd_dev;
247  const uint32_t *buf32 = buf;
248  int i;
249 
250  addr &= 0x7F;
251 
252  /* Return if endpoint is already enabled. */
253  if (USB_DIEPx_TSIZ(addr) & USB_DIEP0TSIZ_PKTCNT) {
254  return 0;
255  }
256 
257  /* Enable endpoint for transmission. */
258  USB_DIEPx_TSIZ(addr) = USB_DIEP0TSIZ_PKTCNT | len;
259  USB_DIEPx_CTL(addr) |= USB_DIEP0CTL_EPENA |
260  USB_DIEP0CTL_CNAK;
261  volatile uint32_t *fifo = USB_FIFOxD(addr);
262 
263  /* Copy buffer to endpoint FIFO, note - memcpy does not work */
264  for (i = len; i > 0; i -= 4) {
265  *fifo++ = *buf32++;
266  }
267 
268  return len;
269 }

271 static uint16_t efm32lg_ep_read_packet(usbd_device *usbd_dev, uint8_t addr,
272  void *buf, uint16_t len)
273 {
274  int i;
275  uint32_t *buf32 = buf;
276  uint32_t extra;
277 
278  len = MIN(len, usbd_dev->rxbcnt);
279  usbd_dev->rxbcnt -= len;
280 
281  volatile uint32_t *fifo = USB_FIFOxD(addr);
282  for (i = len; i >= 4; i -= 4) {
283  *buf32++ = *fifo++;
284  }
285 
286  if (i) {
287  extra = *fifo++;
288  memcpy(buf32, &extra, i);
289  }
290 
291  USB_DOEPx_TSIZ(addr) = usbd_dev->doeptsiz[addr];
292  USB_DOEPx_CTL(addr) |= USB_DOEP0CTL_EPENA |
293  (usbd_dev->force_nak[addr] ?
294  USB_DOEP0CTL_SNAK : USB_DOEP0CTL_CNAK);
295 
296  return len;
297 }

299 static void efm32lg_poll(usbd_device *usbd_dev)
300 {
301  /* Read interrupt status register. */
302  uint32_t intsts = USB_GINTSTS;
303  int i;
304 
305  if (intsts & USB_GINTSTS_ENUMDNE) {
306  /* Handle USB RESET condition. */
307  USB_GINTSTS = USB_GINTSTS_ENUMDNE;
308  usbd_dev->fifo_mem_top = usbd_dev->driver->rx_fifo_size;
309  _usbd_reset(usbd_dev);
310  return;
311  }
312 
313  /* Note: RX and TX handled differently in this device. */
314  if (intsts & USB_GINTSTS_RXFLVL) {
315  /* Receive FIFO non-empty. */
316  uint32_t rxstsp = USB_GRXSTSP;
317  uint32_t pktsts = rxstsp & USB_GRXSTSP_PKTSTS_MASK;
318  if ((pktsts != USB_GRXSTSP_PKTSTS_OUT) &&
319  (pktsts != USB_GRXSTSP_PKTSTS_SETUP)) {
320  return;
321  }
322 
323  uint8_t ep = rxstsp & USB_GRXSTSP_EPNUM_MASK;
324  uint8_t type;
325  if (pktsts == USB_GRXSTSP_PKTSTS_SETUP) {
326  type = USB_TRANSACTION_SETUP;
327  } else {
328  type = USB_TRANSACTION_OUT;
329  }
330 
331  /* Save packet size for stm32f107_ep_read_packet(). */
332  usbd_dev->rxbcnt = (rxstsp & USB_GRXSTSP_BCNT_MASK) >> 4;
333 
334  /*
335  * FIXME: Why is a delay needed here?
336  * This appears to fix a problem where the first 4 bytes
337  * of the DATA OUT stage of a control transaction are lost.
338  */
339  for (i = 0; i < 1000; i++) {
340  __asm__("nop");
341  }
342 
343  if (usbd_dev->user_callback_ctr[ep][type]) {
344  usbd_dev->user_callback_ctr[ep][type] (usbd_dev, ep);
345  }
346 
347  /* Discard unread packet data. */
348  for (i = 0; i < usbd_dev->rxbcnt; i += 4) {
349  (void)*USB_FIFOxD(ep);
350  }
351 
352  usbd_dev->rxbcnt = 0;
353  }
354 
355  /*
356  * There is no global interrupt flag for transmit complete.
357  * The XFRC bit must be checked in each USB_DIEPx_INT(x).
358  */
359  for (i = 0; i < ENDPOINT_COUNT; i++) { /* Iterate over endpoints. */
360  if (USB_DIEPx_INT(i) & USB_DIEP_INT_XFRC) {
361  /* Transfer complete. */
362  if (usbd_dev->user_callback_ctr[i]
363  [USB_TRANSACTION_IN]) {
364  usbd_dev->user_callback_ctr[i]
365  [USB_TRANSACTION_IN](usbd_dev, i);
366  }
367 
368  USB_DIEPx_INT(i) = USB_DIEP_INT_XFRC;
369  }
370  }
371 
372  if (intsts & USB_GINTSTS_USBSUSP) {
373  if (usbd_dev->user_callback_suspend) {
374  usbd_dev->user_callback_suspend();
375  }
376  USB_GINTSTS = USB_GINTSTS_USBSUSP;
377  }
378 
379  if (intsts & USB_GINTSTS_WKUPINT) {
380  if (usbd_dev->user_callback_resume) {
381  usbd_dev->user_callback_resume();
382  }
383  USB_GINTSTS = USB_GINTSTS_WKUPINT;
384  }
385 
386  if (intsts & USB_GINTSTS_SOF) {
387  if (usbd_dev->user_callback_sof) {
388  usbd_dev->user_callback_sof();
389  }
390  USB_GINTSTS = USB_GINTSTS_SOF;
391  }
392 
393  if (usbd_dev->user_callback_sof) {
394  USB_GINTMSK |= USB_GINTMSK_SOFM;
395  } else {
396  USB_GINTMSK &= ~USB_GINTMSK_SOFM;
397  }
398 }
