340 static void scsi_read_capacity(usbd_mass_storage *ms,
341  struct usb_msc_trans *trans,
342  enum trans_event event)
343 {
344  if (EVENT_CBW_VALID == event) {
345  trans->msd_buf[0] = ms->block_count >> 24;
346  trans->msd_buf[1] = 0xff & (ms->block_count >> 16);
347  trans->msd_buf[2] = 0xff & (ms->block_count >> 8);
348  trans->msd_buf[3] = 0xff & ms->block_count;

350  /* Block size: 512 */

354  trans->msd_buf[7] = 0;

356  set_sbc_status_good(ms);

434 static void scsi_inquiry(usbd_mass_storage *ms,
435  struct usb_msc_trans *trans,
436  enum trans_event event)
437 {
438  if (EVENT_CBW_VALID == event) {
439  uint8_t evpd;
440  uint8_t *buf;
441 
442  buf = get_cbw_buf(trans);
443  evpd = 1 & buf[1];
444 
445  if (0 == evpd) {
446  size_t len;
447  trans->bytes_to_write = sizeof(_spc3_inquiry_response);
448  memcpy(trans->msd_buf, _spc3_inquiry_response, sizeof(_spc3_inquiry_response));
449 
450  len = strlen(ms->vendor_id);
451  len = MIN(len, 8);
452  memcpy(&trans->msd_buf[8], ms->vendor_id, len);
453 
454  len = strlen(ms->product_id);
455  len = MIN(len, 16);
456  memcpy(&trans->msd_buf[16], ms->product_id, len);
457 
458  len = strlen(ms->product_revision_level);
459  len = MIN(len, 4);
460  memcpy(&trans->msd_buf[32], ms->product_revision_level, len);
461 
462  trans->csw.csw.dCSWDataResidue = sizeof(_spc3_inquiry_response);
463 
464  set_sbc_status_good(ms);
465  } else {
466  /* TODO: Add VPD 0x83 support */
467  /* TODO: Add VPD 0x00 support */

472 static void scsi_command(usbd_mass_storage *ms,
473  struct usb_msc_trans *trans,
474  enum trans_event event)
475 {
476  if (EVENT_CBW_VALID == event) {
477  /* Setup the default success */
478  trans->csw_sent = 0;
479  trans->csw.csw.dCSWSignature = CSW_SIGNATURE;
480  trans->csw.csw.dCSWTag = trans->cbw.cbw.dCBWTag;
481  trans->csw.csw.dCSWDataResidue = 0;
482  trans->csw.csw.bCSWStatus = CSW_STATUS_SUCCESS;
483 
484  trans->bytes_to_write = 0;
485  trans->bytes_to_read = 0;
486  trans->byte_count = 0;
487  }
488 
489  switch (trans->cbw.cbw.CBWCB[0]) {
490  case SCSI_TEST_UNIT_READY:
491  case SCSI_SEND_DIAGNOSTIC:
492  /* Do nothing, just send the success. */
493  set_sbc_status_good(ms);
494  break;
495  case SCSI_FORMAT_UNIT:
496  scsi_format_unit(ms, trans, event);
497  break;
498  case SCSI_REQUEST_SENSE:
499  scsi_request_sense(ms, trans, event);
500  break;
501  case SCSI_MODE_SENSE_6:
502  scsi_mode_sense_6(ms, trans, event);
503  break;
504  case SCSI_READ_6:
505  scsi_read_6(ms, trans, event);
506  break;
507  case SCSI_INQUIRY:
508  scsi_inquiry(ms, trans, event);
509  break;
510  case SCSI_READ_CAPACITY:
511  scsi_read_capacity(ms, trans, event);
512  break;
513  case SCSI_READ_10:
514  scsi_read_10(ms, trans, event);
515  break;
516  case SCSI_WRITE_6:
517  scsi_write_6(ms, trans, event);
518  break;
519  case SCSI_WRITE_10:
520  scsi_write_10(ms, trans, event);
521  break;
522  default:
523  set_sbc_status(ms, SBC_SENSE_KEY_ILLEGAL_REQUEST,
524  SBC_ASC_INVALID_COMMAND_OPERATION_CODE,
525  SBC_ASCQ_NA);
526 
527  trans->bytes_to_write = 0;
528  trans->bytes_to_read = 0;
529  trans->csw.csw.bCSWStatus = CSW_STATUS_FAILED;
530  break;
531  }
532 }

536 /** @brief Handle the USB 'OUT' requests. */
537 static void msc_data_rx_cb(usbd_device *usbd_dev, uint8_t ep)
538 {
539  usbd_mass_storage *ms;
540  struct usb_msc_trans *trans;
541  int len, max_len, left;
542  void *p;
543 
544  ms = &_mass_storage;
545  trans = &ms->trans;
546 
547  /* RX only */
548  left = sizeof(struct usb_msc_cbw) - trans->cbw_cnt;
549  if (0 < left) {
550  max_len = MIN(ms->ep_out_size, left);
551  p = &trans->cbw.buf[0x1ff & trans->cbw_cnt];
552  len = usbd_ep_read_packet(usbd_dev, ep, p, max_len);
553  trans->cbw_cnt += len;
554 
555  if (sizeof(struct usb_msc_cbw) == trans->cbw_cnt) {
556  scsi_command(ms, trans, EVENT_CBW_VALID);
557  if (trans->byte_count < trans->bytes_to_read) {
558  /* We must wait until there is something to
559  * read again. */
560  return;
561  }
562  }
563  }
564 
565  if (trans->byte_count < trans->bytes_to_read) {
566  if (0 < trans->block_count) {
567  if ((0 == trans->byte_count) && (NULL != ms->lock)){
568  (*ms->lock)();
569  }
570  }
571 
572  left = trans->bytes_to_read - trans->byte_count;
573  max_len = MIN(ms->ep_out_size, left);
574  p = &trans->msd_buf[0x1ff & trans->byte_count];
575  len = usbd_ep_read_packet(usbd_dev, ep, p, max_len);
576  trans->byte_count += len;
577 
578  if (0 < trans->block_count) {
579  if (0 == (0x1ff & trans->byte_count)) {
580  uint32_t lba;
581 
582  lba = trans->lba_start + trans->current_block;
583  if (0 != (*ms->write_block)(lba, trans->msd_buf)) {
584  /* Error */
585  }
586  trans->current_block++;
587  }
588  }
589  } else if (trans->byte_count < trans->bytes_to_write) {
590  if (0 < trans->block_count) {
591  if ((0 == trans->byte_count) && (NULL != ms->lock)) {
592  (*ms->lock)();
593  }
594 
595  if (0 == (0x1ff & trans->byte_count)) {
596  uint32_t lba;
597

599  if (0 != (*ms->read_block)(lba, trans->msd_buf)) {
600  /* Error */
601  }
602  trans->current_block++;
603  }
604  }
605

607  max_len = MIN(ms->ep_out_size, left);

609  len = usbd_ep_write_packet(usbd_dev, ms->ep_in, p, max_len);
610  trans->byte_count += len;
611  } else {
612  if (0 < trans->block_count) {
613  if (trans->current_block == trans->block_count) {
614  uint32_t lba;
615 
616  lba = trans->lba_start + trans->current_block;
617  if (0 != (*ms->write_block)(lba, trans->msd_buf)) {
618  /* Error */
619  }
620 
621  trans->current_block = 0;
622  if (NULL != ms->unlock){
623  (*ms->unlock)();
624  }
625  }
626  }
627  if (false == trans->csw_valid) {
628  scsi_command(ms, trans, EVENT_NEED_STATUS);
629  trans->csw_valid = true;
630  }
631 
632  left = sizeof(struct usb_msc_csw) - trans->csw_sent;
633  if (0 < left) {
634  max_len = MIN(ms->ep_out_size, left);
635  p = &trans->csw.buf[trans->csw_sent];
636  len = usbd_ep_write_packet(usbd_dev, ms->ep_in, p, max_len);
637  trans->csw_sent += len;
638  }
639  }
