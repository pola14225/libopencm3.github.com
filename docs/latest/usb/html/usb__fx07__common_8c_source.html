29 /* The FS core and the HS core have the same register layout.
30  * As the code can be used on both cores, the registers offset is modified
31  * according to the selected cores base address. */
32 #define dev_base_address (usbd_dev->driver->base_address)
33 #define REBASE(x) MMIO32((x) + (dev_base_address))
34 #define REBASE_FIFO(x) (&MMIO32((dev_base_address) + (OTG_FIFO(x))))

41 void stm32fx07_ep_setup(usbd_device *usbd_dev, uint8_t addr, uint8_t type,
42  uint16_t max_size,
43  void (*callback) (usbd_device *usbd_dev, uint8_t ep))
44 {
45  /*
46  * Configure endpoint address and type. Allocate FIFO memory for
47  * endpoint. Install callback funciton.
48  */
49  uint8_t dir = addr & 0x80;
50  addr &= 0x7f;
51 
52  if (addr == 0) { /* For the default control endpoint */
53  /* Configure IN part. */
54  if (max_size >= 64) {
55  REBASE(OTG_DIEPCTL0) = OTG_DIEPCTL0_MPSIZ_64;
56  } else if (max_size >= 32) {
57  REBASE(OTG_DIEPCTL0) = OTG_DIEPCTL0_MPSIZ_32;
58  } else if (max_size >= 16) {
59  REBASE(OTG_DIEPCTL0) = OTG_DIEPCTL0_MPSIZ_16;
60  } else {
61  REBASE(OTG_DIEPCTL0) = OTG_DIEPCTL0_MPSIZ_8;
62  }
63 
64  REBASE(OTG_DIEPTSIZ0) =
65  (max_size & OTG_DIEPSIZ0_XFRSIZ_MASK);
66  REBASE(OTG_DIEPCTL0) |=
67  OTG_DIEPCTL0_EPENA | OTG_DIEPCTL0_SNAK;
68 
69  /* Configure OUT part. */
70  usbd_dev->doeptsiz[0] = OTG_DIEPSIZ0_STUPCNT_1 |
71  OTG_DIEPSIZ0_PKTCNT |
72  (max_size & OTG_DIEPSIZ0_XFRSIZ_MASK);
73  REBASE(OTG_DOEPTSIZ(0)) = usbd_dev->doeptsiz[0];
74  REBASE(OTG_DOEPCTL(0)) |=
75  OTG_DOEPCTL0_EPENA | OTG_DIEPCTL0_SNAK;
76 
77  REBASE(OTG_GNPTXFSIZ) = ((max_size / 4) << 16) |
78  usbd_dev->driver->rx_fifo_size;
79  usbd_dev->fifo_mem_top += max_size / 4;
80  usbd_dev->fifo_mem_top_ep0 = usbd_dev->fifo_mem_top;
81 
82  return;
83  }
84 
85  if (dir) {
86  REBASE(OTG_DIEPTXF(addr)) = ((max_size / 4) << 16) |
87  usbd_dev->fifo_mem_top;
88  usbd_dev->fifo_mem_top += max_size / 4;
89 
90  REBASE(OTG_DIEPTSIZ(addr)) =
91  (max_size & OTG_DIEPSIZ0_XFRSIZ_MASK);
92  REBASE(OTG_DIEPCTL(addr)) |=
93  OTG_DIEPCTL0_EPENA | OTG_DIEPCTL0_SNAK | (type << 18)
94  | OTG_DIEPCTL0_USBAEP | OTG_DIEPCTLX_SD0PID
95  | (addr << 22) | max_size;
96 
97  if (callback) {
98  usbd_dev->user_callback_ctr[addr][USB_TRANSACTION_IN] =
99  (void *)callback;
100  }
101  }
102 
103  if (!dir) {
104  usbd_dev->doeptsiz[addr] = OTG_DIEPSIZ0_PKTCNT |
105  (max_size & OTG_DIEPSIZ0_XFRSIZ_MASK);
106  REBASE(OTG_DOEPTSIZ(addr)) = usbd_dev->doeptsiz[addr];
107  REBASE(OTG_DOEPCTL(addr)) |= OTG_DOEPCTL0_EPENA |
108  OTG_DOEPCTL0_USBAEP | OTG_DIEPCTL0_CNAK |
109  OTG_DOEPCTLX_SD0PID | (type << 18) | max_size;
110 
111  if (callback) {
112  usbd_dev->user_callback_ctr[addr][USB_TRANSACTION_OUT] =
113  (void *)callback;
114  }
115  }
116 }

118 void stm32fx07_endpoints_reset(usbd_device *usbd_dev)
119 {
120  int i;
121  /* The core resets the endpoints automatically on reset. */
122  usbd_dev->fifo_mem_top = usbd_dev->fifo_mem_top_ep0;
123 
124  /* Disable any currently active endpoints */
125  for (i = 1; i < 4; i++) {
126  if (REBASE(OTG_DOEPCTL(i)) & OTG_DOEPCTL0_EPENA) {
127  REBASE(OTG_DOEPCTL(i)) |= OTG_DOEPCTL0_EPDIS;
128  }
129  if (REBASE(OTG_DIEPCTL(i)) & OTG_DIEPCTL0_EPENA) {
130  REBASE(OTG_DIEPCTL(i)) |= OTG_DIEPCTL0_EPDIS;
131  }
132  }
133 
134  /* Flush all tx/rx fifos */
135  REBASE(OTG_GRSTCTL) = OTG_GRSTCTL_TXFFLSH | OTG_GRSTCTL_TXFNUM_ALL | OTG_GRSTCTL_RXFFLSH;
136 }

138 void stm32fx07_ep_stall_set(usbd_device *usbd_dev, uint8_t addr, uint8_t stall)
139 {
140  if (addr == 0) {
141  if (stall) {
142  REBASE(OTG_DIEPCTL(addr)) |= OTG_DIEPCTL0_STALL;
143  } else {
144  REBASE(OTG_DIEPCTL(addr)) &= ~OTG_DIEPCTL0_STALL;
145  }
146  }
147 
148  if (addr & 0x80) {
149  addr &= 0x7F;
150 
151  if (stall) {
152  REBASE(OTG_DIEPCTL(addr)) |= OTG_DIEPCTL0_STALL;
153  } else {
154  REBASE(OTG_DIEPCTL(addr)) &= ~OTG_DIEPCTL0_STALL;
155  REBASE(OTG_DIEPCTL(addr)) |= OTG_DIEPCTLX_SD0PID;
156  }
157  } else {
158  if (stall) {
159  REBASE(OTG_DOEPCTL(addr)) |= OTG_DOEPCTL0_STALL;
160  } else {
161  REBASE(OTG_DOEPCTL(addr)) &= ~OTG_DOEPCTL0_STALL;
162  REBASE(OTG_DOEPCTL(addr)) |= OTG_DOEPCTLX_SD0PID;
163  }
164  }

195 uint16_t stm32fx07_ep_write_packet(usbd_device *usbd_dev, uint8_t addr,
196  const void *buf, uint16_t len)
197 {
198  const uint32_t *buf32 = buf;
199  int i;
200 
201  addr &= 0x7F;
202 
203  /* Return if endpoint is already enabled. */
204  if (REBASE(OTG_DIEPTSIZ(addr)) & OTG_DIEPSIZ0_PKTCNT) {
205  return 0;
206  }
207 
208  /* Enable endpoint for transmission. */
209  REBASE(OTG_DIEPTSIZ(addr)) = OTG_DIEPSIZ0_PKTCNT | len;
210  REBASE(OTG_DIEPCTL(addr)) |= OTG_DIEPCTL0_EPENA |
211  OTG_DIEPCTL0_CNAK;
212  volatile uint32_t *fifo = REBASE_FIFO(addr);
213 
214  /* Copy buffer to endpoint FIFO, note - memcpy does not work */
215  for (i = len; i > 0; i -= 4) {
216  *fifo++ = *buf32++;
217  }
218 
219  return len;
220 }

222 uint16_t stm32fx07_ep_read_packet(usbd_device *usbd_dev, uint8_t addr,
223  void *buf, uint16_t len)
224 {
225  int i;
226  uint32_t *buf32 = buf;
227  uint32_t extra;
228 
229  len = MIN(len, usbd_dev->rxbcnt);
230  usbd_dev->rxbcnt -= len;
231 
232  volatile uint32_t *fifo = REBASE_FIFO(addr);
233  for (i = len; i >= 4; i -= 4) {
234  *buf32++ = *fifo++;
235  }
236 
237  if (i) {
238  extra = *fifo++;
239  memcpy(buf32, &extra, i);
240  }
241 
242  REBASE(OTG_DOEPTSIZ(addr)) = usbd_dev->doeptsiz[addr];
243  REBASE(OTG_DOEPCTL(addr)) |= OTG_DOEPCTL0_EPENA |
244  (usbd_dev->force_nak[addr] ?
245  OTG_DOEPCTL0_SNAK : OTG_DOEPCTL0_CNAK);
246 
247  return len;
248 }

250 void stm32fx07_poll(usbd_device *usbd_dev)
251 {
252  /* Read interrupt status register. */
253  uint32_t intsts = REBASE(OTG_GINTSTS);
254  int i;
255 
256  if (intsts & OTG_GINTSTS_ENUMDNE) {
257  /* Handle USB RESET condition. */
258  REBASE(OTG_GINTSTS) = OTG_GINTSTS_ENUMDNE;
259  usbd_dev->fifo_mem_top = usbd_dev->driver->rx_fifo_size;
260  _usbd_reset(usbd_dev);
261  return;
262  }
263 
264  /* Note: RX and TX handled differently in this device. */
265  if (intsts & OTG_GINTSTS_RXFLVL) {
266  /* Receive FIFO non-empty. */
267  uint32_t rxstsp = REBASE(OTG_GRXSTSP);
268  uint32_t pktsts = rxstsp & OTG_GRXSTSP_PKTSTS_MASK;
269  if ((pktsts != OTG_GRXSTSP_PKTSTS_OUT) &&
270  (pktsts != OTG_GRXSTSP_PKTSTS_SETUP)) {
271  return;
272  }
273 
274  uint8_t ep = rxstsp & OTG_GRXSTSP_EPNUM_MASK;
275  uint8_t type;
276  if (pktsts == OTG_GRXSTSP_PKTSTS_SETUP) {
277  type = USB_TRANSACTION_SETUP;
278  } else {
279  type = USB_TRANSACTION_OUT;
280  }
281 
282  /* Save packet size for stm32f107_ep_read_packet(). */
283  usbd_dev->rxbcnt = (rxstsp & OTG_GRXSTSP_BCNT_MASK) >> 4;
284 
285  /*
286  * FIXME: Why is a delay needed here?
287  * This appears to fix a problem where the first 4 bytes
288  * of the DATA OUT stage of a control transaction are lost.
289  */
290  for (i = 0; i < 1000; i++) {
291  __asm__("nop");
292  }
293 
294  if (usbd_dev->user_callback_ctr[ep][type]) {
295  usbd_dev->user_callback_ctr[ep][type] (usbd_dev, ep);
296  }
297 
298  /* Discard unread packet data. */
299  for (i = 0; i < usbd_dev->rxbcnt; i += 4) {
300  (void)*REBASE_FIFO(ep);
301  }
302 
303  usbd_dev->rxbcnt = 0;
304  }
305 
306  /*
307  * There is no global interrupt flag for transmit complete.
308  * The XFRC bit must be checked in each OTG_DIEPINT(x).
309  */
310  for (i = 0; i < 4; i++) { /* Iterate over endpoints. */
311  if (REBASE(OTG_DIEPINT(i)) & OTG_DIEPINTX_XFRC) {
312  /* Transfer complete. */
313  if (usbd_dev->user_callback_ctr[i]
314  [USB_TRANSACTION_IN]) {
315  usbd_dev->user_callback_ctr[i]
316  [USB_TRANSACTION_IN](usbd_dev, i);
317  }
318 
319  REBASE(OTG_DIEPINT(i)) = OTG_DIEPINTX_XFRC;
320  }
321  }
322 
323  if (intsts & OTG_GINTSTS_USBSUSP) {
324  if (usbd_dev->user_callback_suspend) {
325  usbd_dev->user_callback_suspend();
326  }
327  REBASE(OTG_GINTSTS) = OTG_GINTSTS_USBSUSP;
328  }
329 
330  if (intsts & OTG_GINTSTS_WKUPINT) {
331  if (usbd_dev->user_callback_resume) {
332  usbd_dev->user_callback_resume();
333  }
334  REBASE(OTG_GINTSTS) = OTG_GINTSTS_WKUPINT;
335  }
336 
337  if (intsts & OTG_GINTSTS_SOF) {
338  if (usbd_dev->user_callback_sof) {
339  usbd_dev->user_callback_sof();
340  }
341  REBASE(OTG_GINTSTS) = OTG_GINTSTS_SOF;
342  }
343 
344  if (usbd_dev->user_callback_sof) {
345  REBASE(OTG_GINTMSK) |= OTG_GINTMSK_SOFM;
346  } else {
347  REBASE(OTG_GINTMSK) &= ~OTG_GINTMSK_SOFM;
348  }
349 }
