65 static uint16_t build_config_descriptor(usbd_device *usbd_dev,
66  uint8_t index, uint8_t *buf, uint16_t len)
67 {
68  uint8_t *tmpbuf = buf;
69  const struct usb_config_descriptor *cfg = &usbd_dev->config[index];
70  uint16_t count, total = 0, totallen = 0;
71  uint16_t i, j, k;
72 
73  memcpy(buf, cfg, count = MIN(len, cfg->bLength));
74  buf += count;
75  len -= count;
76  total += count;
77  totallen += cfg->bLength;
78 
79  /* For each interface... */
80  for (i = 0; i < cfg->bNumInterfaces; i++) {
81  /* Interface Association Descriptor, if any */
82  if (cfg->interface[i].iface_assoc) {
83  const struct usb_iface_assoc_descriptor *assoc =
84  cfg->interface[i].iface_assoc;
85  memcpy(buf, assoc, count = MIN(len, assoc->bLength));
86  buf += count;
87  len -= count;
88  total += count;
89  totallen += assoc->bLength;
90  }
91  /* For each alternate setting... */
92  for (j = 0; j < cfg->interface[i].num_altsetting; j++) {
93  const struct usb_interface_descriptor *iface =
94  &cfg->interface[i].altsetting[j];
95  /* Copy interface descriptor. */
96  memcpy(buf, iface, count = MIN(len, iface->bLength));
97  buf += count;
98  len -= count;
99  total += count;
100  totallen += iface->bLength;
101  /* Copy extra bytes (function descriptors). */
102  if (iface->extra) {
103  memcpy(buf, iface->extra,
104  count = MIN(len, iface->extralen));
105  buf += count;
106  len -= count;
107  total += count;
108  totallen += iface->extralen;
109  }
110  /* For each endpoint... */
111  for (k = 0; k < iface->bNumEndpoints; k++) {
112  const struct usb_endpoint_descriptor *ep =
113  &iface->endpoint[k];
114  memcpy(buf, ep, count = MIN(len, ep->bLength));
115  buf += count;
116  len -= count;
117  total += count;
118  totallen += ep->bLength;
119  /* Copy extra bytes (class specific). */
120  if (ep->extra) {
121  memcpy(buf, ep->extra,
122  count = MIN(len, ep->extralen));
123  buf += count;
124  len -= count;
125  total += count;
126  totallen += ep->extralen;
127  }
128  }
129  }
130  }
131 
132  /* Fill in wTotalLength. */
133  *(uint16_t *)(tmpbuf + 2) = totallen;
134 
135  return total;
136 }

148 static int usb_standard_get_descriptor(usbd_device *usbd_dev,
149  struct usb_setup_data *req,
150  uint8_t **buf, uint16_t *len)
151 {
152  int i, array_idx, descr_idx;
153  struct usb_string_descriptor *sd;
154 
155  descr_idx = usb_descriptor_index(req->wValue);
156 
157  switch (usb_descriptor_type(req->wValue)) {
158  case USB_DT_DEVICE:
159  *buf = (uint8_t *) usbd_dev->desc;
160  *len = MIN(*len, usbd_dev->desc->bLength);
161  return USBD_REQ_HANDLED;
162  case USB_DT_CONFIGURATION:
163  *buf = usbd_dev->ctrl_buf;
164  *len = build_config_descriptor(usbd_dev, descr_idx, *buf, *len);
165  return USBD_REQ_HANDLED;
166  case USB_DT_STRING:
167  sd = (struct usb_string_descriptor *)usbd_dev->ctrl_buf;
168 
169  if (descr_idx == 0) {
170  /* Send sane Language ID descriptor... */
171  sd->wData[0] = USB_LANGID_ENGLISH_US;
172  sd->bLength = sizeof(sd->bLength) +
173  sizeof(sd->bDescriptorType) +
174  sizeof(sd->wData[0]);
175 
176  *len = MIN(*len, sd->bLength);
177  } else {
178  array_idx = descr_idx - 1;
179 
180  if (!usbd_dev->strings) {
181  /* Device doesn't support strings. */
182  return USBD_REQ_NOTSUPP;
183  }
184 
185  /* Check that string index is in range. */
186  if (array_idx >= usbd_dev->num_strings) {
187  return USBD_REQ_NOTSUPP;
188  }
189 
190  /* Strings with Language ID differnet from
191  * USB_LANGID_ENGLISH_US are not supported */
192  if (req->wIndex != USB_LANGID_ENGLISH_US) {
193  return USBD_REQ_NOTSUPP;
194  }
195 
196  /* This string is returned as UTF16, hence the
197  * multiplication
198  */
199  sd->bLength = strlen(usbd_dev->strings[array_idx]) * 2 +
200  sizeof(sd->bLength) +
201  sizeof(sd->bDescriptorType);
202 
203  *len = MIN(*len, sd->bLength);
204 
205  for (i = 0; i < (*len / 2) - 1; i++) {
206  sd->wData[i] =
207  usbd_dev->strings[array_idx][i];
208  }
209  }
210 
211  sd->bDescriptorType = USB_DT_STRING;
212  *buf = (uint8_t *)sd;
213 
214  return USBD_REQ_HANDLED;
215  }
216  return USBD_REQ_NOTSUPP;
217 }

245 static int usb_standard_set_configuration(usbd_device *usbd_dev,
246  struct usb_setup_data *req,
247  uint8_t **buf, uint16_t *len)
248 {
249  unsigned i;
250  int found_index = -1;
251  const struct usb_config_descriptor *cfg;
252 
253  (void)req;
254  (void)buf;
255  (void)len;
256 
257  if (req->wValue > 0) {
258  for (i = 0; i < usbd_dev->desc->bNumConfigurations; i++) {
259  if (req->wValue
260  == usbd_dev->config[i].bConfigurationValue) {
261  found_index = i;
262  break;
263  }
264  }
265  if (found_index < 0) {
266  return USBD_REQ_NOTSUPP;
267  }
268  }
269 
270  usbd_dev->current_config = found_index + 1;
271 
272  if (usbd_dev->current_config > 0) {
273  cfg = &usbd_dev->config[usbd_dev->current_config - 1];
274 
275  /* reset all alternate settings configuration */
276  for (i = 0; i < cfg->bNumInterfaces; i++) {
277  if (cfg->interface[i].cur_altsetting) {
278  *cfg->interface[i].cur_altsetting = 0;
279  }
280  }
281  }
282 
283  /* Reset all endpoints. */
284  usbd_dev->driver->ep_reset(usbd_dev);
285 
286  if (usbd_dev->user_callback_set_config[0]) {
287  /*
288  * Flush control callbacks. These will be reregistered
289  * by the user handler.
290  */
291  for (i = 0; i < MAX_USER_CONTROL_CALLBACK; i++) {
292  usbd_dev->user_control_callback[i].cb = NULL;
293  }
294 
295  for (i = 0; i < MAX_USER_SET_CONFIG_CALLBACK; i++) {
296  if (usbd_dev->user_callback_set_config[i]) {
297  usbd_dev->user_callback_set_config[i](usbd_dev,
298  req->wValue);
299  }
300  }
301  }
302 
303  return 1;
304 }

459 int _usbd_standard_request_device(usbd_device *usbd_dev,
460  struct usb_setup_data *req, uint8_t **buf,
461  uint16_t *len)
462 {
463  int (*command)(usbd_device *usbd_dev, struct usb_setup_data *req,
464  uint8_t **buf, uint16_t *len) = NULL;
465 
466  switch (req->bRequest) {
467  case USB_REQ_CLEAR_FEATURE:
468  case USB_REQ_SET_FEATURE:
469  if (req->wValue == USB_FEAT_DEVICE_REMOTE_WAKEUP) {
470  /* Device wakeup code goes here. */
471  }
472 
473  if (req->wValue == USB_FEAT_TEST_MODE) {
474  /* Test mode code goes here. */
475  }
476 
477  break;
478  case USB_REQ_SET_ADDRESS:
479  /*
480  * SET ADDRESS is an exception.
481  * It is only processed at STATUS stage.
482  */
483  command = usb_standard_set_address;
484  break;
485  case USB_REQ_SET_CONFIGURATION:
486  command = usb_standard_set_configuration;
487  break;
488  case USB_REQ_GET_CONFIGURATION:
489  command = usb_standard_get_configuration;
490  break;
491  case USB_REQ_GET_DESCRIPTOR:
492  command = usb_standard_get_descriptor;
493  break;
494  case USB_REQ_GET_STATUS:
495  /*
496  * GET_STATUS always responds with zero reply.
497  * The application may override this behaviour.
498  */
499  command = usb_standard_device_get_status;
500  break;
501  case USB_REQ_SET_DESCRIPTOR:
502  /* SET_DESCRIPTOR is optional and not implemented. */
503  break;
504  }
505 
506  if (!command) {
507  return 0;
508  }
509 
510  return command(usbd_dev, req, buf, len);
511 }
