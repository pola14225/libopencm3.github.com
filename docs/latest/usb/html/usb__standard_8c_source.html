59 static uint16_t build_config_descriptor(usbd_device *usbd_dev,
60  uint8_t index, uint8_t *buf, uint16_t len)
61 {
62  uint8_t *tmpbuf = buf;
63  const struct usb_config_descriptor *cfg = &usbd_dev->config[index];
64  uint16_t count, total = 0, totallen = 0;
65  uint16_t i, j, k;
66 
67  memcpy(buf, cfg, count = MIN(len, cfg->bLength));
68  buf += count;
69  len -= count;
70  total += count;
71  totallen += cfg->bLength;
72 
73  /* For each interface... */
74  for (i = 0; i < cfg->bNumInterfaces; i++) {
75  /* Interface Association Descriptor, if any */
76  if (cfg->interface[i].iface_assoc) {
77  const struct usb_iface_assoc_descriptor *assoc =
78  cfg->interface[i].iface_assoc;
79  memcpy(buf, assoc, count = MIN(len, assoc->bLength));
80  buf += count;
81  len -= count;
82  total += count;
83  totallen += assoc->bLength;
84  }
85  /* For each alternate setting... */
86  for (j = 0; j < cfg->interface[i].num_altsetting; j++) {
87  const struct usb_interface_descriptor *iface =
88  &cfg->interface[i].altsetting[j];
89  /* Copy interface descriptor. */
90  memcpy(buf, iface, count = MIN(len, iface->bLength));
91  buf += count;
92  len -= count;
93  total += count;
94  totallen += iface->bLength;
95  /* Copy extra bytes (function descriptors). */
96  if (iface->extra) {
97  memcpy(buf, iface->extra,
98  count = MIN(len, iface->extralen));
99  buf += count;
100  len -= count;
101  total += count;
102  totallen += iface->extralen;
103  }
104  /* For each endpoint... */
105  for (k = 0; k < iface->bNumEndpoints; k++) {
106  const struct usb_endpoint_descriptor *ep =
107  &iface->endpoint[k];
108  memcpy(buf, ep, count = MIN(len, ep->bLength));
109  buf += count;
110  len -= count;
111  total += count;
112  totallen += ep->bLength;
113  /* Copy extra bytes (class specific). */
114  if (ep->extra) {
115  memcpy(buf, ep->extra,
116  count = MIN(len, ep->extralen));
117  buf += count;
118  len -= count;
119  total += count;
120  totallen += ep->extralen;
121  }
122  }
123  }
124  }
125 
126  /* Fill in wTotalLength. */
127  *(uint16_t *)(tmpbuf + 2) = totallen;
128 
129  return total;
130 }

142 static int usb_standard_get_descriptor(usbd_device *usbd_dev,
143  struct usb_setup_data *req,
144  uint8_t **buf, uint16_t *len)
145 {
146  int i, array_idx, descr_idx;
147  struct usb_string_descriptor *sd;
148 
149  descr_idx = usb_descriptor_index(req->wValue);
150 
151  switch (usb_descriptor_type(req->wValue)) {
152  case USB_DT_DEVICE:
153  *buf = (uint8_t *) usbd_dev->desc;
154  *len = MIN(*len, usbd_dev->desc->bLength);
155  return USBD_REQ_HANDLED;
156  case USB_DT_CONFIGURATION:
157  *buf = usbd_dev->ctrl_buf;
158  *len = build_config_descriptor(usbd_dev, descr_idx, *buf, *len);
159  return USBD_REQ_HANDLED;
160  case USB_DT_STRING:
161  sd = (struct usb_string_descriptor *)usbd_dev->ctrl_buf;
162 
163  if (descr_idx == 0) {
164  /* Send sane Language ID descriptor... */
165  sd->wData[0] = USB_LANGID_ENGLISH_US;
166  sd->bLength = sizeof(sd->bLength) +
167  sizeof(sd->bDescriptorType) +
168  sizeof(sd->wData[0]);
169 
170  *len = MIN(*len, sd->bLength);
171  } else {
172  array_idx = descr_idx - 1;
173 
174  if (!usbd_dev->strings) {
175  /* Device doesn't support strings. */
176  return USBD_REQ_NOTSUPP;
177  }
178 
179  /* Check that string index is in range. */
180  if (array_idx >= usbd_dev->num_strings) {
181  return USBD_REQ_NOTSUPP;
182  }
183 
184  /* Strings with Language ID differnet from
185  * USB_LANGID_ENGLISH_US are not supported */
186  if (req->wIndex != USB_LANGID_ENGLISH_US) {
187  return USBD_REQ_NOTSUPP;
188  }
189 
190  /* Ths string is returned as UTF16, hence the
191  * multiplication
192  */
193  sd->bLength = strlen(usbd_dev->strings[array_idx]) * 2 +
194  sizeof(sd->bLength) +
195  sizeof(sd->bDescriptorType);
196 
197  *len = MIN(*len, sd->bLength);
198 
199  for (i = 0; i < (*len / 2) - 1; i++) {
200  sd->wData[i] =
201  usbd_dev->strings[array_idx][i];
202  }
203  }
204 
205  sd->bDescriptorType = USB_DT_STRING;
206  *buf = (uint8_t *)sd;
207 
208  return USBD_REQ_HANDLED;
209  }
210  return USBD_REQ_NOTSUPP;
211 }

240  struct usb_setup_data *req,
241  uint8_t **buf, uint16_t *len)
242 {
243  int i;
244 
245  (void)req;
246  (void)buf;
247  (void)len;
248 
249  /* Is this correct, or should we reset alternate settings. */
250  if (req->wValue == usbd_dev->current_config) {
251  return 1;
252  }
253

256  /* Reset all endpoints. */

258

263  */

265  usbd_dev->user_control_callback[i].cb = NULL;

269  if (usbd_dev->user_callback_set_config[i]) {
270  usbd_dev->user_callback_set_config[i](usbd_dev,
271  req->wValue);
272  }
273  }
274  }
275 
276  return 1;

402 int _usbd_standard_request_device(usbd_device *usbd_dev,
403  struct usb_setup_data *req, uint8_t **buf,
404  uint16_t *len)
405 {
406  int (*command)(usbd_device *usbd_dev, struct usb_setup_data *req,
407  uint8_t **buf, uint16_t *len) = NULL;
408 
409  switch (req->bRequest) {
410  case USB_REQ_CLEAR_FEATURE:
411  case USB_REQ_SET_FEATURE:
412  if (req->wValue == USB_FEAT_DEVICE_REMOTE_WAKEUP) {
413  /* Device wakeup code goes here. */
414  }
415 
416  if (req->wValue == USB_FEAT_TEST_MODE) {
417  /* Test mode code goes here. */
418  }
419 
420  break;
421  case USB_REQ_SET_ADDRESS:
422  /*
423  * SET ADDRESS is an exception.
424  * It is only processed at STATUS stage.
425  */
426  command = usb_standard_set_address;
427  break;
428  case USB_REQ_SET_CONFIGURATION:
429  command = usb_standard_set_configuration;
430  break;
431  case USB_REQ_GET_CONFIGURATION:
432  command = usb_standard_get_configuration;
433  break;
434  case USB_REQ_GET_DESCRIPTOR:
435  command = usb_standard_get_descriptor;
436  break;
437  case USB_REQ_GET_STATUS:
438  /*
439  * GET_STATUS always responds with zero reply.
440  * The application may override this behaviour.
441  */
442  command = usb_standard_device_get_status;
443  break;
444  case USB_REQ_SET_DESCRIPTOR:
445  /* SET_DESCRIPTOR is optional and not implemented. */
446  break;
447  }
448 
449  if (!command) {
450  return 0;
451  }
452 
453  return command(usbd_dev, req, buf, len);
454 }
