52 /** Initialize the USB device controller hardware of the STM32. */
53 static usbd_device *stm32f207_usbd_init(void)
54 {
55  OTG_HS_GINTSTS = OTG_HS_GINTSTS_MMIS;
56 
57  OTG_HS_GUSBCFG |= OTG_HS_GUSBCFG_PHYSEL;
58  /* Enable VBUS sensing in device mode and power down the PHY. */
59  OTG_HS_GCCFG |= OTG_HS_GCCFG_VBUSBSEN | OTG_HS_GCCFG_PWRDWN;
60 
61  /* Wait for AHB idle. */
62  while (!(OTG_HS_GRSTCTL & OTG_HS_GRSTCTL_AHBIDL));
63  /* Do core soft reset. */
64  OTG_HS_GRSTCTL |= OTG_HS_GRSTCTL_CSRST;
65  while (OTG_HS_GRSTCTL & OTG_HS_GRSTCTL_CSRST);
66 
67  /* Force peripheral only mode. */
68  OTG_HS_GUSBCFG |= OTG_HS_GUSBCFG_FDMOD | OTG_HS_GUSBCFG_TRDT_MASK;
69 
70  /* Full speed device. */
71  OTG_HS_DCFG |= OTG_HS_DCFG_DSPD;
72 
73  /* Restart the PHY clock. */
74  OTG_HS_PCGCCTL = 0;
75 
76  OTG_HS_GRXFSIZ = stm32f207_usb_driver.rx_fifo_size;
77  usbd_dev.fifo_mem_top = stm32f207_usb_driver.rx_fifo_size;
78 
79  /* Unmask interrupts for TX and RX. */
80  OTG_HS_GAHBCFG |= OTG_HS_GAHBCFG_GINT;
81  OTG_HS_GINTMSK = OTG_HS_GINTMSK_ENUMDNEM |
82  OTG_HS_GINTMSK_RXFLVLM |
83  OTG_HS_GINTMSK_IEPINT |
84  OTG_HS_GINTMSK_USBSUSPM |
85  OTG_HS_GINTMSK_WUIM |
86  OTG_HS_GINTMSK_SOFM;
87  OTG_HS_DAINTMSK = 0xF;
88  OTG_HS_DIEPMSK = OTG_HS_DIEPMSK_XFRCM;
89 
90  return &usbd_dev;
91 }
