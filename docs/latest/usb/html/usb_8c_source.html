42 /**
43  * Main initialization entry point.
44  *
45  * Initialize the USB firmware library to implement the USB device described
46  * by the descriptors provided.
47  *
48  * It is required that the 48MHz USB clock is already available.
49  *
50  * @param driver TODO
51  * @param dev Pointer to USB device descriptor. This must not be changed while
52  * the device is in use.
53  * @param conf Pointer to array of USB configuration descriptors. These must
54  * not be changed while the device is in use. The length of this
55  * array is determined by the bNumConfigurations field in the
56  * device descriptor.
57  * @param strings TODO
58  * @param control_buffer Pointer to array that would hold the data
59  * received during control requests with DATA
60  * stage
61  * @param control_buffer_size Size of control_buffer
62  * @return Zero on success (currently cannot fail).
63  */
64 usbd_device *usbd_init(const usbd_driver *driver,
65  const struct usb_device_descriptor *dev,
66  const struct usb_config_descriptor *conf,
67  const char **strings, int num_strings,
68  uint8_t *control_buffer, uint16_t control_buffer_size)
69 {
70  usbd_device *usbd_dev;
71 
72  usbd_dev = driver->init();
73 
74  usbd_dev->driver = driver;
75  usbd_dev->desc = dev;
76  usbd_dev->config = conf;
77  usbd_dev->strings = strings;
78  usbd_dev->num_strings = num_strings;
79  usbd_dev->ctrl_buf = control_buffer;
80  usbd_dev->ctrl_buf_len = control_buffer_size;
81 
82  usbd_dev->user_callback_ctr[0][USB_TRANSACTION_SETUP] =
83  _usbd_control_setup;
84  usbd_dev->user_callback_ctr[0][USB_TRANSACTION_OUT] =
85  _usbd_control_out;
86  usbd_dev->user_callback_ctr[0][USB_TRANSACTION_IN] =
87  _usbd_control_in;

89  return usbd_dev;
90 }

114 void _usbd_reset(usbd_device *usbd_dev)
115 {
116  usbd_dev->current_address = 0;
117  usbd_dev->current_config = 0;
118  usbd_ep_setup(usbd_dev, 0, USB_ENDPOINT_ATTR_CONTROL, 64, NULL);
119  usbd_dev->driver->set_address(usbd_dev, 0);
120 
121  if (usbd_dev->user_callback_reset) {
122  usbd_dev->user_callback_reset();
123  }
124 }
