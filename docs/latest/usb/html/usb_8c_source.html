43 /**
44  * Main initialization entry point.
45  *
46  * Initialize the USB firmware library to implement the USB device described
47  * by the descriptors provided.
48  *
49  * It is required that the 48MHz USB clock is already available.
50  *
51  * @param driver TODO
52  * @param dev Pointer to USB device descriptor. This must not be changed while
53  * the device is in use.
54  * @param conf Pointer to array of USB configuration descriptors. These must
55  * not be changed while the device is in use. The length of this
56  * array is determined by the bNumConfigurations field in the
57  * device descriptor.
58  * @param strings TODO
59  * @return Zero on success (currently cannot fail).
60  */
61 usbd_device *usbd_init(const usbd_driver *driver,
62  const struct usb_device_descriptor *dev,
63  const struct usb_config_descriptor *conf,
64  const char **strings, int num_strings)
65 {
66  usbd_device *usbd_dev;
67 
68  usbd_dev = driver->init();
69 
70  usbd_dev->driver = driver;
71  usbd_dev->desc = dev;
72  usbd_dev->config = conf;
73  usbd_dev->strings = strings;
74  usbd_dev->num_strings = num_strings;
75  usbd_dev->ctrl_buf = usbd_control_buffer;
76  usbd_dev->ctrl_buf_len = sizeof(usbd_control_buffer);
77 
78  usbd_dev->user_callback_ctr[0][USB_TRANSACTION_SETUP] =
79  _usbd_control_setup;
80  usbd_dev->user_callback_ctr[0][USB_TRANSACTION_OUT] =
81  _usbd_control_out;
82  usbd_dev->user_callback_ctr[0][USB_TRANSACTION_IN] =
83  _usbd_control_in;
84 
85  return usbd_dev;
86 }

115 void _usbd_reset(usbd_device *usbd_dev)
116 {
117  usbd_dev->current_address = 0;
118  usbd_dev->current_config = 0;
119  usbd_ep_setup(usbd_dev, 0, USB_ENDPOINT_ATTR_CONTROL, 64, NULL);
120  usbd_dev->driver->set_address(usbd_dev, 0);
