44 /** @brief I2C Reset.
45 
46 The I2C peripheral and all its associated configuration registers are placed in
47 the reset condition. The reset is effected via the RCC peripheral reset system.
48 
49 @param[in] i2c Unsigned int32. I2C peripheral identifier @ref i2c_reg_base.
50 */
51 
52 void i2c_reset(uint32_t i2c)
53 {
54  switch (i2c) {
55  case I2C1:
56  rcc_peripheral_reset(&RCC_APB1RSTR, RCC_APB1RSTR_I2C1RST);
57  rcc_peripheral_clear_reset(&RCC_APB1RSTR, RCC_APB1RSTR_I2C1RST);
58  break;
59  case I2C2:
60  rcc_peripheral_reset(&RCC_APB1RSTR, RCC_APB1RSTR_I2C2RST);
61  rcc_peripheral_clear_reset(&RCC_APB1RSTR, RCC_APB1RSTR_I2C2RST);
62  break;
63  }
64 }

192 /*---------------------------------------------------------------------------*/
193 /** @brief I2C Set Peripheral Clock Frequency.
194 
195 Set the peripheral clock frequency: 2MHz to 36MHz (the APB frequency). Note
196 that this is <b> not </b> the I2C bus clock. This is set in conjunction with
197 the Clock Control register to generate the Master bus clock, see @ref
198 i2c_set_ccr
199 
200 @param[in] i2c Unsigned int32. I2C register base address @ref i2c_reg_base.
201 @param[in] freq Unsigned int8. Clock Frequency Setting @ref i2c_clock.
202 */
203 
204 void i2c_set_clock_frequency(uint32_t i2c, uint8_t freq)
205 {
206  uint16_t reg16;
207  reg16 = I2C_CR2(i2c) & 0xffc0; /* Clear bits [5:0]. */
208  reg16 |= freq;
209  I2C_CR2(i2c) = reg16;
210 }

213 /** @brief I2C Set Bus Clock Frequency.
214 
215 Set the bus clock frequency. This is a 12 bit number (0...4095) calculated
216 from the formulae given in the STM32F1 reference manual in the description
217 of the CCR field. It is a divisor of the peripheral clock frequency
218 @ref i2c_set_clock_frequency modified by the fast mode setting
219 @ref i2c_set_fast_mode
220 
221 @todo provide additional API assitance to set the clock, eg macros
222 
223 @param[in] i2c Unsigned int32. I2C register base address @ref i2c_reg_base.
224 @param[in] freq Unsigned int16. Bus Clock Frequency Setting 0...4095.

226 
227 void i2c_set_ccr(uint32_t i2c, uint16_t freq)
228 {
229  uint16_t reg16;
230  reg16 = I2C_CCR(i2c) & 0xf000; /* Clear bits [11:0]. */
231  reg16 |= freq;
