32 void exti_set_trigger(uint32_t extis, enum exti_trigger_type trig)
33 {
34  switch (trig) {
35  case EXTI_TRIGGER_RISING:
36  EXTI_RTSR |= extis;

42  break;
43  case EXTI_TRIGGER_BOTH:

85 /*
86  * Remap an external interrupt line to the corresponding pin on the
87  * specified GPIO port.
88  *
89  * TODO: This could be rewritten in fewer lines of code.
90  */
91 void exti_select_source(uint32_t exti, uint32_t gpioport)
92 {
93  uint32_t line;
94  for (line = 0; line < 16; line++) {
95  if (!(exti & (1 << line))) {
96  continue;
97  }
98 
99  uint32_t bits = 0, mask = 0x0F;
100 
101  switch (gpioport) {
102  case GPIOA:
103  bits = 0;
104  break;
105  case GPIOB:
106  bits = 1;
107  break;
108  case GPIOC:
109  bits = 2;
110  break;
111  case GPIOD:
112  bits = 3;
113  break;
114 #if defined(GPIOE) && defined(GPIO_PORT_E_BASE)
115  case GPIOE:
116  bits = 4;
117  break;
118 #endif
119 #if defined(GPIOF) && defined(GPIO_PORT_F_BASE)
120  case GPIOF:
121  bits = 5;
122  break;
123 #endif
124 #if defined(GPIOG) && defined(GPIO_PORT_G_BASE)
125  case GPIOG:
126  bits = 6;
127  break;
128 #endif
129 #if defined(GPIOH) && defined(GPIO_PORT_H_BASE)
130  case GPIOH:
131  bits = 7;
132  break;
133 #endif
134 #if defined(GPIOI) && defined(GPIO_PORT_I_BASE)
135  case GPIOI:
136  bits = 8;
137  break;
138 #endif
139 #if defined(GPIOJ) && defined(GPIO_PORT_J_BASE)
140  case GPIOJ:
141  bits = 9;
142  break;
143 #endif
144 #if defined(GPIOK) && defined(GPIO_PORT_K_BASE)
145  case GPIOK:
146  bits = 10;
147  break;
148 #endif
149  }
150 
151  uint8_t shift = (uint8_t)(4 * (line % 4));
152  uint32_t reg = line / 4;
153  bits <<= shift;
154  mask <<= shift;
155 
156 #if defined(AFIO_BASE)
157  AFIO_EXTICR(reg) = (AFIO_EXTICR(reg) & ~mask) | bits;
158 #else
159  SYSCFG_EXTICR(reg) = (SYSCFG_EXTICR(reg) & ~mask) | bits;
160 #endif
161  };
162 }
