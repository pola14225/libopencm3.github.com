59 /** @brief IWDG Set Period in Milliseconds
60 
61 The countdown period is converted into count and prescale values. The maximum
62 period is 32.76 seconds; values above this are truncated. Periods less than 1ms
63 are not supported by this library.
64 
65 A delay of up to 5 clock cycles of the LSI clock (about 156 microseconds)
66 can occasionally occur if the prescale or preload registers are currently busy
67 loading a previous value.
68 
69 @param[in] period uint32_t Period in milliseconds (< 32760) from a watchdog
70 reset until a system reset is issued.
71 */
72 
73 void iwdg_set_period_ms(uint32_t period)
74 {
75  uint32_t count, prescale, reload, exponent;
76 
77  /* Set the count to represent ticks of the 32kHz LSI clock */
78  count = (period << 5);
79 
80  /* Strip off the first 12 bits to get the prescale value required */
81  prescale = (count >> 12);
82  if (prescale > 256) {
83  exponent = IWDG_PR_DIV256; reload = COUNT_MASK;
84  } else if (prescale > 128) {
85  exponent = IWDG_PR_DIV256; reload = (count >> 8);
86  } else if (prescale > 64) {
87  exponent = IWDG_PR_DIV128; reload = (count >> 7);
88  } else if (prescale > 32) {
89  exponent = IWDG_PR_DIV64; reload = (count >> 6);
90  } else if (prescale > 16) {
91  exponent = IWDG_PR_DIV32; reload = (count >> 5);
92  } else if (prescale > 8) {
93  exponent = IWDG_PR_DIV16; reload = (count >> 4);
94  } else if (prescale > 4) {
95  exponent = IWDG_PR_DIV8; reload = (count >> 3);
96  } else {
97  exponent = IWDG_PR_DIV4; reload = (count >> 2);
98  }
99 
100  /* Avoid the undefined situation of a zero count */
101  if (count == 0) {
102  count = 1;
103  }
104 
105  while (iwdg_prescaler_busy());
106  IWDG_KR = IWDG_KR_UNLOCK;
107  IWDG_PR = exponent;
108  while (iwdg_reload_busy());
109  IWDG_KR = IWDG_KR_UNLOCK;
110  IWDG_RLR = (reload & COUNT_MASK);
111 }
