163 void rcc_wait_for_osc_ready(osc_t osc)
164 {
165  switch (osc) {
166  case RCC_PLL:
167  while ((RCC_CR & RCC_CR_PLLRDY) == 0);
168  break;
169  case RCC_HSE:
170  while ((RCC_CR & RCC_CR_HSERDY) == 0);
171  break;
172  case RCC_HSI:
173  while ((RCC_CR & RCC_CR_HSIRDY) == 0);
174  break;
175  case RCC_LSE:
176  while ((RCC_BDCR & RCC_BDCR_LSERDY) == 0);
177  break;
178  case RCC_LSI:
179  while ((RCC_CSR & RCC_CSR_LSIRDY) == 0);
180  break;
181  }

184 void rcc_wait_for_sysclk_status(osc_t osc)
185 {
186  switch (osc) {
187  case RCC_PLL:
188  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_PLL);
189  break;
190  case RCC_HSE:
191  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_HSE);
192  break;
193  case RCC_HSI:
194  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_HSI);
195  break;
196  default:
197  /* Shouldn't be reached. */
198  break;
199  }
200 }

367 void rcc_clock_setup_hse_3v3(const struct rcc_clock_scale *clock)
368 {
369  /* Enable internal high-speed oscillator. */
370  rcc_osc_on(RCC_HSI);
371  rcc_wait_for_osc_ready(RCC_HSI);
372 
373  /* Select HSI as SYSCLK source. */
374  rcc_set_sysclk_source(RCC_CFGR_SW_HSI);
375 
376  /* Enable external high-speed oscillator 8MHz. */
377  rcc_osc_on(RCC_HSE);
378  rcc_wait_for_osc_ready(RCC_HSE);
379 
380  /*
381  * Set prescalers for AHB, ADC, ABP1, ABP2.
382  * Do this before touching the PLL (TODO: why?).
383  */
384  rcc_set_hpre(clock->hpre);
385  rcc_set_ppre1(clock->ppre1);
386  rcc_set_ppre2(clock->ppre2);
387 
388  rcc_set_main_pll_hse(clock->pllm, clock->plln,
389  clock->pllp, clock->pllq);
390 
391  /* Enable PLL oscillator and wait for it to stabilize. */
392  rcc_osc_on(RCC_PLL);
393  rcc_wait_for_osc_ready(RCC_PLL);
394 
395  /* Configure flash settings. */
396  flash_set_ws(clock->flash_config);
397 
398  /* Select PLL as SYSCLK source. */
399  rcc_set_sysclk_source(RCC_CFGR_SW_PLL);
400 
401  /* Wait for PLL clock to be selected. */
402  rcc_wait_for_sysclk_status(RCC_PLL);
403 
404  /* Set the peripheral clock frequencies used. */
405  rcc_apb1_frequency = clock->apb1_frequency;
406  rcc_apb2_frequency = clock->apb2_frequency;
407 }
