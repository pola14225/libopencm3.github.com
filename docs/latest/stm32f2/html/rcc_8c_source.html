183 void rcc_wait_for_sysclk_status(osc_t osc)
184 {
185  switch (osc) {
186  case PLL:
187  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_PLL);
188  break;
189  case HSE:
190  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_HSE);
191  break;
192  case HSI:
193  while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_HSI);
194  break;
195  default:
196  /* Shouldn't be reached. */
197  break;
198  }

366 void rcc_clock_setup_hse_3v3(const clock_scale_t *clock)
367 {
368  /* Enable internal high-speed oscillator. */
369  rcc_osc_on(HSI);
370  rcc_wait_for_osc_ready(HSI);
371 
372  /* Select HSI as SYSCLK source. */
373  rcc_set_sysclk_source(RCC_CFGR_SW_HSI);
374 
375  /* Enable external high-speed oscillator 8MHz. */
376  rcc_osc_on(HSE);
377  rcc_wait_for_osc_ready(HSE);
378 
379  /*
380  * Set prescalers for AHB, ADC, ABP1, ABP2.
381  * Do this before touching the PLL (TODO: why?).
382  */
383  rcc_set_hpre(clock->hpre);
384  rcc_set_ppre1(clock->ppre1);
385  rcc_set_ppre2(clock->ppre2);
386 
387  rcc_set_main_pll_hse(clock->pllm, clock->plln,
388  clock->pllp, clock->pllq);
389 
390  /* Enable PLL oscillator and wait for it to stabilize. */
391  rcc_osc_on(PLL);
392  rcc_wait_for_osc_ready(PLL);
393 
394  /* Configure flash settings. */
395  flash_set_ws(clock->flash_config);
396 
397  /* Select PLL as SYSCLK source. */
398  rcc_set_sysclk_source(RCC_CFGR_SW_PLL);
399 
400  /* Wait for PLL clock to be selected. */
401  rcc_wait_for_sysclk_status(PLL);
402 
403  /* Set the peripheral clock frequencies used. */
404  rcc_ppre1_frequency = clock->apb1_frequency;
405  rcc_ppre2_frequency = clock->apb2_frequency;
406 }
