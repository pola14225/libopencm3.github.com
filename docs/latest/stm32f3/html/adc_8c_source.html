1 /** @defgroup adc_file ADC
2  *
3  * @ingroup STM32F3xx
4  *
5  * @brief <b>libopencm3 STM32F3xx Analog to Digital Converters</b>
6  *
7  * @author @htmlonly &copy; @endhtmlonly 2012
8  * Ken Sarkies <ksarkies@internode.on.net>
9  *
10  * @date 30 August 2012
11  *
12  * This library supports the A/D Converter Control System in the STM32 series
13  * of ARM Cortex Microcontrollers by ST Microelectronics.
14  *
15  * Devices can have up to three A/D converters each with their own set of
16  * registers. However all the A/D converters share a common clock which is
17  * prescaled from the APB2 clock by default by a minimum factor of 2 to a
18  * maximum of 8. The ADC resolution can be set to 12, 10, 8 or 6 bits.
19  *
20  * Each A/D converter has up to 19 channels:
21  * @li On ADC1 the analog channels 16 is internally connected to the
22  * temperature sensor, channel 17 to V<sub>REFINT</sub>, and channel 18
23  * to V<sub>BATT</sub>.
24  * @li On ADC2 and ADC3 the analog channels 16 - 18 are not used.
25  *
26  * The conversions can occur as a one-off conversion whereby the process stops
27  * once conversion is complete. The conversions can also be continuous wherein
28  * a new conversion starts immediately the previous conversion has ended.
29  *
30  * Conversion can occur as a single channel conversion or a scan of a group of
31  * channels in either continuous or one-off mode. If more than one channel is
32  * converted in a scan group, DMA must be used to transfer the data as there is
33  * only one result register available. An interrupt can be set to occur at the
34  * end*
35  * of conversion, which occurs after all channels have been scanned.
36  *
37  * A discontinuous mode allows a subgroup of group of a channels to be
38  * converted in bursts of a given length.
39  *
40  * Injected conversions allow a second group of channels to be converted
41  * separately from the regular group. An interrupt can be set to occur at the
42  * end of conversion, which occurs after all channels have been scanned.
43  *
44  * @section adc_f3_api_ex Basic ADC Handling API.
45  *
46  * Example 1: Simple single channel conversion polled. Enable the peripheral
47  * clock and ADC, reset ADC and set the prescaler divider. Set multiple mode to
48  * independent.
49  *
50  * @code
51  * gpio_mode_setup(GPIOA, GPIO_MODE_ANALOG, GPIO_PUPD_NONE, GPIO1);
52  * rcc_peripheral_enable_clock(&RCC_APB2ENR, RCC_APB2ENR_ADC1EN);
53  * adc_set_clk_prescale(RCC_CFGR_ADCPRE_BY2);
54  * adc_disable_scan_mode(ADC1);
55  * adc_set_single_conversion_mode(ADC1);
56  * adc_set_sample_time(ADC1, ADC_CHANNEL0, ADC_SMPR1_SMP_1DOT5CYC);
57  * uint8_t channels[] = ADC_CHANNEL0;
58  * adc_set_regular_sequence(ADC1, 1, channels);
59  * adc_set_multi_mode(ADC_CCR_MULTI_INDEPENDENT);
60  * adc_power_on(ADC1);
61  * adc_start_conversion_regular(ADC1);
62  * while (! adc_eoc(ADC1));
63  * reg16 = adc_read_regular(ADC1);
64  * @endcode
65  *
66  * LGPL License Terms @ref lgpl_license
67  */

151 /** @brief ADC Enable Discontinuous Mode for Regular Conversions
152  *
153  * In this mode the ADC converts, on each trigger, a subgroup of up to 8 of the
154  * defined regular channel group. The subgroup is defined by the number of
155  * consecutive channels to be converted. After a subgroup has been converted
156  * the next trigger will start conversion of the immediately following subgroup
157  * of the same length or until the whole group has all been converted. When the
158  * whole group has been converted, the next trigger will restart conversion of
159  * the subgroup at the beginning of the whole group.
160  *
161  * @param[in] adc Unsigned int32. ADC block register address base @ref

165

169  return;

171  ADC_CFGR1(adc) |= ADC_CFGR1_DISCEN;
172  ADC_CFGR1(adc) |= ((length-1) << ADC_CFGR1_DISCNUM_SHIFT);

493 /** @brief ADC Set an Injected Channel Conversion Sequence
494  *
495  * Defines a sequence of channels to be converted as an injected group with a
496  * length from 1 to 4 channels. If this is called during conversion, the current
497  * conversion is reset and conversion begins again with the newly defined group.
498  *
499  * @param[in] adc Unsigned int32. ADC block register address base
500  * @ref adc_reg_base
501  * @param[in] length Unsigned int8. Number of channels in the group.
502  * @param[in] channel Unsigned int8[]. Set of channels in sequence, integers
503  * 0..18
504  */
505 
506 void adc_set_injected_sequence(uint32_t adc, uint8_t length, uint8_t channel[])
507 {
508  uint32_t reg32 = 0;
509  uint8_t i = 0;
510 
511  /* Maximum sequence length is 4 channels. Minimum sequence is 1.*/
512  if ((length - 1) > 3) {
513  return;
514  }
515 
516  for (i = 0; i < length; i++) {
517  reg32 |= ADC_JSQR_JSQ_VAL(4 - i, channel[length - i - 1]);
518  }
519 
520  reg32 |= ADC_JSQR_JL_VAL(length);
521 
522  ADC_JSQR(adc) = reg32;
523 }
