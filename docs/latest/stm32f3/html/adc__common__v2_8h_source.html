38 /* ----- ADC registers -----------------------------------------------------*/
39 /* ADC interrupt and status register */
40 #define ADC_ISR(adc) MMIO32((adc) + 0x00)
41 /* Interrupt Enable Register */
42 #define ADC_IER(adc) MMIO32((adc) + 0x04)
43 /* Control Register */
44 #define ADC_CR(adc) MMIO32((adc) + 0x08)
45 /* Configuration Register 1 */
46 #define ADC_CFGR1(adc) MMIO32((adc) + 0x0C)
47 /* Configuration Register 2 */
48 #define ADC_CFGR2(adc) MMIO32((adc) + 0x10)
49 /* Sample Time Register 1 */
50 #define ADC_SMPR1(adc) MMIO32((adc) + 0x14)
51 /* Watchdog Threshold Register 1*/
52 #define ADC_TR1(adc) MMIO32((adc) + 0x20)
53 /* Regular Data Register */
54 #define ADC_DR(adc) MMIO32((adc) + 0x40)
55 /* CALFACT for all but f0 :(*/
56 
57 /* ADC common (shared) registers */
58 #define ADC_CCR(adc) MMIO32((adc) + 0x300 + 0x8)
