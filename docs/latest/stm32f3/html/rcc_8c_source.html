386 void rcc_clock_setup_hsi(const clock_scale_t *clock)
387 {
388  /* Enable internal high-speed oscillator. */
389  rcc_osc_on(HSI);
390  rcc_wait_for_osc_ready(HSI);
391  /* Select HSI as SYSCLK source. */
392  rcc_set_sysclk_source(RCC_CFGR_SW_HSI); /* XXX: se cayo */
393  rcc_wait_for_sysclk_status(HSI);
394 
395  rcc_osc_off(PLL);
396  rcc_wait_for_osc_not_ready(PLL);
397  rcc_set_pll_source(clock->pllsrc);
398  rcc_set_main_pll_hsi(clock->pll);
399  /* Enable PLL oscillator and wait for it to stabilize. */
400  rcc_osc_on(PLL);
401  rcc_wait_for_osc_ready(PLL);
402  /*
403  * Set prescalers for AHB, ADC, ABP1, ABP2.
404  * Do this before touching the PLL (TODO: why?).
405  */
406  rcc_set_hpre(clock->hpre);
407  rcc_set_ppre2(clock->ppre2);
408  rcc_set_ppre1(clock->ppre1);
409  /* Configure flash settings. */
410  flash_set_ws(clock->flash_config);
411  /* Select PLL as SYSCLK source. */
412  rcc_set_sysclk_source(RCC_CFGR_SW_PLL); /* XXX: se cayo */
413  /* Wait for PLL clock to be selected. */
414  rcc_wait_for_sysclk_status(PLL);
415 
416  /* Set the peripheral clock frequencies used. */
417  rcc_apb1_frequency = clock->apb1_frequency;
418  rcc_apb2_frequency = clock->apb2_frequency;
419 }
