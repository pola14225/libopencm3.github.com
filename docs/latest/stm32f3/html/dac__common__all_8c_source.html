1 /** @addtogroup dac_file
2 
3 @author @htmlonly &copy; @endhtmlonly 2012 Ken Sarkies ksarkies@internode.on.net
4 
5 This library supports the Digital to Analog Conversion System in the
6 STM32F series of ARM Cortex Microcontrollers by ST Microelectronics.
7 
8 The DAC is present only in a limited set of devices, notably some
9 of the connection line, high density and XL devices.
10 
11 Two DAC channels are available, however unlike the ADC channels these
12 are separate DAC devices controlled by the same register block.
13 
14 The DAC is on APB1. Its clock must be enabled in RCC and depending on
15 specific family, the GPIO
16 ports set to alternate function output before it can be used.
17 On most families, the GPIO pins should be configured to Analog IN to
18 avoid parasitic consumption.
19 The digital output driver is disabled so the output driver mode
20 (push-pull/open drain) is arbitrary.
21 
22 The DAC has a holding (buffer) register and an output register from
23 which the analog output is derived. The holding register must be
24 loaded first. If triggering is enabled the output register is loaded
25 from the holding register after a trigger occurs. If triggering is
26 not enabled the holding register contents are transferred directly
27 to the output register.
28 
29 @note To avoid nonlinearities, do not allow outputs to range close
30 to zero or V_analog.
31 
32 @section dac_api_dual Dual Channel Conversion
33 
34 There are dual modes in which both DACs are used to output data
35 simultaneously or independently on both channels. The data must be
36 presented according to the formats described in the datasheets. A
37 convenience function @ref dac_load_data_buffer_dual is provided
38 for software controlled use.
39 
40 A variety of modes are available depending on whether independent
41 or simultaneous output is desired, and whether waveforms are to be
42 superimposed. Refer to the datasheets.
43 
44 If DMA is used, only enable it for one of the channels. The DMA
45 requests will then serve data in dual format to the data register
46 dedicated to dual mode. The data will then be split and loaded to the
47 appropriate DAC following the next trigger. There are three registers
48 available, one for each of the formats: 12 bit right-aligned, 12 bit
49 left-aligned and 8 bit right-aligned. The desired format is determined
50 by specifying the appropriate register to the DMA controller.
51 
52 @section dac_api_basic_ex Basic DAC handling API.
53 
54 Set the DAC's GPIO port to Analog IN. Enable the
55 DAC clock. Enable the DAC, set a trigger source and load the buffer
56 with the first value. After the DAC is triggered, load the buffer with
57 the next value. This example uses software triggering and added noise.
58 The trigger and further buffer load calls are made when data is to be
59 sent out.
60 
61 @code
62  gpio_set_mode(GPIOA, GPIO_MODE_OUTPUT_50_MHZ,
63  GPIO_CNF_OUTPUT_ALTFN_PUSHPULL, GPIO4);
64  rcc_peripheral_enable_clock(&RCC_APB1ENR, RCC_APB1ENR_DACEN);
65  dac_disable(CHANNEL_1);
66  dac_set_waveform_characteristics(DAC_CR_MAMP1_8);
67  dac_set_waveform_generation(DAC_CR_WAVE1_NOISE);
68  dac_enable(CHANNEL_1);
69  dac_set_trigger_source(DAC_CR_TSEL1_SW);
70  dac_load_data_buffer_single(0, RIGHT12, CHANNEL_1);
71  ....
72  dac_software_trigger(CHANNEL_1);
73  dac_load_data_buffer_single(value, RIGHT12, CHANNEL_1);
74 @endcode
75 
76 @section dac_api_dma_ex Simultaneous Dual DAC with DMA.
77 
78 This example in part sets up the DAC channel 1 DMA (DMA2 channel 3) to read
79 16 bit data from memory into the right-aligned 8 bit dual register DAC_DHR8RD.
80 Both DAC channels are enabled, and both triggers are set to the same timer
81 2 input as required for simultaneous operation. DMA is enabled for DAC channel
82 1 only to ensure that only one DMA request is generated.
83 
84 @code
85  dma_set_memory_size(DMA2,DMA_CHANNEL3,DMA_CCR_MSIZE_16BIT);
86  dma_set_peripheral_size(DMA2,DMA_CHANNEL3,DMA_CCR_PSIZE_16BIT);
87  dma_set_read_from_memory(DMA2,DMA_CHANNEL3);
88  dma_set_peripheral_address(DMA2,DMA_CHANNEL3,(uint32_t) &DAC_DHR8RD);
89  dma_enable_channel(DMA2,DMA_CHANNEL3);
90  ...
91  dac_trigger_enable(CHANNEL_D);
92  dac_set_trigger_source(DAC_CR_TSEL1_T2 | DAC_CR_TSEL2_T2);
93  dac_dma_enable(CHANNEL_1);
94  dac_enable(CHANNEL_D);
95 @endcode

406 Loads the appropriate digital to analog converter data register with 12 or 8 bit

409 @li right-aligned 12 bit data in bits 0-11
410 @li left aligned 12 bit data in bits 4-15
411

418  data_channel dac_channel)

424  break;
425  case RIGHT12:
426  DAC_DHR12R1 = dac_data;
427  break;
428  case LEFT12:
429  DAC_DHR12L1 = dac_data;
430  break;
431  }
432  } else if (dac_channel == CHANNEL_2) {
433  switch (dac_data_format) {
434  case RIGHT8:
435  DAC_DHR8R2 = dac_data;
436  break;
437  case RIGHT12:

448 /** @brief Load DAC Dual Data Register.
449 
450 Loads the appropriate digital to analog converter dual data register with 12 or
451 8 bit data to be converted for both channels. This allows high bandwidth
452 simultaneous or independent analog output. The data in both channels are aligned
453 identically.
454 
455 @param[in] dac_data1 uint16_t for channel 1 with appropriate alignment.
456 @param[in] dac_data2 uint16_t for channel 2 with appropriate alignment.
457 @param[in] dac_data_format enum ::data_align. Right or left aligned, and 8 or
458 12 bit.
459 */
460 
461 void dac_load_data_buffer_dual(uint16_t dac_data1, uint16_t dac_data2,
462  data_align dac_data_format)
463 {
464  switch (dac_data_format) {
465  case RIGHT8:
466  DAC_DHR8RD = ((dac_data1 & MASK8) | ((dac_data2 & MASK8) << 8));
467  break;
468  case RIGHT12:
469  DAC_DHR12RD = ((dac_data1 & MASK12) |
470  ((dac_data2 & MASK12) << 16));
471  break;
472  case LEFT12:
473  DAC_DHR12LD = ((dac_data1 & MASK12) |
474  ((dac_data2 & MASK12) << 16));
475  break;
476  }
477 }
