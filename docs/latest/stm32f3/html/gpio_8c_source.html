36 void gpio_mode_setup(uint32_t gpioport, uint8_t mode, uint8_t pull_up_down,
37  uint16_t gpios)
38 {
39  uint16_t i;
40  uint32_t moder, pupd;
41 
42  /*
43  * We want to set the config only for the pins mentioned in gpios,
44  * but keeping the others, so read out the actual config first.
45  */
46  moder = GPIO_MODER(gpioport);
47  pupd = GPIO_PUPDR(gpioport);
48 
49  for (i = 0; i < 16; i++) {
50  if (!((1 << i) & gpios)) {
51  continue;
52  }
53 
54  moder &= ~GPIO_MODE_MASK(i);
55  moder |= GPIO_MODE(i, mode);
56  pupd &= ~GPIO_PUPD_MASK(i);
57  pupd |= GPIO_PUPD(i, pull_up_down);
58  }
59 
60  /* Set mode and pull up/down control registers. */
61  GPIO_MODER(gpioport) = moder;
62  GPIO_PUPDR(gpioport) = pupd;
63 }

148 void gpio_port_config_lock(uint32_t gpioport, uint16_t gpios)
149 {
150  uint32_t reg32;
151 
152  /* Special "Lock Key Writing Sequence", see datasheet. */
153  GPIO_LCKR(gpioport) = GPIO_LCKK | gpios; /* Set LCKK. */
154  GPIO_LCKR(gpioport) = ~GPIO_LCKK & gpios; /* Clear LCKK. */
155  GPIO_LCKR(gpioport) = GPIO_LCKK | gpios; /* Set LCKK. */
156  reg32 = GPIO_LCKR(gpioport); /* Read LCKK. */
157  reg32 = GPIO_LCKR(gpioport); /* Read LCKK again. */
158 
159  /*
160  * Tell the compiler the variable is actually used.
161  * It will get optimized out anyways.
162  */
163  reg32 = reg32;
164 
165  /* If (reg32 & GPIO_LCKK) is true, the lock is now active. */
166 }
