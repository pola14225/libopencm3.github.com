37 /** @brief I2C Reset.
38  *
39  * The I2C peripheral and all its associated configuration registers are placed
40  * in the reset condition. The reset is effected via the RCC peripheral reset
41  * system.
42  *
43  * @param[in] i2c Unsigned int32. I2C peripheral identifier @ref i2c_reg_base.
44  */
45 
46 void i2c_reset(uint32_t i2c)
47 {
48  switch (i2c) {
49  case I2C1:
50  rcc_peripheral_reset(&RCC_APB1RSTR, RCC_APB1RSTR_I2C1RST);
51  rcc_peripheral_clear_reset(&RCC_APB1RSTR, RCC_APB1RSTR_I2C1RST);
52  break;
53  case I2C2:
54  rcc_peripheral_reset(&RCC_APB1RSTR, RCC_APB1RSTR_I2C2RST);
55  rcc_peripheral_clear_reset(&RCC_APB1RSTR, RCC_APB1RSTR_I2C2RST);
56  break;
57  }
58 }

411 void write_i2c(uint32_t i2c, uint8_t i2c_addr, uint8_t reg, uint8_t size,
412  uint8_t *data)
413 {
414  int wait;
415  int i;
416  while (i2c_busy(i2c) == 1);
417  while (i2c_is_start(i2c) == 1);
418  /*Setting transfer properties*/
419  i2c_set_bytes_to_transfer(i2c, size + 1);
420  i2c_set_7bit_address(i2c, (i2c_addr & 0x7F));
421  i2c_set_write_transfer_dir(i2c);
422  i2c_enable_autoend(i2c);
423  /*start transfer*/
424  i2c_send_start(i2c);
425 
426  wait = true;
427  while (wait) {
428  if (i2c_transmit_int_status(i2c)) {
429  wait = false;
430  }
431  while (i2c_nack(i2c));
432  }
433 
434  i2c_send_data(i2c, reg);
435  for (i = 0; i < size; i++) {
436  wait = true;
437  while (wait) {
438  if (i2c_transmit_int_status(i2c)) {
439  wait = false;
440  }
441  while (i2c_nack(i2c));
442  }
443  i2c_send_data(i2c, data[i]);
444  }
445 }

447 void read_i2c(uint32_t i2c, uint8_t i2c_addr, uint8_t reg, uint8_t size,
448  uint8_t *data)
449 {
450  int wait;
451  int i;
452  while (i2c_busy(i2c) == 1);
453  while (i2c_is_start(i2c) == 1);
454  /*Setting transfer properties*/

456  i2c_set_7bit_address(i2c, i2c_addr);
457  i2c_set_write_transfer_dir(i2c);
458  i2c_disable_autoend(i2c);
459  /*start transfer*/
460  i2c_send_start(i2c);
461 
462  wait = true;
463  while (wait) {
464  if (i2c_transmit_int_status(i2c)) {
465  wait = false;
466  }
467  while (i2c_nack(i2c)); /* Some error */
468  }
469  i2c_send_data(i2c, reg);
470 
471  while (i2c_is_start(i2c) == 1);
472  /*Setting transfer properties*/
473  i2c_set_bytes_to_transfer(i2c, size);
474  i2c_set_7bit_address(i2c, i2c_addr);
475  i2c_set_read_transfer_dir(i2c);
476  i2c_enable_autoend(i2c);
477  /*start transfer*/
