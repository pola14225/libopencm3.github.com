35 {
36  USART_MR(usart) = (USART_MR(usart) & ~USART_MR_NBSTOP_MASK) |

45 void usart_set_mode(uint32_t usart, enum usart_mode mode)
46 {
47  USART_CR(usart) =
48  (mode & USART_MODE_RX) ? USART_CR_RXEN : USART_CR_RXDIS;
49  USART_CR(usart) = (mode & USART_MODE_TX) ? USART_CR_TXEN
50  : USART_CR_TXDIS;
51 }

53 void usart_set_flow_control(uint32_t usart, enum usart_flowcontrol fc)
54 {
55  USART_MR(usart) = (USART_MR(usart) & ~USART_MR_MODE_MASK) |
56  (fc ? USART_MR_MODE_HW_HANDSHAKING : 0);
