31 /** @brief USART enable data inversion
32 
33  Logical data from the data register are send/received in negative/inverse logic.
34  (1=L, 0=H). The parity bit is also inverted.
35 
36  @note This bit field can only be written when the USART is disabled.
37 
38  @param[in] usart USART block register address base @ref usart_reg_base
39  */

118 /** @brief USART Enable Half-duplex
119 
120  - The TX and RX lines are internally connected.
121  - The RX pin is no longer used
122  - The TX pin is always released when no data is transmitted. Thus,
123  it acts as a standard I/O in idle or in reception. It means
124  that the I/O must be configured so that TX is configured as
125  alternate function open-drain with an external pull-up.
126 
127  Apart from this, the communication protocol is similar to normal USART mode.
128  Any conflicts on the line must be managed by software
129 
130  @This bit field can only be written when the USART is disabled.
131 
132  @param[in] usart USART block register address base @ref usart_reg_base
133  */
134 void usart_enable_halfduplex(uint32_t usart)
135 {
136  USART_CR3(usart) |= USART_CR3_HDSEL;
137 }

152 /** @brief USART Set receiver timeout value
153 
154  Sets the receive timeout value in terms of number of bit duration.
155  The RTOF @ref usart_isr_rtof is set if, after the last received character,
156  no new start bit is detected for more than the receive timeout value.
157 
158  @note The timeout value can also be written when USART is enabled.
159  If the new value is lower/equals the internal hardware counter,
160  the RTOF flag will be set.
161 
162  @param[in] usart USART block register address base @ref usart_reg_base
163  @param[in] value The receive timeout value in terms of number of bit duration.
164  */
165 void usart_set_rx_timeout_value(uint32_t usart, uint32_t value)
166 {
167  uint32_t reg;
168  reg = USART_RTOR(usart) & ~USART_RTOR_RTO_MASK;
169  reg |= (USART_RTOR_RTO_VAL(value) & USART_RTOR_RTO_MASK);
170  USART_RTOR(usart) = reg;
171 }
