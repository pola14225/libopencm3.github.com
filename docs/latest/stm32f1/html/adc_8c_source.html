5 @brief <b>libopencm3 STM32F1xx Analog to Digital Converters</b>
6 
7 @version 1.0.0
8 
9 @author @htmlonly &copy; @endhtmlonly 2009
10 Edward Cheeseman <evbuilder@users.sourceforge.net>
11 @author @htmlonly &copy; @endhtmlonly 2012
12 Ken Sarkies <ksarkies@internode.on.net>
13 
14 @date 18 August 2012
15 
16 This library supports the A/D Converter Control System in the STM32F1xx series
17 of ARM Cortex Microcontrollers by ST Microelectronics.
18 
19 Devices can have up to three A/D converters each with their own set of
20 registers. However all the A/D converters share a common clock which is
21 prescaled from the APB2 clock by default by a minimum factor of 2 to a maximum
22 of 8.
23 
24 Each A/D converter has up to 18 channels:
25 @li On ADC1 the analog channels 16 and 17 are internally connected to the
26 temperature
27 sensor and V<sub>REFINT</sub>, respectively.
28 @li On ADC2 the analog channels 16 and 17 are internally connected to
29 V<sub>SS</sub>.
30 @li On ADC3 the analog channels 9, 14, 15, 16 and 17 are internally connected
31 to V<sub>SS</sub>.
32 
33 The conversions can occur as a one-off conversion whereby the process stops
34 once conversion is complete. The conversions can also be continuous wherein a
35 new conversion starts immediately the previous conversion has ended.
36 
37 Conversion can occur as a single channel conversion or a scan of a group of
38 channels in either continuous or one-off mode. If more than one channel is
39 converted in a scan group, DMA must be used to transfer the data as there is
40 only one result register available. An interrupt can be set to occur at the end
41 of conversion, which occurs after all channels have been scanned.
42 
43 A discontinuous mode allows a subgroup of group of a channels to be converted
44 in bursts of a given length.
45 
46 Injected conversions allow a second group of channels to be converted
47 separately from the regular group. An interrupt can be set to occur at the end
48 of conversion, which occurs after all channels have been scanned.
49 
50 @section adc_api_ex Basic ADC Handling API.
51 
52 Example 1: Simple single channel conversion polled. Enable the peripheral clock
53 and ADC, reset ADC and set the prescaler divider. Set dual mode to independent
54 (default). Enable triggering for a software trigger.
55 
56 @code
57  rcc_peripheral_enable_clock(&RCC_APB2ENR, RCC_APB2ENR_ADC1EN);
58  adc_off(ADC1);
59  rcc_peripheral_reset(&RCC_APB2RSTR, RCC_APB2RSTR_ADC1RST);
60  rcc_peripheral_clear_reset(&RCC_APB2RSTR, RCC_APB2RSTR_ADC1RST);
61  rcc_set_adcpre(RCC_CFGR_ADCPRE_PCLK2_DIV2);
62  adc_set_dual_mode(ADC_CR1_DUALMOD_IND);
63  adc_disable_scan_mode(ADC1);
64  adc_set_single_conversion_mode(ADC1);
65  adc_set_sample_time(ADC1, ADC_CHANNEL0, ADC_SMPR1_SMP_1DOT5CYC);
66  adc_set_single_channel(ADC1, ADC_CHANNEL0);
67  adc_enable_trigger(ADC1, ADC_CR2_EXTSEL_SWSTART);
68  adc_power_on(ADC1);
69  adc_reset_calibration(ADC1);
70  adc_calibration(ADC1);
71  adc_start_conversion_regular(ADC1);
72  while (! adc_eoc(ADC1));
73  reg16 = adc_read_regular(ADC1);
74 @endcode

154 /** @brief ADC Set Dual A/D Mode
155 
156 The dual mode uses ADC1 as master and ADC2 in a slave arrangement. This setting
157 is applied to ADC1 only. Start of conversion when triggered can cause
158 simultaneous conversion with ADC2, or alternate conversion. Regular and
159 injected conversions can be configured, each one being separately simultaneous
160 or alternate.
161 
162 Fast interleaved mode starts ADC1 immediately on trigger, and ADC2 seven clock
163 cycles later.
164 
165 Slow interleaved mode starts ADC1 immediately on trigger, and ADC2 fourteen
166 clock cycles later, followed by ADC1 fourteen cycles later again. This can only
167 be used on a single channel.
168 
169 Alternate trigger mode must occur on an injected channel group, and alternates
170 between the ADCs on each trigger.
171 
172 Note that sampling must not overlap between ADCs on the same channel.
173 
174 Dual A/D converter modes possible:
175 
176 @li IND: Independent mode.
177 @li CRSISM: Combined regular simultaneous + injected simultaneous mode.
178 @li CRSATM: Combined regular simultaneous + alternate trigger mode.
179 @li CISFIM: Combined injected simultaneous + fast interleaved mode.
180 @li CISSIM: Combined injected simultaneous + slow interleaved mode.
181 @li ISM: Injected simultaneous mode only.
182 @li RSM: Regular simultaneous mode only.
183 @li FIM: Fast interleaved mode only.
184 @li SIM: Slow interleaved mode only.
185 @li ATM: Alternate trigger mode only.
186 
187 @param[in] mode Unsigned int32. Dual mode selection from @ref adc_cr1_dualmod
188 */
189 
190 void adc_set_dual_mode(uint32_t mode)
191 {
192  ADC1_CR1 |= mode;
193 }

391 /** @brief ADC Set the Sample Time for a Single Channel
392 
393 The sampling time can be selected in ADC clock cycles from 1.5 to 239.5.
394 
395 @param[in] adc Unsigned int32. ADC block register address base @ref
396 adc_reg_base.
397 @param[in] channel Unsigned int8. ADC Channel integer 0..18 or from @ref
398 adc_channel.
399 @param[in] time Unsigned int8. Sampling time selection from @ref adc_sample_rg.
400  * * NOTE Common with f2 and f37x and f4
401 */
402 
403 void adc_set_sample_time(uint32_t adc, uint8_t channel, uint8_t time)
404 {
405  uint32_t reg32;
406 
407  if (channel < 10) {
408  reg32 = ADC_SMPR2(adc);
409  reg32 &= ~(0x7 << (channel * 3));
410  reg32 |= (time << (channel * 3));
411  ADC_SMPR2(adc) = reg32;
412  } else {
413  reg32 = ADC_SMPR1(adc);
414  reg32 &= ~(0x7 << ((channel - 10) * 3));
415  reg32 |= (time << ((channel - 10) * 3));
416  ADC_SMPR1(adc) = reg32;
417  }
418 }
419 
420 /*---------------------------------------------------------------------------*/
421 /** @brief ADC Set the Sample Time for All Channels
422 
423 The sampling time can be selected in ADC clock cycles from 1.5 to 239.5, same
424 for all channels.
425 
426 @param[in] adc Unsigned int32. ADC block register address base @ref
427 adc_reg_base.
428 @param[in] time Unsigned int8. Sampling time selection from @ref adc_sample_rg.
429  * * NOTE Common with f2 and f37x and f4
430 */
431 
432 void adc_set_sample_time_on_all_channels(uint32_t adc, uint8_t time)
433 {
434  uint8_t i;
435  uint32_t reg32 = 0;
436 
437  for (i = 0; i <= 9; i++) {
438  reg32 |= (time << (i * 3));
439  }
440  ADC_SMPR2(adc) = reg32;
441 
442  for (i = 10; i <= 17; i++) {
443  reg32 |= (time << ((i - 10) * 3));
444  }
445  ADC_SMPR1(adc) = reg32;
446 }
