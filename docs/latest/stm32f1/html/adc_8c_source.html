5 @brief <b>libopencm3 STM32F1xx Analog to Digital Converters</b>
6

8 
9 @author @htmlonly &copy; @endhtmlonly 2009
10 Edward Cheeseman <evbuilder@users.sourceforge.net>
11 @author @htmlonly &copy; @endhtmlonly 2012
12 Ken Sarkies <ksarkies@internode.on.net>
13 
14 @date 18 August 2012
15 
16 This library supports the A/D Converter Control System in the STM32F1xx series
17 of ARM Cortex Microcontrollers by ST Microelectronics.
18 
19 Devices can have up to three A/D converters each with their own set of
20 registers. However all the A/D converters share a common clock which is
21 prescaled from the APB2 clock by default by a minimum factor of 2 to a maximum
22 of 8.
23 
24 Each A/D converter has up to 18 channels:
25 @li On ADC1 the analog channels 16 and 17 are internally connected to the
26 temperature
27 sensor and V<sub>REFINT</sub>, respectively.
28 @li On ADC2 the analog channels 16 and 17 are internally connected to
29 V<sub>SS</sub>.
30 @li On ADC3 the analog channels 9, 14, 15, 16 and 17 are internally connected
31 to V<sub>SS</sub>.
32 
33 The conversions can occur as a one-off conversion whereby the process stops
34 once conversion is complete. The conversions can also be continuous wherein a
35 new conversion starts immediately the previous conversion has ended.
36 
37 Conversion can occur as a single channel conversion or a scan of a group of
38 channels in either continuous or one-off mode. If more than one channel is
39 converted in a scan group, DMA must be used to transfer the data as there is
40 only one result register available. An interrupt can be set to occur at the end
41 of conversion, which occurs after all channels have been scanned.
42 
43 A discontinuous mode allows a subgroup of group of a channels to be converted
44 in bursts of a given length.
45 
46 Injected conversions allow a second group of channels to be converted
47 separately from the regular group. An interrupt can be set to occur at the end
48 of conversion, which occurs after all channels have been scanned.
49 
50 @section adc_api_ex Basic ADC Handling API.
51 
52 Example 1: Simple single channel conversion polled. Enable the peripheral clock
53 and ADC, reset ADC and set the prescaler divider. Set dual mode to independent
54 (default). Enable triggering for a software trigger.
55 
56 @code
57  rcc_peripheral_enable_clock(&RCC_APB2ENR, RCC_APB2ENR_ADC1EN);
58  adc_off(ADC1);
59  rcc_peripheral_reset(&RCC_APB2RSTR, RCC_APB2RSTR_ADC1RST);
60  rcc_peripheral_clear_reset(&RCC_APB2RSTR, RCC_APB2RSTR_ADC1RST);
61  rcc_set_adcpre(RCC_CFGR_ADCPRE_PCLK2_DIV2);
62  adc_set_dual_mode(ADC_CR1_DUALMOD_IND);
63  adc_disable_scan_mode(ADC1);
64  adc_set_single_conversion_mode(ADC1);
65  adc_set_sample_time(ADC1, ADC_CHANNEL0, ADC_SMPR1_SMP_1DOT5CYC);
66  adc_set_single_channel(ADC1, ADC_CHANNEL0);
67  adc_enable_trigger(ADC1, ADC_CR2_EXTSEL_SWSTART);
68  adc_power_on(ADC1);
69  adc_reset_calibration(ADC1);
70  adc_calibration(ADC1);
71  adc_start_conversion_regular(ADC1);
72  while (! adc_eoc(ADC1));
73  reg16 = adc_read_regular(ADC1);
74 @endcode

153 /** @brief ADC Set Dual A/D Mode
154 
155 The dual mode uses ADC1 as master and ADC2 in a slave arrangement. This setting
156 is applied to ADC1 only. Start of conversion when triggered can cause
157 simultaneous conversion with ADC2, or alternate conversion. Regular and
158 injected conversions can be configured, each one being separately simultaneous
159 or alternate.
160 
161 Fast interleaved mode starts ADC1 immediately on trigger, and ADC2 seven clock
162 cycles later.
163 
164 Slow interleaved mode starts ADC1 immediately on trigger, and ADC2 fourteen
165 clock cycles later, followed by ADC1 fourteen cycles later again. This can only
166 be used on a single channel.
167 
168 Alternate trigger mode must occur on an injected channel group, and alternates
169 between the ADCs on each trigger.
170 
171 Note that sampling must not overlap between ADCs on the same channel.
172 
173 Dual A/D converter modes possible:
174 
175 @li IND: Independent mode.
176 @li CRSISM: Combined regular simultaneous + injected simultaneous mode.
177 @li CRSATM: Combined regular simultaneous + alternate trigger mode.
178 @li CISFIM: Combined injected simultaneous + fast interleaved mode.
179 @li CISSIM: Combined injected simultaneous + slow interleaved mode.
180 @li ISM: Injected simultaneous mode only.
181 @li RSM: Regular simultaneous mode only.
182 @li FIM: Fast interleaved mode only.
183 @li SIM: Slow interleaved mode only.
184 @li ATM: Alternate trigger mode only.
185 
186 @param[in] mode Unsigned int32. Dual mode selection from @ref adc_cr1_dualmod
187 */
188 
189 void adc_set_dual_mode(uint32_t mode)
190 {
191  ADC1_CR1 |= mode;
192 }

243 /** @brief ADC Read from an Injected Conversion Result Register
244 
245 The result read back from the selected injected result register (one of four)
246 is 12 bits, right or left aligned within the first 16 bits. The result can have
247 a negative value if the injected channel offset has been set @see
248 adc_set_injected_offset.
249 
250 @param[in] adc Unsigned int32. ADC block register address base @ref
251 adc_reg_base.
252 @param[in] reg Unsigned int8. Register number (1 ... 4).
253 @returns Unsigned int32 conversion result.
254 */
255 
256 uint32_t adc_read_injected(uint32_t adc, uint8_t reg)
257 {
258  switch (reg) {
259  case 1:
260  return ADC_JDR1(adc);
261  case 2:
262  return ADC_JDR2(adc);
263  case 3:
264  return ADC_JDR3(adc);
265  case 4:
266  return ADC_JDR4(adc);
267  }
268  return 0;
269 }

359 /** @brief ADC Enable Discontinuous Mode for Regular Conversions
360 
361 In this mode the ADC converts, on each trigger, a subgroup of up to 8 of the
362 defined regular channel group. The subgroup is defined by the number of
363 consecutive channels to be converted. After a subgroup has been converted
364 the next trigger will start conversion of the immediately following subgroup
365 of the same length or until the whole group has all been converted. When the
366 the whole group has been converted, the next trigger will restart conversion

368 
369 @param[in] adc Unsigned int32. ADC block register address base @ref
370 adc_reg_base.

374 
375 void adc_enable_discontinuous_mode_regular(uint32_t adc, uint8_t length)
376 {
377  if ((length-1) > 7) {
378  return;
379  }
380  ADC_CR1(adc) |= ADC_CR1_DISCEN;
381  ADC_CR1(adc) |= ((length-1) << ADC_CR1_DISCNUM_SHIFT);
382 }
