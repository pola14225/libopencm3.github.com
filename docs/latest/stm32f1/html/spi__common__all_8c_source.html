73 /** @brief SPI Reset.
74 
75 The SPI peripheral and all its associated configuration registers are placed in
76 the reset condition. The reset is effected via the RCC peripheral reset system.
77 
78 @param[in] spi_peripheral Unsigned int32. SPI peripheral identifier @ref
79 spi_reg_base.
80 */
81 
82 void spi_reset(uint32_t spi_peripheral)
83 {
84  switch (spi_peripheral) {
85  case SPI1:
86  rcc_peripheral_reset(&RCC_APB2RSTR, RCC_APB2RSTR_SPI1RST);
87  rcc_peripheral_clear_reset(&RCC_APB2RSTR, RCC_APB2RSTR_SPI1RST);
88  break;
89  case SPI2:
90  rcc_peripheral_reset(&RCC_APB1RSTR, RCC_APB1RSTR_SPI2RST);
91  rcc_peripheral_clear_reset(&RCC_APB1RSTR, RCC_APB1RSTR_SPI2RST);
92  break;
93 #if defined(STM32F1) || defined(STM32F2) || defined(STM32F3) || defined(STM32F4)
94  case SPI3:
95  rcc_peripheral_reset(&RCC_APB1RSTR, RCC_APB1RSTR_SPI3RST);
96  rcc_peripheral_clear_reset(&RCC_APB1RSTR, RCC_APB1RSTR_SPI3RST);
97  break;
98 #endif
99  }
100 }

137 /** @brief SPI Clean Disable.
138 
139 Disable the SPI peripheral according to the procedure in section 23.3.8 of the
140 reference manual. This prevents corruption of any ongoing transfers and
141 prevents the BSY flag from becoming unreliable.
142 
143 @param[in] spi Unsigned int32. SPI peripheral identifier @ref spi_reg_base.
144 @returns data Unsigned int16. 8 or 16 bit data from final read.
145 */
146 
147 uint16_t spi_clean_disable(uint32_t spi)
148 {
149  /* Wait to receive last data */
150  while (!(SPI_SR(spi) & SPI_SR_RXNE));
151 
152  uint16_t data = SPI_DR(spi);
153 
154  /* Wait to transmit last data */
155  while (!(SPI_SR(spi) & SPI_SR_TXE));
156 
157  /* Wait until not busy */
158  while (SPI_SR(spi) & SPI_SR_BSY);
159 
160  SPI_CR1(spi) &= ~SPI_CR1_SPE;
161 
162  return data;
163 }
