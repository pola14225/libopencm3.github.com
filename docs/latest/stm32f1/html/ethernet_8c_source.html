22 void eth_smi_write(uint8_t phy, uint8_t reg, uint16_t data)
23 {
24  /* Set PHY and register addresses for write access. */
25  ETH_MACMIIAR &= ~(ETH_MACMIIAR_MR | ETH_MACMIIAR_PA);
26  ETH_MACMIIAR |= (phy << 11) | (reg << 6) | ETH_MACMIIAR_MW;
27 
28  /* Set register value. */
29  ETH_MACMIIDR = data;
30 
31  /* Begin transaction. */
32  ETH_MACMIIAR |= ETH_MACMIIAR_MB;
33 
34  /* Wait for not busy. */
35  while (ETH_MACMIIAR & ETH_MACMIIAR_MB);
36 }
