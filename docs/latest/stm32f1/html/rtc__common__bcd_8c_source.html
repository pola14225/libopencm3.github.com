32 void rtc_set_prescaler(u32 sync, u32 async) {
33  /*
34  * Even if only one of the two fields needs to be changed,
35  * 2 separate write accesses must be performed to the RTC_PRER register.
36  */
37  RTC_PRER = (sync & RTC_PRER_PREDIV_S_MASK);
38  RTC_PRER |= (async << RTC_PRER_PREDIV_A_SHIFT);
39 }

79 /** @brief Sets the wakeup time auto-reload value
80 
81 */
82 void rtc_set_wakeup_time(u16 wkup_time, u8 rtc_cr_wucksel) {
83 // FTFM:
84 // The following sequence is required to configure or change the wakeup timer
85 // auto-reload value (WUT[15:0] in RTC_WUTR):
86 // 1. Clear WUTE in RTC_CR to disable the wakeup timer.
87  RTC_CR &= ~RTC_CR_WUTE;
88 // 2. Poll WUTWF until it is set in RTC_ISR to make sure the access to wakeup
89 // auto-reload counter and to WUCKSEL[2:0] bits is allowed. It takes around 2
90 // RTCCLK clock cycles (due to clock synchronization).
91  while (!((RTC_ISR) & (RTC_ISR_WUTWF))) { }
92 // 3. Program the wakeup auto-reload value WUT[15:0], and the wakeup clock
93 // selection (WUCKSEL[2:0] bits in RTC_CR).Set WUTE in RTC_CR to enable the
94 // timer again. The wakeup timer restarts down-counting.
95  RTC_WUTR = wkup_time;
96  RTC_CR |= (rtc_cr_wucksel << RTC_CR_WUCLKSEL_SHIFT);
97  RTC_CR |= RTC_CR_WUTE;
98 }
