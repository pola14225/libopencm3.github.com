43 void rtc_awake_from_off(enum rcc_osc clock_source)
44 {
45  uint32_t reg32;
46 
47  /* Enable power and backup interface clocks. */
48  RCC_APB1ENR |= (RCC_APB1ENR_PWREN | RCC_APB1ENR_BKPEN);
49 
50  /* Enable access to the backup registers and the RTC. */
51  PWR_CR |= PWR_CR_DBP;
52 
53  /*
54  * Reset the backup domain, clears everything RTC related.
55  * If not wanted use the rtc_awake_from_standby() function.
56  */
57  rcc_backupdomain_reset();
58 
59  switch (clock_source) {
60  case LSE:
61  /* Turn the LSE on and wait while it stabilises. */
62  RCC_BDCR |= RCC_BDCR_LSEON;
63  while ((reg32 = (RCC_BDCR & RCC_BDCR_LSERDY)) == 0);
64 
65  /* Choose LSE as the RTC clock source. */
66  RCC_BDCR &= ~((1 << 8) | (1 << 9));
67  RCC_BDCR |= (1 << 8);
68  break;
69  case LSI:
70  /* Turn the LSI on and wait while it stabilises. */
71  RCC_CSR |= RCC_CSR_LSION;
72  while ((reg32 = (RCC_CSR & RCC_CSR_LSIRDY)) == 0);
73 
74  /* Choose LSI as the RTC clock source. */
75  RCC_BDCR &= ~((1 << 8) | (1 << 9));
76  RCC_BDCR |= (1 << 9);
77  break;
78  case HSE:
79  /* Turn the HSE on and wait while it stabilises. */
80  RCC_CR |= RCC_CR_HSEON;
81  while ((reg32 = (RCC_CR & RCC_CR_HSERDY)) == 0);
82 
83  /* Choose HSE as the RTC clock source. */
84  RCC_BDCR &= ~((1 << 8) | (1 << 9));
85  RCC_BDCR |= (1 << 9) | (1 << 8);
86  break;
87  case PLL:
88  case PLL2:
89  case PLL3:
90  case HSI:
91  /* Unusable clock source, here to prevent warnings. */
92  /* Turn off clock sources to RTC. */
93  RCC_BDCR &= ~((1 << 8) | (1 << 9));
94  break;
95  }
96 
97  /* Enable the RTC. */
98  RCC_BDCR |= RCC_BDCR_RTCEN;
99 
100  /* Wait for the RSF bit in RTC_CRL to be set by hardware. */
101  RTC_CRL &= ~RTC_CRL_RSF;
102  while ((reg32 = (RTC_CRL & RTC_CRL_RSF)) == 0);
103 
104  /* Wait for the last write operation to finish. */
105  /* TODO: Necessary? */
106  while ((reg32 = (RTC_CRL & RTC_CRL_RTOFF)) == 0);
107 }

109 void rtc_enter_config_mode(void)
110 {
111  uint32_t reg32;
112 
113  /* Wait until the RTOFF bit is 1 (no RTC register writes ongoing). */
114  while ((reg32 = (RTC_CRL & RTC_CRL_RTOFF)) == 0);
115 
116  /* Enter configuration mode. */
117  RTC_CRL |= RTC_CRL_CNF;
118 }

121 {
122  uint32_t reg32;
123 
124  /* Exit configuration mode. */
125  RTC_CRL &= ~RTC_CRL_CNF;
126 
127  /* Wait until the RTOFF bit is 1 (our RTC register write finished). */
128  while ((reg32 = (RTC_CRL & RTC_CRL_RTOFF)) == 0);
129 }
