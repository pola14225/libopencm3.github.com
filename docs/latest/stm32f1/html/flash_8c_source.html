113 void flash_program_word(u32 address, u32 data)
114 {
115  /* Ensure that all flash operations are complete. */
116  flash_wait_for_last_operation();
117 
118  /* Enable writes to flash. */
119  FLASH_CR |= FLASH_CR_PG;
120 
121  /* Program the first half of the word. */
122  (*(volatile u16 *)address) = (u16)data;
123 
124  /* Wait for the write to complete. */
125  flash_wait_for_last_operation();
126 
127  /* Program the second half of the word. */
128  (*(volatile u16 *)(address + 2)) = data >> 16;
129 
130  /* Wait for the write to complete. */
131  flash_wait_for_last_operation();
132 
133  /* Disable writes to flash. */
134  FLASH_CR &= ~FLASH_CR_PG;
135 }

173 void flash_erase_option_bytes(void)
174 {
175  flash_wait_for_last_operation();
176 
177  if ((FLASH_CR & FLASH_CR_OPTWRE) == 0)
178  flash_unlock_option_bytes();
179 
180  FLASH_CR |= FLASH_CR_OPTER; /* Enable option byte erase. */

187 {
188  flash_wait_for_last_operation();
189 
190  if ((FLASH_CR & FLASH_CR_OPTWRE) == 0)
191  flash_unlock_option_bytes();
192 
193  FLASH_CR |= FLASH_CR_OPTPG; /* Enable option byte programming. */
