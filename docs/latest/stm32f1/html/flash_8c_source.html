116 void flash_program_word(uint32_t address, uint32_t data)
117 {
118  /* Ensure that all flash operations are complete. */
119  flash_wait_for_last_operation();
120 
121  /* Enable writes to flash. */
122  FLASH_CR |= FLASH_CR_PG;
123 
124  /* Program the first half of the word. */
125  MMIO16(address) = (uint16_t)data;
126 
127  /* Wait for the write to complete. */
128  flash_wait_for_last_operation();
129 
130  /* Enable writes to flash. */
131  FLASH_CR |= FLASH_CR_PG;
132 
133  /* Program the second half of the word. */
134  MMIO16(address + 2) = data >> 16;
135 
136  /* Wait for the write to complete. */
137  flash_wait_for_last_operation();
138 }

174 void flash_erase_option_bytes(void)
175 {
176  flash_wait_for_last_operation();
177 
178  if ((FLASH_CR & FLASH_CR_OPTWRE) == 0) {
179  flash_unlock_option_bytes();
180  }
181 
182  FLASH_CR |= FLASH_CR_OPTER; /* Enable option byte erase. */
183  FLASH_CR |= FLASH_CR_STRT;
184  flash_wait_for_last_operation();
185  FLASH_CR &= ~FLASH_CR_OPTER; /* Disable option byte erase. */
186 }

189 {
