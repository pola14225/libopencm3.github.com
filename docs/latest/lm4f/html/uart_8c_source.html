45 /** @defgroup uart_config UART configuration
46  * @ingroup uart_file
47  *
48  * \brief <b>Enabling and configuring the UART</b>
49  *
50  * Enabling the UART is a two step process. The GPIO on which the UART resides
51  * must be enabled, and the UART pins must be configured as alternate function,
52  * digital pins. Pins must also be muxed to the appropriate alternate function.
53  * This is done with the GPIO API.
54  *
55  * The second step involves enabling and the UART itself. The UART should be
56  * disabled while it is being configured.
57  * -# The UART clock must be enabled with @ref periph_clock_enable().
58  * -# The UART must be disabled with @ref uart_disable().
59  * -# The UART clock source should be chosen before setting the baudrate.
60  * -# Baudrate, data bits, stop bit length, and parity can be configured.
61  * -# If needed, enable CTS or RTS lines via the @ref uart_set_flow_control().
62  * -# The UART can now be enabled with @ref uart_enable().
63  *
64  * For example, to enable UART1 at 115200 8n1 with hardware flow control:
65  * @code{.c}
66  * // Enable the UART clock
67  * periph_clock_enable(RCC_UART1);
68  * // We need a brief delay before we can access UART config registers
69  * __asm__("nop"); __asm__("nop"); __asm__("nop");
70  * // Disable the UART while we mess with its settings
71  * uart_disable(UART1);
72  * // Configure the UART clock source as precision internal oscillator
73  * uart_clock_from_piosc();
74  * // Set communication parameters
75  * uart_set_baudrate(UART1, 115200);
76  * uart_set_databits(UART1, 8);
77  * uart_set_parity(UART1, UART_PARITY_NONE);
78  * uart_set_stopbits(UART1, 1);
79  * // Enable RTC and CTS lines
80  * uart_set_flow_control(UART1, UART_FLOWCTL_HARD_RTS_CTS);
81  * // Now that we're done messing with the settings, enable the UART
82  * uart_enable(UART1);
83  * @endcode
84  */
85 /**@{*/

108 /**
109  * \brief Set UART baudrate
110  *
111  * @param[in] uart UART block register address base @ref uart_reg_base
112  * @param[in] baud Baud rate in bits per second (bps).*
113  */
114 void uart_set_baudrate(u32 uart, u32 baud)
115 {
116  u32 clock;
117 
118  /* Are we running off the internal clock or system clock? */
119  if (UART_CC(uart) == UART_CC_CS_PIOSC)
120  clock = 16000000;
121  else
122  clock = rcc_get_system_clock_frequency();
123 
124  /* Find the baudrate divisor */
125  u32 div = (((clock * 8) / baud) + 1) / 2;
126 
127  /* Set the baudrate divisors */
128  UART_IBRD(uart) = div / 64;
129  UART_FBRD(uart) = div % 64;
130 }

132 /**
133  * \brief Set UART databits
134  *
135  * @param[in] uart UART block register address base @ref uart_reg_base
136  * @param[in] databits number of data bits per transmission.
137  */
138 void uart_set_databits(u32 uart, u8 databits)
139 {
140  u32 reg32, bits32;
141 
142  /* This has the same effect as using UART_LCRH_WLEN_5/6/7/8 directly */
143  bits32 = (databits - 5) << 5;
144 
145  /* TODO: What about 9 data bits? */
146 
147  reg32 = UART_LCRH(uart);
148  reg32 &= ~UART_LCRH_WLEN_MASK;
149  reg32 |= bits32;
150  UART_LCRH(uart) = reg32;
151 }

167 /**
168  * \brief Set UART parity
169  *
170  * @param[in] uart UART block register address base @ref uart_reg_base
171  * @param[in] bits the requested parity scheme.
172  */
173 void uart_set_parity(u32 uart, enum uart_parity parity)
174 {
175  u32 reg32;
176 
177  reg32 = UART_LCRH(uart);
178  reg32 |= UART_LCRH_PEN;
179  reg32 &= ~(UART_LCRH_SPS | UART_LCRH_EPS);
180 
181  switch (parity)
182  {
183  case UART_PARITY_NONE:
184  /* Once we disable parity the other bits are meaningless */
185  UART_LCRH(uart) &= ~UART_LCRH_PEN;
186  return;
187  case UART_PARITY_ODD:
188  break;
189  case UART_PARITY_EVEN:
190  reg32 |= UART_LCRH_EPS;
191  break;
192  case UART_PARITY_STICK_0:
193  reg32 |= (UART_LCRH_SPS | UART_LCRH_EPS);
194  break;
195  case UART_PARITY_STICK_1:
196  reg32 |= UART_LCRH_SPS;
197  break;
198  }
199 
200  UART_LCRH(uart) = reg32;
201 }

203 /**
204  * \brief Set the flow control scheme
205  *
206  * Set the flow control scheme by enabling or disabling RTS and CTS lines. This
207  * will only have effect if the given UART supports the RTS and CTS lines.
208  *
209  * @param[in] uart UART block register address base @ref uart_reg_base
210  * @param[in] flow The flow control scheme to use (none, RTS, CTS or both) \n
211  * UART_FLOWCTL_RTS -- enable the RTS line \n
212  * UART_FLOWCTL_CTS -- enable the CTS line \n
213  * UART_FLOWCTL_RTS_CTS -- enable both RTS and CTS lines
214  */
215 void uart_set_flow_control(u32 uart, enum uart_flowctl flow)
216 {
217  u32 reg32 = UART_CTL(uart);
218 
219  reg32 &= ~(UART_CTL_RTSEN | UART_CTL_CTSEN);
220 
221  if (flow == UART_FLOWCTL_RTS)
222  reg32 |= UART_CTL_RTSEN;
223  else if (flow == UART_FLOWCTL_CTS)
224  reg32 |= UART_CTL_CTSEN;
225  else if (flow == UART_FLOWCTL_RTS_CTS)
226  reg32 |= (UART_CTL_RTSEN | UART_CTL_CTSEN);
227 
228  UART_CTL(uart) = reg32;
229 }

349 /** @defgroup uart_irq UART Interrupt control
350  * @ingroup uart_file
351  *
352  * \brief <b>Configuring interrupts from the UART</b>
353  *
354  * To have an event generate an interrupt, its interrupt source must be
355  * unmasked. This can be achieved with @ref uart_enable_interrupts(). Interrupts
356  * which are no longer needed can be disabled through
357  * @ref uart_disable_interrupts().
358  *
359  * In order for the interrupt to generate an IRQ and a call to the interrupt
360  * service routine, the interrupt for the target UART must be routed through the
361  * NVIC with @ref nvic_enable_irq(). For this last step, the nvic.h header is
362  * needed:
363  * @code{.c}
364  * #include <libopencm3/lm4f/nvic.h>
365  * @endcode
366  *
367  * Enabling an interrupt is as simple as unmasking the desired interrupt, and
368  * routing the desired UART's interrupt through the NVIC.
369  * @code{.c}
370  * // Unmask receive interrupt
371  * uart_enable_rx_interrupt(UART0);
372  * // Make sure the interrupt is routed through the NVIC
373  * nvic_enable_irq(NVIC_UART0_IRQ);
374  * @endcode
375  *
376  * If a more than one interrupt is to be enabled at one time, the interrupts
377  * can be enabled by a single call to @ref uart_enable_interrupts().
378  * For example:
379  * @code{.c}
380  * // Unmask receive, CTS, and RI, interrupts
381  * uart_enable_interrupts(UART0, UART_INT_RX | UART_INT_RI | UART_INT_CTS);
382  * @endcode
383  *
384  * After interrupts are properly enabled and routed through the NVIC, when an
385  * event occurs, the appropriate IRQ flag is set by hardware, and execution
386  * jumps to the UART ISR. The ISR should query the IRQ flags to determine which
387  * event caused the interrupt. For this, use @ref uart_is_interrupt_source(),
388  * with the desired UART_INT flag. After one or more interrupt sources are
389  * serviced, the IRQ flags must be cleared by the ISR. This can be done with
390  * @ref uart_clear_interrupt_flag().
391  *
392  * A typical UART ISR may look like the following:
393  * @code{.c}
394  * void uart0_isr(void)
395  * {
396  * u32 serviced_irqs = 0;
397  *
398  * // Process individual IRQs
399  * if (uart_is_interrupt_source(UART0, UART_INT_RX)) {
400  * process_rx_event();
401  * serviced_irq |= UART_INT_RX;
402  * }
403  * if (uart_is_interrupt_source(UART0, UART_INT_CTS)) {
404  * process_cts_event();
405  * serviced_irq |= UART_INT_CTS;
406  * }
407  *
408  * // Clear the interupt flag for the processed IRQs
409  * uart_clear_interrupt_flag(UART0, serviced_irqs);
410  * }
411  * @endcode
412  */
413 /**@{*/
